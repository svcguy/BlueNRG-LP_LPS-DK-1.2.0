234 HAL_StatusTypeDef HAL_RNG_GenerateRandomNumber(RNG_HandleTypeDef *hrng, uint32_t *random16bit)
235 {
236  uint32_t tickstart;
237  HAL_StatusTypeDef status = HAL_OK;
238 
239  /* Process Locked */
240  __HAL_LOCK(hrng);
241 
242  /* Check RNG peripheral state */
243  if (hrng->State == HAL_RNG_STATE_READY)
244  {
245  /* Change RNG peripheral state */
246  hrng->State = HAL_RNG_STATE_BUSY;
247 
248  /* Get tick */
249  tickstart = HAL_GetTick();
250 
251  /* Check if data register contains valid random data */
252  while (__HAL_RNG_GET_FLAG(hrng, RNG_FLAG_DRDY) == RESET)
253  {
254  if ((HAL_GetTick() - tickstart) > RNG_TIMEOUT_VALUE)
255  {
256  hrng->State = HAL_RNG_STATE_READY;
257  hrng->ErrorCode |= HAL_RNG_ERROR_TIMEOUT;
258  /* Process Unlocked */
259  __HAL_UNLOCK(hrng);
260  return HAL_ERROR;
261  }
262  }
263 
264  /* Get a 16bit Random number */
265  hrng->RandomNumber = hrng->Instance->VAL;
266  *random16bit = hrng->RandomNumber;
267 
268  hrng->State = HAL_RNG_STATE_READY;
269  }
270  else
271  {
272  status = HAL_ERROR;
273  }
274 
275  /* Process Unlocked */
276  __HAL_UNLOCK(hrng);
277 
278  return status;
279 }
