200 uint32_t HAL_RCCEx_GetPeriphCLKFreq(uint32_t PeriphClk)
201 {
202  uint32_t frequency;
203 
204  /* Check the parameters */
205  assert_param(IS_RCC_PERIPHCLOCK(PeriphClk));
206 
207  /* No if clausule is covered */
208  frequency = 0;
209 
210  if(PeriphClk == RCC_PERIPHCLK_RF)
211  {
212 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
213  switch(__HAL_RCC_RF_RC64MPLL_GET_CONFIG())
214  {
215  case RCC_RF_RC64MPLL_DIV2:
216  frequency = 32000000;
217  break;
218  case RCC_RF_RC64MPLL_DIV4:
219  frequency = 16000000;
220  break;
221  }
222 #endif
223  }
224  else if (PeriphClk == RCC_PERIPHCLK_SMPS)
225  {
226  switch( __HAL_RCC_GET_SMPS_DIV())
227  {
228  case RCC_SMPSCLK_DIV2:
229  frequency = 16000000/2;
230  break;
231  case RCC_SMPSCLK_DIV4:
232  frequency = 16000000/4;
233  break;
234  }
235  }
236 
237 #if defined(SPI2)
238  if (PeriphClk == RCC_PERIPHCLK_SPI2_I2S)
239  {
240  if (__HAL_RCC_GET_SPI2I2S_SOURCE() == RCC_SPI2I2S_CLKSOURCE_16M)
241  {
242  frequency = 16000000;
243  }
244  else
245  {
246  frequency = 32000000;
247  }
248  }
249 #endif
250 
251  if (PeriphClk == RCC_PERIPHCLK_SPI3_I2S)
252  {
253  if (__HAL_RCC_GET_SPI3I2S_SOURCE() == RCC_SPI3I2S_CLKSOURCE_16M)
254  {
255  frequency = 16000000;
256  }
257  else
258  {
259  frequency = 32000000;
260  }
261  }
262 
263  return (frequency);
264 }

399 {
400  HAL_StatusTypeDef status;
401  uint32_t trimming = 15U;
402 
403  assert_param(IS_RCC_TRIMOSC(OscillatorType));
404 
405  if(OscillatorType == RCC_OSCILLATORTYPE_LSI)
406  {
407  if (LL_RCC_LSI_IsReady() == 0U)
408  {
409  status = HAL_ERROR;
410  }
411  else
412  {
413  /* LSI Trimming procedure */
414  /* 1/2 LSI is already enable and ready as initial condition */
415 
416  /* 3. Write LSITRIM to its max value b1111 */
417  LL_RCC_LSI_SetTrimming(trimming);
418 
419  /* 4. Enable LSI trimming */
420  LL_RCC_LSI_EnableTrimming();
421 
422  /* 5. Check if the LSI trimming value is OK by reading LSITRIMOK bit
423  till trimming value reach 0 */
424  const uint32_t temp_lsi1trimmed = LL_RCC_LSI_IsTrimmed();
425  const uint32_t temp_lsi2trimmed = LL_RCC_LSI_GetTrimming();
426  while((temp_lsi1trimmed != 1U) && (temp_lsi2trimmed != 0U))
427  {
428  /* 6/7 Decrease LSITRIM value by 1 */
429  trimming = trimming - 1U;
430  LL_RCC_LSI_SetTrimming(trimming);
431  }
432 
433  /* Check if trimming is successful */
434  if (LL_RCC_LSI_IsTrimmed() == 1U)
435  {
436  status = HAL_OK;
437  }
438  /* Otherwise, it means we reach minimal trimming value (0) wtihout trimming ok */
439  else
440  {
441  status = HAL_ERROR;
442  }
443  /* 8. Disable LSI trimming */
444  LL_RCC_LSI_DisableTrimming();
445  }
446  }
447  else
448  {
449  status = HAL_ERROR;
450  }
451  return status;
