208 static uint8_t IO_IRQ_Enabled(uint32_t wkSource, uint8_t *IRQA_enabled, uint8_t *IRQB_enabled)
209 {
210  uint32_t IO_A, IO_B;
211  uint8_t ret;
212 
213  ret = 0;
214  *IRQA_enabled = 0;
215  *IRQB_enabled = 0;
216 
217  /* Select GPIOA wakeup sources */
218  IO_A = WAKEUP_IOA_MASK(wkSource);
219 
220  /* Select GPIOB wakeup sources */
221  IO_B = WAKEUP_IOB_MASK(wkSource);
222 
223  /* Check if the GPIOB interrupt is enabled */
224  if (IO_B && (SYSCFG->IO_IER & (IO_B << 16))) {
225  *IRQB_enabled = 1;
226  ret = 1;
227  }
228 
229  /* Check if the GPIOA interrupt is enabled */
230  if (IO_A && (SYSCFG->IO_IER & IO_A)) {
231  *IRQA_enabled = 1;
232  ret = 1;
233  }
234 
235  return ret;
236 }
