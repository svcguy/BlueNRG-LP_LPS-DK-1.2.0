763  FlagStatus procEndFlag = __HAL_PKA_GET_FLAG(hpka, PKA_FLAG_PROCEND);
764 
765  /* Address error interrupt occurred */

767  {
768  hpka->ErrorCode |= HAL_PKA_ERROR_ADDRERR;
769

778

780  __HAL_PKA_CLEAR_FLAG(hpka, PKA_FLAG_RAMERR);
781  }

784  if (hpka->ErrorCode != HAL_PKA_ERROR_NONE)
785  {
786 #if (USE_HAL_PKA_REGISTER_CALLBACKS == 1)
787  hpka->ErrorCallback(hpka);

789  HAL_PKA_ErrorCallback(hpka);
790 #endif /* USE_HAL_PKA_REGISTER_CALLBACKS */
791  }
792 
793  /* End Of Operation interrupt occurred */
794  if ((__HAL_PKA_GET_IT_SOURCE(hpka, PKA_IT_PROCEND) == SET) && (procEndFlag == SET))
795  {
796  /* Clear PROCEND flag */
797  __HAL_PKA_CLEAR_FLAG(hpka, PKA_FLAG_PROCEND);
798 
799  /* Set the state to ready */
800  hpka->State = HAL_PKA_STATE_READY;
801

803  hpka->OperationCpltCallback(hpka);
804 #else
