808 
809  /* Check that a Tx process is not already ongoing */
810  if (hsmartcard->gState == HAL_SMARTCARD_STATE_READY)
811  {
812  if ((ptmpdata == NULL) || (Size == 0U))
813  {
814  return HAL_ERROR;
815  }
816

818  __HAL_LOCK(hsmartcard);
819 
820  hsmartcard->gState = HAL_SMARTCARD_STATE_BUSY_TX;
821

823  tickstart = HAL_GetTick();
824

826  CLEAR_BIT(hsmartcard->Instance->CR1, USART_CR1_UE);
