569 HAL_StatusTypeDef HAL_UARTEx_ReceiveToIdle(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size, uint16_t *RxLen, uint32_t Timeout)
570 {
571  uint8_t *pdata8bits;
572  uint16_t *pdata16bits;
573  uint16_t uhMask;
574  uint32_t tickstart;
575 
576  /* Check that a Rx process is not already ongoing */
577  if (huart->RxState == HAL_UART_STATE_READY)
578  {
579  if ((pData == NULL) || (Size == 0U))
580  {
581  return HAL_ERROR;
582  }
583 
584  __HAL_LOCK(huart);
585 
586  huart->ErrorCode = HAL_UART_ERROR_NONE;
587  huart->RxState = HAL_UART_STATE_BUSY_RX;
588  huart->ReceptionType = HAL_UART_RECEPTION_TOIDLE;
589 
590  /* Init tickstart for timeout management */
591  tickstart = HAL_GetTick();
592 
593  huart->RxXferSize = Size;
594  huart->RxXferCount = Size;
595 
596  /* Computation of UART mask to apply to RDR register */
597  UART_MASK_COMPUTATION(huart);
598  uhMask = huart->Mask;
599 
600  /* In case of 9bits/No Parity transfer, pRxData needs to be handled as a uint16_t pointer */
601  if ((huart->Init.WordLength == UART_WORDLENGTH_9B) && (huart->Init.Parity == UART_PARITY_NONE))
602  {
603  pdata8bits = NULL;
604  pdata16bits = (uint16_t *) pData;
605  }
606  else
607  {
608  pdata8bits = pData;
609  pdata16bits = NULL;
610  }
611 
612  __HAL_UNLOCK(huart);
613 
614  /* Initialize output number of received elements */
615  *RxLen = 0U;
616 
617  /* as long as data have to be received */
618  while (huart->RxXferCount > 0U)
619  {
620  /* Check if IDLE flag is set */
621  if (__HAL_UART_GET_FLAG(huart, UART_FLAG_IDLE))
622  {
623  /* Clear IDLE flag in ISR */
624  __HAL_UART_CLEAR_FLAG(huart, UART_CLEAR_IDLEF);
625 
626  /* If Set, but no data ever received, clear flag without exiting loop */
627  /* If Set, and data has already been received, this means Idle Event is valid : End reception */
628  if (*RxLen > 0U)
629  {
630  huart->RxState = HAL_UART_STATE_READY;
631 
632  return HAL_OK;
633  }
634  }
635 
636  /* Check if RXNE flag is set */
637  if (__HAL_UART_GET_FLAG(huart, UART_FLAG_RXNE))
638  {
639  if (pdata8bits == NULL)
640  {
641  *pdata16bits = (uint16_t)(huart->Instance->RDR & uhMask);
642  pdata16bits++;
643  }
644  else
645  {
646  *pdata8bits = (uint8_t)(huart->Instance->RDR & (uint8_t)uhMask);
647  pdata8bits++;
648  }
649  /* Increment number of received elements */
650  *RxLen += 1U;
651  huart->RxXferCount--;
652  }
653 
654  /* Check for the Timeout */
655  if (Timeout != HAL_MAX_DELAY)
656  {
657  if (((HAL_GetTick() - tickstart) > Timeout) || (Timeout == 0U))
658  {
659  huart->RxState = HAL_UART_STATE_READY;
660 
661  return HAL_TIMEOUT;
662  }
663  }
664  }
665 
666  /* Set number of received elements in output parameter : RxLen */
667  *RxLen = huart->RxXferSize - huart->RxXferCount;
668  /* At end of Rx process, restore huart->RxState to Ready */
669  huart->RxState = HAL_UART_STATE_READY;
670 
671  return HAL_OK;
672  }
673  else
674  {
675  return HAL_BUSY;
676  }
677 }

827 static void UARTEx_SetNbDataToProcess(UART_HandleTypeDef *huart)
828 {
829  uint8_t rx_fifo_depth;
830  uint8_t tx_fifo_depth;
831  uint8_t rx_fifo_threshold;
832  uint8_t tx_fifo_threshold;
833  uint8_t numerator[] = {1U, 1U, 1U, 3U, 7U, 1U, 0U, 0U};
834  uint8_t denominator[] = {8U, 4U, 2U, 4U, 8U, 1U, 1U, 1U};
835 
836  if (huart->FifoMode == UART_FIFOMODE_DISABLE)
837  {
838  huart->NbTxDataToProcess = 1U;
839  huart->NbRxDataToProcess = 1U;
840  }
841  else
842  {
843  rx_fifo_depth = RX_FIFO_DEPTH;
844  tx_fifo_depth = TX_FIFO_DEPTH;
845  rx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_RXFTCFG) >> USART_CR3_RXFTCFG_Pos);
846  tx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_TXFTCFG) >> USART_CR3_TXFTCFG_Pos);
847  huart->NbTxDataToProcess = ((uint16_t)tx_fifo_depth * numerator[tx_fifo_threshold]) / (uint16_t)denominator[tx_fifo_threshold];
848  huart->NbRxDataToProcess = ((uint16_t)rx_fifo_depth * numerator[rx_fifo_threshold]) / (uint16_t)denominator[rx_fifo_threshold];
849  }
