167 {
168  uint32_t position = 0x00u;
169  uint32_t iocurrent;
170  uint32_t temp;
171  uint8_t both_edge;
172 
173  /* Check the parameters */
174  assert_param(IS_GPIO_ALL_INSTANCE(GPIOx));
175  assert_param(IS_GPIO_PIN(GPIO_Init->Pin));
176  assert_param(IS_GPIO_MODE(GPIO_Init->Mode));
177  assert_param(IS_GPIO_PULL(GPIO_Init->Pull));
178 
179  /* Configure the port pins */
180  while (((GPIO_Init->Pin) >> position) != 0x00u)
181  {
182  /* Get current io position */
183  iocurrent = (GPIO_Init->Pin) & (1uL << position);
184 
185  if (iocurrent != 0x00u)
186  {
187  /*--------------------- GPIO Mode Configuration ------------------------*/
188  /* In case of Alternate function mode selection */
189  if((GPIO_Init->Mode == GPIO_MODE_AF_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
190  {
191  /* Check the Alternate function parameters */
192  assert_param(IS_GPIO_AF_INSTANCE(GPIOx));
193  assert_param(IS_GPIO_AF(GPIO_Init->Alternate));
194 
195  /* Configure Alternate function mapped with the current IO */
196  temp = GPIOx->AFR[position >> 3u];
197  temp &= ~(0xFu << ((position & 0x07u) * 4u));
198  temp |= ((GPIO_Init->Alternate) << ((position & 0x07u) * 4u));
199  GPIOx->AFR[position >> 3u] = temp;
200  }
201 
202  /* Configure IO Direction mode (Input, Output, Alternate or Analog) */
203  temp = GPIOx->MODER;
204  temp &= ~(GPIO_MODER_MODE0 << (position * 2u));
205  temp |= ((GPIO_Init->Mode & GPIO_MODE) << (position * 2u));
206  GPIOx->MODER = temp;
207 
208  /* In case of Output or Alternate function mode selection */
209  if((GPIO_Init->Mode == GPIO_MODE_OUTPUT_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_PP) ||
210  (GPIO_Init->Mode == GPIO_MODE_OUTPUT_OD) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
211  {
212  /* Check the Speed parameter */
213  assert_param(IS_GPIO_SPEED(GPIO_Init->Speed));
214  /* Configure the IO Speed */
215  temp = GPIOx->OSPEEDR;
216  temp &= ~(GPIO_OSPEEDR_OSPEED0 << (position * 2u));
217  temp |= (GPIO_Init->Speed << (position * 2u));
218  GPIOx->OSPEEDR = temp;
219 
220  /* Configure the IO Output Type */
221  temp = GPIOx->OTYPER;
222  temp &= ~(GPIO_OTYPER_OT0 << position) ;
223  temp |= (((GPIO_Init->Mode & GPIO_OUTPUT_TYPE) >> 4u) << position);
224  GPIOx->OTYPER = temp;
225  }
226 
227  /* To use the GPIO register configuration for PULL-UP PULL-DOWN and NO-PULL define USE_GPIO_PULL_REGISTER,
228  * otherwise the IO PULL will be managed from the PWR peripheral block
229  */
230 #ifdef USE_GPIO_PULL_REGISTER
231  /* Activate the Pull-up or Pull down resistor for the current IO */
232  temp = GPIOx->PUPDR;
233  temp &= ~(GPIO_PUPDR_PUPD0 << (position * 2u));
234  temp |= ((GPIO_Init->Pull) << (position * 2u));
235  GPIOx->PUPDR = temp;
236 #else
237  if (GPIOx == GPIOA)
238  {
239  switch (GPIO_Init->Pull)
240  {
241  case GPIO_NOPULL:
242  LL_PWR_SetNoPullA(GPIO_Init->Pin);
243  break;
244  case GPIO_PULLUP:
245  LL_PWR_EnablePUA(GPIO_Init->Pin);
246  break;
247  case GPIO_PULLDOWN:
248  LL_PWR_EnablePDA(GPIO_Init->Pin);
249  break;
250  }
251  }
252  else // GPIOB

258  break;
259  case GPIO_PULLUP:

261  break;
262  case GPIO_PULLDOWN:
263  LL_PWR_EnablePDB(GPIO_Init->Pin);
264  break;
265  }
266  }
267 
268 #endif
269 
270  /*--------------------- EXTI Mode Configuration ------------------------*/
271  /* Configure the External Interrupt for the current IO */
272  if((GPIO_Init->Mode & EXTI_MODE) == EXTI_MODE)
273  {
274  both_edge = FALSE;
275 
276  /* Edge/Level line configuration */
277  temp = SYSCFG->IO_DTR;
278  if (GPIOx == GPIOA)
279  {
280  temp &= ~(1<<position);
281  temp |= (((GPIO_Init->Mode & LEVEL_MODE)>>4) << position);
282  }
283  else
284  {
285  temp &= ~((1<<position) << 16);
286  temp |= ((((GPIO_Init->Mode & LEVEL_MODE)>>4) << position) << 16);
287  }
288  SYSCFG->IO_DTR = temp;
289 
290  /* Both Edge line configuration */
291  if (!(GPIO_Init->Mode & LEVEL_MODE))
292  {
293  temp = SYSCFG->IO_IBER;
294  /* Clear Interrupt Edge register line configuration */
295  if (GPIOx == GPIOA)
296  {
297  temp &= ~(1<<position);
298  }
299  else
300  {
301  temp &= ~((1<<position) << 16);
302  }
303  /* Enable both edges detection line configuration */
304  if ((GPIO_Init->Mode & BOTH_EDGE) == BOTH_EDGE)
305  {
306  both_edge = TRUE;
307  /* Clear Interrupt polarity event register */
308  SYSCFG->IO_IEVR = temp;
309 
310  if (GPIOx == GPIOA)
311  {
312  temp |= (1<< position);
313  }
314  else
315  {
316  temp |= ((1<< position) << 16);
317  }
318  }
319  SYSCFG->IO_IBER = temp;
320  }
321 
322  /* Interrupt polarity line configuration */
323  temp = SYSCFG->IO_IEVR;
324  if (!both_edge) {
325  /* Clear Interrupt polarity event register line configuration */
326  if (GPIOx == GPIOA)
327  {
328  temp &= ~(1<<position);
329  }
330  else
331  {
332  temp &= ~((1<<position) << 16);
333  }
334  /* Enable rising edges detection line configuration */
335  if ((GPIO_Init->Mode & RISING_EDGE_HIGH_LEVEL) == RISING_EDGE_HIGH_LEVEL)
336  {
337  if (GPIOx == GPIOA)
338  {
339  temp |= (1<< position);
340  }
341  else

344  }
345  }
346  }
