168 HAL_StatusTypeDef HAL_USARTEx_EnableSlaveMode(USART_HandleTypeDef *husart)
169 {
170  uint32_t tmpcr1;
171 
172  /* Check parameters */
173  assert_param(IS_UART_SPI_SLAVE_INSTANCE(husart->Instance));
174 
175  /* Process Locked */
176  __HAL_LOCK(husart);
177 
178  husart->State = HAL_USART_STATE_BUSY;
179 
180  /* Save actual USART configuration */
181  tmpcr1 = READ_REG(husart->Instance->CR1);
182 
183  /* Disable USART */
184  __HAL_USART_DISABLE(husart);
185 
186  /* In SPI slave mode mode, the following bits must be kept cleared:
187  - LINEN and CLKEN bit in the USART_CR2 register
188  - HDSEL, SCEN and IREN bits in the USART_CR3 register.*/
189  CLEAR_BIT(husart->Instance->CR2, (USART_CR2_LINEN | USART_CR2_CLKEN));
190  CLEAR_BIT(husart->Instance->CR3, (USART_CR3_SCEN | USART_CR3_HDSEL | USART_CR3_IREN));
191 
192  /* Enable SPI slave mode */
193  SET_BIT(husart->Instance->CR2, USART_CR2_SLVEN);
194 
195  /* Restore USART configuration */
196  WRITE_REG(husart->Instance->CR1, tmpcr1);
197 
198  husart->SlaveMode = USART_SLAVEMODE_ENABLE;
199 
200  husart->State = HAL_USART_STATE_READY;
201 
202  /* Enable USART */
203  __HAL_USART_ENABLE(husart);
204 
205  /* Process Unlocked */
206  __HAL_UNLOCK(husart);
207 
208  return HAL_OK;
209 }

493 static void USARTEx_SetNbDataToProcess(USART_HandleTypeDef *husart)
494 {
495  uint8_t rx_fifo_depth;
496  uint8_t tx_fifo_depth;
497  uint8_t rx_fifo_threshold;
498  uint8_t tx_fifo_threshold;
499  /* 2 0U/1U added for MISRAC2012-Rule-18.1_b and MISRAC2012-Rule-18.1_d */
500  uint8_t numerator[] = {1U, 1U, 1U, 3U, 7U, 1U, 0U, 0U};
501  uint8_t denominator[] = {8U, 4U, 2U, 4U, 8U, 1U, 1U, 1U};
502 
503  if (husart->FifoMode == USART_FIFOMODE_DISABLE)
504  {
505  husart->NbTxDataToProcess = 1U;
506  husart->NbRxDataToProcess = 1U;
507  }
508  else
509  {
510  rx_fifo_depth = RX_FIFO_DEPTH;
511  tx_fifo_depth = TX_FIFO_DEPTH;
512  rx_fifo_threshold = (uint8_t)((READ_BIT(husart->Instance->CR3, USART_CR3_RXFTCFG) >> USART_CR3_RXFTCFG_Pos) & 0xFFU);
513  tx_fifo_threshold = (uint8_t)((READ_BIT(husart->Instance->CR3, USART_CR3_TXFTCFG) >> USART_CR3_TXFTCFG_Pos) & 0xFFU);
514  husart->NbTxDataToProcess = ((uint16_t)tx_fifo_depth * numerator[tx_fifo_threshold]) / (uint16_t)denominator[tx_fifo_threshold];
515  husart->NbRxDataToProcess = ((uint16_t)rx_fifo_depth * numerator[rx_fifo_threshold]) / (uint16_t)denominator[rx_fifo_threshold];
516  }
517 }
