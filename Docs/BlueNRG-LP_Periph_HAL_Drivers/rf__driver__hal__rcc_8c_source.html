481 HAL_StatusTypeDef HAL_RCC_ClockConfig(RCC_ClkInitTypeDef *RCC_ClkInitStruct, uint32_t FLatency)
482 {
483  uint32_t tickstart;
484 
485  /* Check Null pointer */
486  if(RCC_ClkInitStruct == NULL)
487  {
488  return HAL_ERROR;
489  }
490 
491  /* Check the parameters */
492  assert_param(IS_FLASH_WAIT_STATES(FLatency));
493 
494  /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
495  must be correctly programmed according to the frequency of the FLASH clock */
496 
497  /* Setup flash wait states because according the system clock frequency */
498  if (FLatency != __HAL_FLASH_GET_WAIT_STATES())
499  {
500  /* Program the new number of wait states to the LATENCY bits in the FLASH_CONFIG register */
501  __HAL_FLASH_SET_WAIT_STATES(FLatency);
502 
503  /* Get Start Tick*/
504  tickstart = HAL_GetTick();
505 
506  /* Check that the new number of wait states is taken into account to access the Flash
507  memory by reading the FLASH_CONFIG register */
508  while (__HAL_FLASH_GET_WAIT_STATES() != FLatency)
509  {
510  if ((HAL_GetTick() - tickstart) > LATENCY_TIMEOUT_VALUE)
511  {
512  return HAL_TIMEOUT;
513  }
514  }
515  }
516 
517  /*------------------------- SYSCLK Configuration ---------------------------*/
518  assert_param(IS_RCC_SYSCLKSOURCE(RCC_ClkInitStruct->SYSCLKSource));
519  assert_param(IS_RCC_SYSCLK_DIVIDER(RCC_ClkInitStruct->SYSCLKDivider));
520 
521 
522  /* HSE is selected as System Clock Source */
523  if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSE)
524  {
525  /* Check the HSE ready flag */
526  if (LL_RCC_HSE_IsReady() == 0U)
527  {
528  return HAL_ERROR;
529  }
530  }
531  /* HSI is selected as System Clock Source */
532  else
533  {
534  /* Check the HSI ready flag */
535  if (LL_RCC_HSI_IsReady() == 0U)
536  {
537  return HAL_ERROR;
538  }
539  }
540 
541  /* Enable the RC64MPLL*/
542  __HAL_RCC_RC64MPLL_ENABLE();
543 
544  /* Get Start Tick*/
545  tickstart = HAL_GetTick();
546 
547  /* Wait till RC64MPLL is ready */
548  while (LL_RCC_RC64MPLL_IsReady() == 0)
549  {
550  if ((HAL_GetTick() - tickstart) > PLL_TIMEOUT_VALUE)
551  {
552  return HAL_TIMEOUT;
553  }
554  }
555 
556  /* Configure the RC64MPLL multiplication factor */
557  __HAL_RCC_RC64MPLL_PRESC_CONFIG(RCC_ClkInitStruct->SYSCLKDivider);
558 
559  /*---------------------------------------------------------------------------*/
560 
561  /* Update the SystemCoreClock global variable */
562  switch (LL_RCC_GetRC64MPLLPrescaler())
563  {
564  case LL_RCC_RC64MPLL_DIV_1:
565  SystemCoreClock = RC64MPLL_VALUE;
566  break;

568  SystemCoreClock = RC64MPLL_VALUE/2;
569  break;
570  case LL_RCC_RC64MPLL_DIV_4:
571  SystemCoreClock = RC64MPLL_VALUE/4;
572  break;
573  case LL_RCC_RC64MPLL_DIV_8:
574  SystemCoreClock = RC64MPLL_VALUE/8;
575  break;

577  SystemCoreClock = RC64MPLL_VALUE/16;
578  break;
579  case LL_RCC_RC64MPLL_DIV_32:
580  SystemCoreClock = RC64MPLL_VALUE/32;
581  break;
582  case LL_RCC_RC64MPLL_DIV_64:
583  SystemCoreClock = RC64MPLL_VALUE/64;
584  break;
585  }
586 
587  /* Configure the source of time base considering new system clocks settings*/
588  return HAL_InitTick (HAL_GetTickPrio());
