279 void HAL_FLASH_IRQHandler(void)
280 {
281  uint32_t param;
282  uint32_t error;
283 
284  /* Save flash errors. */
285  error = FLASH->IRQSTAT & FLASH_FLAG_ALL_ERRORS;
286 
287  /* A] Set parameter for user or error callbacks */
288 
289  /* Check if the operation was a program or erase */
290  if((pFlash.ProcedureOnGoing & (FLASH_CMD_WRITE | FLASH_CMD_BURSTWRITE | FLASH_CMD_OTPWRITE)) != 0x00u)
291  {
292  /* Return adress being programmed */
293  param = pFlash.Address;
294  }
295  else if((pFlash.ProcedureOnGoing & (FLASH_CMD_MASSERASE | FLASH_CMD_ERASE_PAGES)) != 0x00u)
296  {
297  /* return page number being erased (0 for mass erase) */
298  param = pFlash.Page;
299  }
300  else
301  {
302  param = 0U;
303  }
304 
305  /* B] Check errors */
306  if(error != 0x00u)
307  {
308  /* Save the error code */
309  pFlash.ErrorCode |= error;
310 
311  /* Clear error flags */
312  __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_ALL_ERRORS);
313 
314  /* Stop the procedure ongoing */
315  pFlash.ProcedureOnGoing = 0;
316 
317  /* Error callback */
318  HAL_FLASH_OperationErrorCallback(param);
319  }
320 
321  /* C] Check FLASH End of Operation flag */
322  if(__HAL_FLASH_GET_IT(FLASH_IT_CMDDONE))
323  {
324 
325  /* Clear FLASH End of Operation IT flag pending bit */
326  __HAL_FLASH_CLEAR_IT(FLASH_IT_CMDDONE);
327 
328  /* Clear FLASH End of Operation pending bit */
329  __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_CMDDONE);
330 
331 
332  if(pFlash.ProcedureOnGoing == FLASH_CMD_ERASE_PAGES)
333  {
334  /* Nb of pages to erased can be decreased */
335  pFlash.NbPagesToErase--;
336 
337  /* Check if there are still pages to erase */
338  if(pFlash.NbPagesToErase != 0x00u)
339  {
340  /* Increment page number */
341  pFlash.Page++;
342  FLASH_PageErase(pFlash.Page);
343  }
344  else
345  {
346  pFlash.Page = 0xFFFFFFFFU;
347  pFlash.ProcedureOnGoing = 0;
348  }
349  }
350  else
351  {
352  /*Stop the ongoing procedure */
353  pFlash.Page = 0xFFFFFFFFU;
354  pFlash.ProcedureOnGoing = 0;
355  }
356 
357  /* User callback */
358  HAL_FLASH_EndOfOperationCallback(param);
359  }
360 
361  if(pFlash.ProcedureOnGoing == 0)
362  {
363  /* Disable End of Operation and Error interrupts */
364  __HAL_FLASH_DISABLE_IT(FLASH_IT_CMDDONE | FLASH_IT_CMDERR | FLASH_IT_ILLCMD);
365 
366  /* Process Unlocked */
367  __HAL_UNLOCK(&pFlash);
368  }
369 }

456 HAL_StatusTypeDef FLASH_WaitForLastOperation(uint32_t Timeout)
457 {
458  uint32_t tickstart = HAL_GetTick();
459 
460  /* Wait for the FLASH operation to complete by polling on CMDDONE flag to be set.
461  If the FLASH operation fails, an error flag will be set */
462  while(__HAL_FLASH_GET_FLAG(FLASH_FLAG_CMDDONE) != SET)
463  {
464  if(Timeout != HAL_MAX_DELAY)
465  {
466  if((HAL_GetTick() - tickstart) >= Timeout)
467  {
468  return HAL_TIMEOUT;
469  }
470  }
471  }
472 
473  /* Check flash errors */
474  if (__HAL_FLASH_GET_FLAG(FLASH_FLAG_ALL_ERRORS))
475  {
476  /* Save the error code */
477  pFlash.ErrorCode |= FLASH->IRQSTAT;
478 
479  /* Clear error flags */
480  __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_ALL_ERRORS);
481 
482  return HAL_ERROR;
483  }
484 
485  return HAL_OK;
486 }
