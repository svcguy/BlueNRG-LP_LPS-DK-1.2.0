146 {
147  uint32_t tickstart;
148 
149  /* Check the IWDG handle allocation */
150  if (hiwdg == NULL)

152  return HAL_ERROR;
153  }
154 
155  /* Check the parameters */
156  assert_param(IS_IWDG_ALL_INSTANCE(hiwdg->Instance));
157  assert_param(IS_IWDG_PRESCALER(hiwdg->Init.Prescaler));
158  assert_param(IS_IWDG_RELOAD(hiwdg->Init.Reload));
159  assert_param(IS_IWDG_WINDOW(hiwdg->Init.Window));
160 
161  /* Enable IWDG. LSI is turned on automaticaly */
162  __HAL_IWDG_START(hiwdg);
163 
164  /* Enable write access to IWDG_PR, IWDG_RLR and IWDG_WINR registers by writing
165  0x5555 in KR */
166  IWDG_ENABLE_WRITE_ACCESS(hiwdg);
167 
168  /* Write to IWDG registers the Prescaler & Reload values to work with */
169  hiwdg->Instance->PR = hiwdg->Init.Prescaler;
170  hiwdg->Instance->RLR = hiwdg->Init.Reload;
171 
172  /* Check pending flag, if previous update not done, return timeout */
173  tickstart = HAL_GetTick();
174 
175  /* Wait for register to be updated */
176  while (hiwdg->Instance->SR != 0x00u)
177  {
178  if ((HAL_GetTick() - tickstart) > HAL_IWDG_DEFAULT_TIMEOUT)
179  {
180  return HAL_TIMEOUT;
181  }
182  }
183 
184  /* If window parameter is different than current value, modify window
185  register */
186  if (hiwdg->Instance->WINR != hiwdg->Init.Window)
187  {
188  /* Write to IWDG WINR the IWDG_Window value to compare with. In any case,
189  even if window feature is disabled, Watchdog will be reloaded by writing
190  windows register */
191  hiwdg->Instance->WINR = hiwdg->Init.Window;
192  }
193  else
194  {
195  /* Reload IWDG counter with value defined in the reload register */
196  __HAL_IWDG_RELOAD_COUNTER(hiwdg);
197  }
198 
199  /* Return function status */
200  return HAL_OK;
201 }
