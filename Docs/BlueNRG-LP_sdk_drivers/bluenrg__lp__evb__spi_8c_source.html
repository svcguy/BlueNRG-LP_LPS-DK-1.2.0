167 int32_t BSP_SPI_Write(void *handle, uint8_t Reg, uint8_t *pBuff, uint16_t nBuffSize)
168 {
169  /* Open the SPI communication by drive low the CS pin */
170  LL_GPIO_ResetOutputPin(BSP_SPI_CS_SENSOR1_GPIO_PORT, BSP_SPI_CS_SENSOR1_PIN);
171 
172  while(LL_SPI_IsActiveFlag_TXE(BSP_SPI) == 0); /* TODO: add a SW timeout */
173 
174  /* Send the register address */
175  LL_SPI_TransmitData8(BSP_SPI, Reg);
176 
177  /* Wait for the TX flag */
178  while(LL_SPI_IsActiveFlag_RXNE(BSP_SPI) == 0); /* TODO: add a SW timeout */
179 
180  /* Read the available data */
181  LL_SPI_ReceiveData8(BSP_SPI);
182 
183  for(uint16_t i = 0; i < nBuffSize; i++) {
184  /* Wait for the TX flag */
185  while(LL_SPI_IsActiveFlag_TXE(BSP_SPI) == 0); /* TODO: add a SW timeout */
186 
187  /* Send the data available */
188  LL_SPI_TransmitData8(BSP_SPI, pBuff[i]);
189 
190  /* Wait for the TX flag */
191  while(LL_SPI_IsActiveFlag_RXNE(BSP_SPI) == 0); /* TODO: add a SW timeout */
192 
193  /* Read the available data */
194  LL_SPI_ReceiveData8(BSP_SPI);
195  }
196 
197  /* Wait for the Busy flag */
198  while(LL_SPI_IsActiveFlag_BSY(BSP_SPI) == 1);
199 
200  /* Close the SPI communication by drive high the CS pin */
201  LL_GPIO_SetOutputPin(BSP_SPI_CS_SENSOR1_GPIO_PORT, BSP_SPI_CS_SENSOR1_PIN);
202 
203  return 0;
204 }
