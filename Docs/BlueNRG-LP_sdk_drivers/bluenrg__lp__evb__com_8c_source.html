227 uint8_t BSP_COM_Read(uint8_t *pData)
228 {
229  uint8_t tmp = 0;
230 
231  /* If the user callback is not NULL */
232  if(BSP_COM_RxDataCb.RxDataUserCb != NULL) {
233  tmp = Read_Buffer_Pop(pData);
234 
235 #if defined(__CC_ARM) || (defined(__ARMCC_VERSION) && (__ARMCC_VERSION >= 6100100))
236  if(*(int *)pData == EOF)
237  return 0;
238 #endif
239  }
240  /* If the user callback is NULL, then the management of the reception is done manually */
241  else {
242  if(LL_USART_IsActiveFlag_RXNE(BSP_UART) == 1) {
243  *pData = LL_USART_ReceiveData8(BSP_UART);
244  tmp = 1;
245  }
246  }
247 
248  return tmp; /* 1: character has been read, 0: no character */
249 }
