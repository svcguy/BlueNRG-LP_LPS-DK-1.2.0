533 static void BLETask( void *pvParameters )
534 {
535  /* To make sure no other BLE functions are called from other tasks. */
536  xSemaphoreTake(BLETickSemaphoreHandle, portMAX_DELAY);
537 
538  /* Init the BlueNRG-LP device */
539  Device_Init();
540 
541  /* Start Beacon Non Connectable Mode*/
542  Start_Beaconing();
543 
544  /* BLE is initialized. Let other tasks call BLE functions. */
545  xSemaphoreGive(BLETickSemaphoreHandle);
546 
547  while(1)
548  {
549  /* Take the semaphore to avoid that other ACI functions can interrupt the
550  execution of ModulesTick(). */
551  xSemaphoreTake(BLETickSemaphoreHandle, portMAX_DELAY);
552  ModulesTick();
553  xSemaphoreGive(BLETickSemaphoreHandle);
554  /* Preliminary check if there is something to process at the moment. If
555  something else to process will arrive later, before chip goes to sleep,
556  this will be handled inside HAL_PWR_MNGR_Request(). */
557  if(BLE_STACK_SleepCheck() != POWER_SAVE_LEVEL_RUNNING && HAL_VTIMER_SleepCheck() == TRUE)
558  {
559  xSemaphoreTake(radioActivitySemaphoreHandle, portMAX_DELAY);
560  }
561  }
562 }

566 static void testTask( void *pvParameters )
567 {
568  TickType_t xNextWakeTime;
569 
570  /* Initialise xNextWakeTime - this only needs to be done once. */
571  xNextWakeTime = xTaskGetTickCount();
572 
573  for( ;; )
574  {
575  /* Place this task in the blocked state until it is time to run again.
576  The block time is specified in ticks, the constant used converts ticks
577  to ms. While in the Blocked state this task will not consume any CPU
578  time. */

596 static void changeADVDataTask( void *pvParameters )
597 {
598  TickType_t xNextWakeTime;
599  uint8_t Random_Number[8];
600 
601  /* Initialise xNextWakeTime - this only needs to be done once. */
602  xNextWakeTime = xTaskGetTickCount();
603 
604  for( ;; )
605  {
606  /* Place this task in the blocked state until it is time to run again.
607  The block time is specified in ticks, the constant used converts ticks
608  to ms. While in the Blocked state this task will not consume any CPU
609  time. */
610  vTaskDelayUntil( &xNextWakeTime, ADV_CHANGE_PERIOD );
611 
612  BLE_ACI_PROTECTED(hci_le_rand(Random_Number));
613 
614  adv_data[25] = Random_Number[0];
615 
616  /* In this case there is no need to disable advertising before updating buffer
617  content, since only one byte is changed (atomic operation) and there is no
618  risk to have inconsistent data. */
619 
620  BLE_ACI_PROTECTED(aci_gap_set_advertising_data(0, ADV_COMPLETE_DATA, SHORT_ADV_DATA_LENGTH, adv_data));
621 
622 
623 #if EXTENDED_ADV
624  BLE_ACI_PROTECTED(aci_gap_set_advertising_data(1, ADV_COMPLETE_DATA, sizeof(adv_data), adv_data));
625 #endif
626 
627  PRINTF("ADV change %d\r\n", adv_data[25]);
628 
629  }
630 }
