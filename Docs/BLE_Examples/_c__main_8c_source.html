463 /*******************************************************************************
464 * Function Name : main.
465 * Description : Main routine.
466 * Input : None.
467 * Output : None.
468 * Return : None.
469 *******************************************************************************/
470 int main(void)
471 {
472  uint8_t ret;
473  uint32_t powersave_state = 0;
474  PowerSaveLevels stopMode = POWER_SAVE_LEVEL_STOP_NOTIMER;
475  PowerSaveLevels stopLevel;
476 
477  /* System initialization function */
478  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS)
479  {
480  /* Error during system clock configuration take appropriate action */
481  while(1);
482  }
483  /* Configure IOs for pwer save modes */
484  BSP_IO_Init();
485 
486  /* Configure I/O communication channel */
487  BSP_COM_Init(BSP_COM_RxDataUserCb);
488 
489  ModulesInit();
490 
491  /* Master Init procedure */
492  if (deviceInit() != BLE_STATUS_SUCCESS) {
493  PRINTF("Error during the device init procedure\r\n");
494  }
495 
496  /* Master set security */
497  ret = Device_Security();
498  if (ret != BLE_STATUS_SUCCESS) {
499  PRINTF("Error in Device_Security() 0x%02x\r\n", ret);
500  while(1);
501  }
502 
503  /* Initialize the button */
504  BSP_PB_Init(USER_BUTTON, BUTTON_MODE_EXTI);
505  user_set_wakeup_source(USER_BUTTON);
506 
507  PRINTF("BlueNRG-LP BLE Security Central Application (version: %s, security mode: %s, button: %d)\r\n", BLE_APPLICATION_VERSION_STRING,Security_Configuration, USER_BUTTON);
508 
509 
510  /* Start Master Configure Scanning procedure */
511  if (deviceScanConfiguration() != BLE_STATUS_SUCCESS) {
512  PRINTF("Error during the Scanning procedure configuration\r\n");
513  }
514 
515  /* Start Master Configure Connection procedure */
516  if (deviceConnectionConfiguration() != BLE_STATUS_SUCCESS) {
517  PRINTF("Error during the Connection procedure configuration\r\n");
518  }
519 
520  /* Start Master Device Discovery procedure */
521  if (deviceDiscovery() != BLE_STATUS_SUCCESS) {
522  PRINTF("Error during the device discovery procedure\r\n");
523  }
524 
525  /* Main loop */
526  while(1) {
527 
528  ModulesTick();
529 
530  /* Application Tick */
531  APP_Tick();
532 
533  /* Master Tick */
534  powersave_state = Master_Process();
535 
536  /* Check master library power state LSB bit: if 1 micro can enter in low powe mode */
537  (powersave_state & 0x1) ? (stopMode = POWER_SAVE_LEVEL_STOP_NOTIMER) : (stopMode = POWER_SAVE_LEVEL_RUNNING);
538 
539  /* Power Save management: no timer and wakeup on UART RX, PUSH button */
540  HAL_PWR_MNGR_Request(stopMode, wakeupIO, &stopLevel);
541 
542  }
543 }
