299 /*******************************************************************************
300 * Function Name : main.
301 * Description : Main routine.
302 * Input : None.
303 * Output : None.
304 * Return : None.
305 *******************************************************************************/
306 int main(void)
307 {
308  uint8_t ret;
309  WakeupSourceConfig_TypeDef wakeupIO;
310  PowerSaveLevels stopLevel;
311 
312  /* System initialization function */
313  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS)
314  {
315  /* Error during system clock configuration take appropriate action */
316  while(1);
317  }
318  /* Configure IOs for pwer save modes */
319  BSP_IO_Init();
320 
321  BSP_PB_Init(BSP_PUSH1, BUTTON_MODE_EXTI);
322 #if CLIENT
323  BSP_PB_Init(BSP_PUSH2, BUTTON_MODE_EXTI);
324 #endif
325 
326  /* Configure I/O communication channel */
327  BSP_COM_Init(BSP_COM_RxDataUserCb);
328 
329  /* Configure application LED */
330  BSP_LED_Init(BSP_LED1);
331  BSP_LED_Init(BSP_LED2);
332  BSP_LED_Init(BSP_LED3);
333 
334  ModulesInit();
335 
336  ret = RC_DeviceInit();
337  if (ret != BLE_STATUS_SUCCESS) {
338  PRINTF("RC_DeviceInit() failed: 0x%02x\r\n", ret);
339  while(1);
340  }
341 
342  PRINTF("BlueNRG-LP BLE Remote Control Application (version: %s)\r\n", BLE_RC_VERSION_STRING);
343 
344  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
345  wakeupIO.IO_Mask_Low_polarity = 0;
346  wakeupIO.RTC_enable = 0;
347  wakeupIO.LPU_enable = 0;
348 
349  while(1) {
350 
351  ModulesTick();
352 
353  /* Application Tick */
354  APP_Tick();
355 
356 #if ENABLE_LOW_POWER_MODE
357  /* Power Save Request */
358  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
359 #else
360  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_CPU_HALT, wakeupIO, &stopLevel);
361 #endif
362  }
363 }
