301 int main(void)
302 {
303  uint8_t ret;
304  WakeupSourceConfig_TypeDef wakeupIO;
305  PowerSaveLevels stopLevel;
306 
307  /* System initialization function */
308  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS)
309  {
310  /* Error during system clock configuration take appropriate action */
311  while(1);
312  }
313  /* Configure IOs for pwer save modes */
314  BSP_IO_Init();
315 
316  /* Configure I/O communication channel */
317  BSP_COM_Init(BSP_COM_RxDataUserCb);
318 
319 #if (ENABLE_BLUEVOICE) && !(ENABLE_BLUEVOICE_TIMER)
320  // Init the systick
321  Clock_Init();
322 #endif
323 
324  ModulesInit();
325 
326  /* Application demo Led Init */
327  BSP_LED_Init(BSP_LED1); //Activity led
328  BSP_LED_Init(BSP_LED2);
329  BSP_LED_Init(BSP_LED3); //Error led and ....
330  BSP_LED_On(BSP_LED1);
331  BSP_LED_Off(BSP_LED2);
332  BSP_LED_Off(BSP_LED3);
333 
334  PRINTF("BlueNRG-LP BLE Sensor Demo Application (version: %s)\r\n", BLE_SENSOR_VERSION_STRING);
335 
336 #if CONFIG_OTA_USE_SERVICE_MANAGER
337  /* Initialize the button: to be done before Sensor_DeviceInit for avoiding to
338  overwrite pressure/temperature sensor IO configuration when using BUTTON_2 (IO5) */
339  BSP_PB_Init(USER_BUTTON, BUTTON_MODE_GPIO);
340 #endif /* CONFIG_OTA_USE_SERVICE_MANAGER */
341 
342 
343  /* Sensor Device Init */
344  ret = Sensor_DeviceInit();
345  if (ret != BLE_STATUS_SUCCESS) {
346  BSP_LED_On(BSP_LED3);
347  while(1);
348  }
349 
350  /* No Wakeup Source needed */
351  wakeupIO.IO_Mask_High_polarity = 0;
352  wakeupIO.IO_Mask_Low_polarity = 0;
353  wakeupIO.RTC_enable = 0;
354  wakeupIO.LPU_enable = 0;
355 
356  while(1)
357  {
358  ModulesTick();
359 
360  /* Application Tick */
361  APP_Tick();
362 
363  /* Power Save Request */
364  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_NOTIMER, wakeupIO, &stopLevel);
365 
366 #if ST_OTA_FIRMWARE_UPGRADE_SUPPORT
367  /* Check if the OTA firmware upgrade session has been completed */
368  if (OTA_Tick() == 1)
369  {
370  /* Jump to the new application */
371  OTA_Jump_To_New_Application();
372  }
373 #endif /* ST_OTA_FIRMWARE_UPGRADE_SUPPORT */
374 
375 #if CONFIG_OTA_USE_SERVICE_MANAGER
376  if (BSP_PB_GetState(USER_BUTTON) == SET)
377  {
378  OTA_Jump_To_Service_Manager_Application();
379  }
380 #endif /* CONFIG_OTA_USE_SERVICE_MANAGER */
381  }
382 }
