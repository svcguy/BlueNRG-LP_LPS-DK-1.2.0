329 static void verifyFW(uint32_t appAddress)
330 {
331  uint32_t FW_size;
332  uint32_t *FW_input;
333  uint32_t *FW_sign_2048;
334  uint32_t *R2;
335  uint32_t *Modulus_2048;
336  uint32_t *e_65537;
337  uint32_t Modulus_2048_size, e_65537_size;
338  uint8_t FW_input_digest[CRL_SHA256_SIZE]; /* Buffer to store the computed
339  digest: MUST be allocated
340  before calling the SHA256
341  functions */
342  int32_t outSize, /* Used to store the digest's length after its computation */
343  doubleCheck; /* Used to prevent fault attacks on verification step */
344  int32_t retval;
345 
346  secureBoot_type *secureBoot = (secureBoot_type*)SYSTEM_MEMORY_BASE;
347 
348  /* Disable Peripheral Radio */
349  LL_APB2_DisableClock(LL_APB2_PERIPH_MRBLE);
350  /* Configure 1 FLASH Wait State */
351  LL_FLASH_SetWaitStates(FLASH, LL_FLASH_WAIT_STATES_1);
352  /* Setup the HSI frequency to 64 MHz */
353  LL_RCC_SetRC64MPLLPrescaler(LL_RCC_RC64MPLL_DIV_1);
354 
355  FW_size = *(uint32_t*)(appAddress + 0x18);
356  FW_input = (uint32_t*)appAddress;
357  FW_sign_2048 = (uint32_t*)((uint8_t*)FW_input+FW_size);
358  R2 = (uint32_t*)SECURE_BOOT_OTP_R2_ADDR;
359  Modulus_2048 = (uint32_t*)SECURE_BOOT_OTP_MODULUS_2048_ADDR;
360  Modulus_2048_size = 256;
361  e_65537 = (uint32_t*)SECURE_BOOT_OTP_E_65537_ADDR;
362  e_65537_size = 4;
363 
364  /* -----------------------------------------------------------------------*
365  * Generate digest of the customer's public key (SHA-256) -----------------*
366  * ------------------------------------------------------------------------*/
367  retval = secureBoot->generate_SHA256_digest((uint8_t*)FW_input,
368  FW_size,
369  FW_input_digest,
370  &outSize);
371  if (retval != HASH_SUCCESS) {
372  while(1);
373  }
374 
375  /* -----------------------------------------------------------------------*
376  * RSA PKCS#1 v1.5 Signature verification ---------------------------------*
377  * ------------------------------------------------------------------------*/
378  retval = secureBoot->verify_RSA_precomputedR2((uint8_t*)Modulus_2048, Modulus_2048_size,
379  (uint8_t*)e_65537, e_65537_size,
380  (uint8_t*)R2,
381  (uint8_t*)FW_sign_2048,
382  FW_input_digest,
383  E_SHA256,
384  &doubleCheck);
385  /* Check the return code: the following double check is provided to be
386  * resistant to fault attacks
387  */
388  if ((retval != SIGNATURE_VALID) || ((retval ^ doubleCheck) != FAULT_RESISTANT_PATTERN)) {
389  while(1);
390  }
391 }

400 int main(void)
401 {
402  pFunction Jump_To_Application;
403  uint32_t JumpAddress, appAddress;
404  /* Identifies the valid application where to jump based on the OTA application validity tags values placed on
405  reserved vector table entry: OTA_TAG_VECTOR_TABLE_ENTRY_INDEX */
406  appAddress = OTA_Check_Application_Tags_Value();
407 
408  if (appAddress == 0) {
409  /* This case indicates that no valid application is present and this normally should not happen */
410  while (1);
411  }
412 
413  /* Secure boot activation*/
414  if (*(volatile uint32_t*)SECURE_BOOT_OTP_ACTIVATION_ADDR == SECURE_BOOT_MAGIC_WORD)
415  verifyFW(appAddress);
416 
417  /* Jump to user application */
418  JumpAddress = *(__IO uint32_t*) (appAddress + 4);
419  Jump_To_Application = (pFunction) JumpAddress;
420  /* Initialize user application's Stack Pointer */
421  __set_MSP(*(__IO uint32_t*) appAddress);
422  Jump_To_Application();
423 
424  /* Infinite loop */
425  while (1)
426  {
427  }
428 }
