353  uint32_t appAddress = 0;
354 
355  if (((TAG_VALUE(APP_OTA_SERVICE_ADDRESS) == OTA_SERVICE_MANAGER_TAG) && (TAG_VALUE(APP_WITH_OTA_SERVICE_ADDRESS) == OTA_IN_PROGRESS_TAG))|| /* 10 */

380 static void verifyFW(uint32_t appAddress)
381 {
382  uint32_t FW_size;
383  uint32_t *FW_input;
384  uint32_t *FW_sign_2048;
385  uint32_t *R2;
386  uint32_t *Modulus_2048;
387  uint32_t *e_65537;
388  uint32_t Modulus_2048_size, e_65537_size;
389  uint8_t FW_input_digest[CRL_SHA256_SIZE]; /* Buffer to store the computed
390  digest: MUST be allocated
391  before calling the SHA256
392  functions */
393  int32_t outSize, /* Used to store the digest's length after its computation */
394  doubleCheck; /* Used to prevent fault attacks on verification step */
395  int32_t retval;
396 
397  secureBoot_type *secureBoot = (secureBoot_type*)SYSTEM_MEMORY_BASE;
398 
399  /* Disable Peripheral Radio */
400  LL_APB2_DisableClock(LL_APB2_PERIPH_MRBLE);
401  /* Configure 1 FLASH Wait State */
402  LL_FLASH_SetWaitStates(FLASH, LL_FLASH_WAIT_STATES_1);
403  /* Setup the HSI frequency to 64 MHz */
404  LL_RCC_SetRC64MPLLPrescaler(LL_RCC_RC64MPLL_DIV_1);
405 
406  FW_size = *(uint32_t*)(appAddress + 0x18);
407  FW_input = (uint32_t*)appAddress;
408  FW_sign_2048 = (uint32_t*)((uint8_t*)FW_input+FW_size);
409  R2 = (uint32_t*)SECURE_BOOT_OTP_R2_ADDR;
410  Modulus_2048 = (uint32_t*)SECURE_BOOT_OTP_MODULUS_2048_ADDR;
411  Modulus_2048_size = 256;
412  e_65537 = (uint32_t*)SECURE_BOOT_OTP_E_65537_ADDR;
413  e_65537_size = 4;
414 
415  /* -----------------------------------------------------------------------*
416  * Generate digest of the customer's public key (SHA-256) -----------------*
417  * ------------------------------------------------------------------------*/
418  retval = secureBoot->generate_SHA256_digest((uint8_t*)FW_input,
419  FW_size,
420  FW_input_digest,
421  &outSize);
422  if (retval != HASH_SUCCESS) {
423  while(1);
424  }
425 
426  /* -----------------------------------------------------------------------*
427  * RSA PKCS#1 v1.5 Signature verification ---------------------------------*
428  * ------------------------------------------------------------------------*/
429  retval = secureBoot->verify_RSA_precomputedR2((uint8_t*)Modulus_2048, Modulus_2048_size,
430  (uint8_t*)e_65537, e_65537_size,
431  (uint8_t*)R2,
432  (uint8_t*)FW_sign_2048,
433  FW_input_digest,
434  E_SHA256,
435  &doubleCheck);
436  /* Check the return code: the following double check is provided to be
437  * resistant to fault attacks
438  */
439  if ((retval != SIGNATURE_VALID) || ((retval ^ doubleCheck) != FAULT_RESISTANT_PATTERN)) {
440  while(1);
441  }
442 }

444 int main(void)
445 {
446  pFunction Jump_To_Application;
447  uint32_t JumpAddress, appAddress;
448  uint8_t ret;
449  WakeupSourceConfig_TypeDef wakeupIO;
450  PowerSaveLevels stopLevel;
451 
452  /* Check Service manager RAM Location to verify if a jump to Service Manager has been set from the Application */
453  OTA_Check_ServiceManager_Operation();
454 
455  /* Identifies the valid application where to jump based on the OTA application validity tags values placed on
456  reserved vector table entry: OTA_TAG_VECTOR_TABLE_ENTRY_INDEX */
457  appAddress = OTA_Check_Application_Tags_Value();
458 
459  /* Check if there is a valid application where to jump */
460  if (appAddress == APP_WITH_OTA_SERVICE_ADDRESS)
461  {
462  /* Secure boot activation*/
463  if (*(volatile uint32_t*)SECURE_BOOT_OTP_ACTIVATION_ADDR == SECURE_BOOT_MAGIC_WORD)
464  verifyFW(appAddress);
465 
466  /* Jump to user application */
467  JumpAddress = *(__IO uint32_t*) (appAddress + 4);
468  Jump_To_Application = (pFunction) JumpAddress;
469  /* Initialize user application's Stack Pointer */
470  __set_MSP(*(__IO uint32_t*) appAddress);
471  Jump_To_Application();
472 
473  /* Infinite loop */
474  while (1)
475  {
476  }
477  }
478 
479  /* Here Ota Service Manager Application is started */
480 
481  /* System initialization function */
482  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS)
483  {
484  /* Error during system clock configuration take appropriate action */
485  while(1);
486  }
487  /* Configure IOs for pwer save modes */
488  BSP_IO_Init();
489 
490  /* Init the UART peripheral */
491  BSP_COM_Init(NULL);
492 
493  ModulesInit();
494 
495  PRINTF("\r\nBlueNRG-LP BLE OTA Service Manager (version: %s)\r\n", BLE_OTA_SERVICE_MANAGER_VERSION_STRING);
496 
497  /* Init OTA Service Manager Device */
498  ret = OTA_ServiceManager_DeviceInit();
499  if (ret != BLE_STATUS_SUCCESS) {
500  PRINTF("OTA_ServiceManager_DeviceInit()--> Failed 0x%02x\r\n", ret);
501  while(1);
502  }
503 
504  /* No Wakeup Source needed */
505  wakeupIO.IO_Mask_High_polarity = 0;
506  wakeupIO.IO_Mask_Low_polarity = 0;
507  wakeupIO.RTC_enable = 0;
508  wakeupIO.LPU_enable = 0;
509 
510  /* Erase the storage area from start page to end page */
511  OTA_Erase_Flash(APP_WITH_OTA_SERVICE_PAGE_NUMBER_START,APP_WITH_OTA_SERVICE_PAGE_NUMBER_END);
512 
513 
514  while(1)
515  {
516  ModulesTick();
517 
518  /* Application tick */
519  APP_Tick();
520 
521  /* Check if the OTA firmware upgrade session has been completed */
522  if (OTA_Tick() == 1)
523  {
524  /* Jump to the new application */
525  OTA_Jump_To_New_Application();
526  }
527  /* Power Save Request */
528  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_NOTIMER, wakeupIO, &stopLevel);
529  }
530 
531 }
