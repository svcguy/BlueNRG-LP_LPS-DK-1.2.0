335  0, /* SID */
336  0 /* No scan request notifications */);
337  printf("Advertising configuration (Advertising Inteval: %d): %02X\n", adv_interval, ret);

340 
341  ret = aci_gap_set_advertising_data(0, ADV_COMPLETE_DATA, sizeof(adv_data), adv_data);

419 int main(void)
420 {
421  uint8_t charRead;
422  WakeupSourceConfig_TypeDef wakeupIO;
423  PowerSaveLevels stopLevel;
424 
425  /* System initialization function.
426  Set the clocks combination with the lowest power consumption.
427  Refer to Power_Consumption_Test_parameters.h header file
428  */
429  if (SystemInit(POWER_CONSUMPTION_DEMO_SYSTEM_CLOCK, POWER_CONSUMPTION_DEMO_BLE_CLOCK) != SUCCESS)
430  {
431  /* Error during system clock configuration take appropriate action */
432  while(1);
433  }
434  /* Configure IOs for pwer save modes */
435  BSP_IO_Init();
436 
437  /* Init the UART peripheral */
438  BSP_COM_Init(NULL);
439 
440  ModulesInit();
441 
442  /* Setup the BLE GAP and GATT layer*/
443  DeviceInit();
444 
445  printf("Power Consumption Test\r\n");
446  printf("Enter ? for list of commands\r\n");
447 
448  /* No Wakeup Source needed */
449  wakeupIO.IO_Mask_High_polarity = 0;
450  wakeupIO.IO_Mask_Low_polarity = 0;
451  wakeupIO.RTC_enable = 0;
452  wakeupIO.LPU_enable = 0;
453 
454  device_state = DEVICE_IDLE_STATE;
455  while(1) {
456 
457  ModulesTick();
458 
459  if (BSP_COM_Read(&charRead)) {
460  switch (charRead) {
461  case 'f':
462  if (device_state == DEVICE_IDLE_STATE) {
463  /* Device in discoverable mode with fast adv interval 100 ms */
464  Configure_DeviceAdvertising(ADV_INTERVAL_FAST_MS);
465  Set_DeviceConnectable();
466  device_state = DEVICE_ADV_STATE;
467  } else {
468  printf ("Device already in discoverable or connected state. Please reset the device.\r\n");
469  }
470  break;
471  case 's':
472  if (device_state == DEVICE_IDLE_STATE) {
473  /* Device in discoverable mode with slow adv interval 1000 ms */
474  Configure_DeviceAdvertising(ADV_INTERVAL_SLOW_MS);
475  Set_DeviceConnectable();
476  device_state = DEVICE_ADV_STATE;
477  } else {
478  printf ("Device already in discoverable or connected state. Please reset the device.\r\n");
479  }
480  break;
481  case '?':
482  help();
483  break;
484  case 'r':
485  NVIC_SystemReset();
486  break;
487  case '\r':
488  case '\n':
489  // ignored
490  break;
491  default:
492  printf("Unknown Command\r\n");
493  }
494  }
495 
496  switch(device_state) {
497  case DEVICE_DISCONNECTED_STATE:
498  /* Enable the UART */
499  BSP_COM_Init(NULL);
500  printf ("Device Disconnected!!!\r\n");
501  device_state = DEVICE_IDLE_STATE;
502  break;
503  case DEVICE_CONN_STATE:
504  /* Enable the UART */
505  BSP_COM_Init(NULL);
506  printf("Device Connected with Interval %d * 1.25 ms\r\n", connection_interval);
507  device_state = DEVICE_SLEEP_STATE;
508  break;
509  case DEVICE_IDLE_STATE:
510  /* Nothing to Do */
511  break;
512  default:
513  /* Disable the UART to save power */
514  while(LL_USART_IsEnabled(BSP_UART) && (BSP_COM_TxFifoNotEmpty() || BSP_COM_UARTBusy()));
515  BSP_COM_DeInit();
516 
517  /* Power Save Request */
518  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_NOTIMER, wakeupIO, &stopLevel);
519  }
520  }
521 }
