454 void FlashRoutine(void)
455 {
456  static uint32_t flash_counter = 0;
457  static uint32_t flash_pattern = 0xAAAAAAAA;
458 
459  if(flash_counter == 512) {
460  if (HAL_VTIMER_DiffSysTimeMs(Next_Advertising_SysTime, HAL_VTIMER_GetCurrentSysTime()) > FLASH_ERASE_GUARD_TIME) {
461 
462  if (LL_FLASH_GetFlag(FLASH, LL_FLASH_FLAG_CMDDONE) == SET) {
463  LL_FLASH_Erase(FLASH, LL_FLASH_TYPE_ERASE_PAGES, (BLE_BEACON_TEST_PAGE_ADDRESS-_MEMORY_FLASH_BEGIN_)/_MEMORY_BYTES_PER_PAGE_, 1);
464  flash_pattern = ~flash_pattern;
465  flash_counter = 0;
466  }
467  }
468  }
469  else {
470  if (HAL_VTIMER_DiffSysTimeMs(Next_Advertising_SysTime, HAL_VTIMER_GetCurrentSysTime()) > FLASH_WRITE_GUARD_TIME) {
471 
472  if (LL_FLASH_GetFlag(FLASH, LL_FLASH_FLAG_CMDDONE) == SET) {
473  LL_FLASH_Program(FLASH, (BLE_BEACON_TEST_PAGE_ADDRESS + flash_counter*4), flash_pattern);
474  flash_counter++;
475  }
476  }
477  }
478 }
