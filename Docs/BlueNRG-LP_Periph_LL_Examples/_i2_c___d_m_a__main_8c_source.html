604 void Handle_I2C_Slave(void)
605 {
606 
607  /* (1) Wait ADDR flag and check address match code and direction ************/
608  while(!LL_I2C_IsActiveFlag_ADDR(I2Cx))
609  {
610  }
611 
612  /* Verify the Address Match with the OWN Slave address */
613  if(LL_I2C_GetAddressMatchCode(I2Cx) == SLAVE_OWN_ADDRESS)
614  {
615  /* Verify the transfer direction, a write direction, Slave enters receiver mode */
616  if(LL_I2C_GetTransferDirection(I2Cx) == LL_I2C_DIRECTION_WRITE)
617  {
618  /* (1.1) Enable DMA transfer (before clearing ADDR FLag) ****************/
619  LL_DMA_EnableChannel(DMA1, LL_DMA_CHANNEL_3);
620  /* Clear ADDR flag value in ISR register */
621  LL_I2C_ClearFlag_ADDR(I2Cx);
622  }
623  else
624  {
625  /* Clear ADDR flag value in ISR register */
626  LL_I2C_ClearFlag_ADDR(I2Cx);
627 
628  /* Call Error function */
629  Error_Callback();
630  }
631  }
632  else
633  {
634  /* Clear ADDR flag value in ISR register */
635  LL_I2C_ClearFlag_ADDR(I2Cx);
636 
637  /* Call Error function */
638  Error_Callback();
639  }
640 
641  /* Loop until end of transfer completed (DMA TC raised) *****************/
642 
643 #if (USE_TIMEOUT == 1)
644  Timeout = DMA_SEND_TIMEOUT_TC_MS;
645 #endif /* USE_TIMEOUT */
646 
647  /* Loop until DMA transfer complete event */
648  while(!ubSlaveTransferComplete)
649  {
650 #if (USE_TIMEOUT == 1)
651  /* Check Systick counter flag to decrement the time-out value */
652  if (LL_SYSTICK_IsActiveCounterFlag())
653  {
654  if(Timeout-- == 0)
655  {
656  /* Time-out occurred. Set LED to blinking mode */
657  LED_Blinking(LED_BLINK_SLOW);
658  }
659  }
660 #endif /* USE_TIMEOUT */
661  }
662 
663  /* Loop until end of slave reception completed (STOP flag raised) *******/
664 
665 #if (USE_TIMEOUT == 1)
666  Timeout = I2C_SEND_TIMEOUT_STOP_MS;
667 #endif /* USE_TIMEOUT */
668 
669  /* Loop until STOP flag is raised */
670  while(!LL_I2C_IsActiveFlag_STOP(I2Cx))
671  {
672 #if (USE_TIMEOUT == 1)
673  /* Check Systick counter flag to decrement the time-out value */
674  if (LL_SYSTICK_IsActiveCounterFlag())
675  {
676  if(Timeout-- == 0)
677  {
678  /* Time-out occurred. Set LED2 to blinking mode */
679  LED_Blinking(LED_BLINK_SLOW);
680  }
681  }
682 #endif /* USE_TIMEOUT */
683  }
684 
685  /* Clear pending flags, Data consistency are checking into Slave process */
686 
687  /* End of I2C_SlaveReceiver_MasterTransmitter_DMA Process */
688  LL_I2C_ClearFlag_STOP(I2Cx);
689 
690  /* Check if datas request to turn on the LED2 */
691  if(Buffercmp8((uint8_t*)aReceiveBuffer, (uint8_t*)aLedOn, (ubNbDataToReceive-1)) == 0)
692  {
693  /* Turn LED2 On:
694  * Expected bytes have been received
695  * Slave Rx sequence completed successfully
696  */
697  LED_On();
698  printf("Slave Rx sequence completed successfully\n\r");
699  printf("** Test successfully. ** \n\r\n\r");
700  }
701  else
702  {
703  /* Call Error function */
704  Error_Callback();
705  }
706 }
