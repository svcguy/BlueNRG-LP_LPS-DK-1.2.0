306 static void MX_IWDG_Init(void)
307 {
308  /* Enable APB0 indipendent Watchdog peripherals clock */
309  LL_APB0_EnableClock(LL_APB0_PERIPH_WDG);
310 
311  /* Force WDG peripheral reset */
312  LL_APB0_ForceReset(LL_APB0_PERIPH_WDG);
313  LL_APB0_ReleaseReset(LL_APB0_PERIPH_WDG);
314 
315  /* Check if WDG Reset Release flag interrupt occurred or not */
316  while(LL_RCC_IsActiveFlag_WDGRSTREL() == 0)
317  {
318  }
319 
320  LL_IWDG_Enable(IWDG);
321  LL_IWDG_EnableWriteAccess(IWDG);
322  LL_IWDG_SetPrescaler(IWDG, LL_IWDG_PRESCALER_4);
323  LL_IWDG_SetWindow(IWDG, 0xFFF);
324  LL_IWDG_SetReloadCounter(IWDG, 0xFFE);
325  while (LL_IWDG_IsReady(IWDG) != 1)
326  {
327  }
328 
329  LL_IWDG_ReloadCounter(IWDG);
330 }
