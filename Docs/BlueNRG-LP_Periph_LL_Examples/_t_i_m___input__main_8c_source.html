374 void TimerCaptureCompare_Callback(void)
375 {
376  /* Capture index */
377  static uint16_t uhCaptureIndex = 0;
378 
379  /* Captured Values */
380  static uint32_t uwICValue1 = 0;
381  static uint32_t uwICValue2 = 0;
382  static uint32_t uwDiffCapture = 0;
383 
384  uint32_t TIMxCLK;
385  uint32_t PSC;
386  uint32_t IC1PSC;
387  uint32_t IC1Polarity;
388 
389  if(uhCaptureIndex == 0)
390  {
391  /* Get the 1st Input Capture value */
392  uwICValue1 = LL_TIM_IC_GetCaptureCH1(TIMx);
393  uhCaptureIndex = 1;
394  }
395  else if(uhCaptureIndex == 1)
396  {
397  /* Get the 2nd Input Capture value */
398  uwICValue2 = LL_TIM_IC_GetCaptureCH1(TIMx);
399 
400  /* Capture computation */
401  if (uwICValue2 > uwICValue1)
402  {
403  uwDiffCapture = (uwICValue2 - uwICValue1);
404  }
405  else if (uwICValue2 < uwICValue1)
406  {
407  uwDiffCapture = ((TIMx_ARR_MAX - uwICValue1) + uwICValue2) + 1;
408  }
409  else
410  {
411  /* If capture values are equal, we have reached the limit of frequency */
412  /* measures. */
413  LED_Blinking(LED_BLINK_ERROR);
414  }
415 
416  /* The signal frequency is calculated as follows: */
417  /* Frequency = (TIMx*IC1PSC) / (Capture*(PSC+1)*IC1Polarity) */
418  /* where: */
419  /* Capture is the difference between two consecutive captures */
420  /* TIMxCLK is the timer counter clock frequency */
421  /* PSC is the timer prescaler value */
422  /* IC1PSC is the input capture prescaler value */
423  /* IC1Polarity value depends on the capture sensitivity: */
424  /* 1 if the input is sensitive to rising or falling edges */
425  /* 2 if the input is sensitive to both rising and falling edges */
426 
427  /* Retrieve actual TIMx counter clock frequency */
428  TIMxCLK = LL_TIM_GetPeriphClock(TIMx);
429 
430  /* Retrieve actual TIMx prescaler value */
431  PSC = LL_TIM_GetPrescaler(TIMx);
432 
433  /* Retrieve actual IC1 prescaler ratio */
434  IC1PSC = __LL_TIM_GET_ICPSC_RATIO(LL_TIM_IC_GetPrescaler(TIMx, LL_TIM_CHANNEL_CH1));
435 
436  /* Retrieve actual IC1 polarity setting */
437  if (LL_TIM_IC_GetPolarity(TIMx, LL_TIM_CHANNEL_CH1) == LL_TIM_IC_POLARITY_BOTHEDGE)
438  IC1Polarity = 2;
439  else
440  IC1Polarity = 1;
441 
442  /* Calculate input signal frequency */
443  uwMeasuredFrequency = (TIMxCLK *IC1PSC) / (uwDiffCapture*(PSC+1)*IC1Polarity);
444 
445  /* reset capture index */
446  uhCaptureIndex = 0;
447  }
448 }
