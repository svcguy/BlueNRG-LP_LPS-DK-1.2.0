373 void Configure_RTC(void)
374 {
375  /* Enable Peripheral Clock */
376  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
377 
378  /* RTC peripheral reset */
379  LL_RCC_ClearFlag_RTCRSTREL();
380  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
381  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
382  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
383  LL_RCC_ClearFlag_RTCRSTREL();
384 
385  /* Disable the write protection for RTC registers */
386  LL_RTC_DisableWriteProtection(RTC);
387 
388  /* Set Initialization mode */
389  LL_RTC_EnableInitMode(RTC);
390  /* Check if the Initialization mode is set */
391  while (LL_RTC_IsActiveFlag_INIT(RTC) != 1)
392  {
393  }
394 
395  /* Set prescaler according to source clock */
396  LL_RTC_SetAsynchPrescaler(RTC, RTC_ASYNCH_PREDIV);
397  LL_RTC_SetSynchPrescaler(RTC, RTC_SYNCH_PREDIV);
398 
399  /* Disable wake up timer to modify it */
400  LL_RTC_WAKEUP_Disable(RTC);
401 
402  /* Wait until it is allow to modify wake up reload value */
403  while (LL_RTC_IsActiveFlag_WUTW(RTC) != 1)
404  {
405  }
406 
407  /* Exit Initialization mode */
408  LL_RTC_DisableInitMode(RTC);
409  /* Clear RSF flag */
410  LL_RTC_ClearFlag_RS(RTC);
411  /* Wait the registers to be synchronised */
412  while (LL_RTC_IsActiveFlag_RS(RTC) != 1)
413  {
414  }
415 
416  /* Enable RTC registers write protection */
417  LL_RTC_EnableWriteProtection(RTC);
418 }
