542 void WaitAndCheckEndOfTransfer(void)
543 {

547  }

549 
550  /* Disable DMA1 Tx Channel */
551  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_7);
552 
553  /* Wait end of reception */
554  while (ubReceptionComplete != 1)
555  {
556  }
557  /* Disable DMA1 Rx Channel */
558  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_6);
559 
560  PrintEcho();
561 
562  /* Compare received string to expected one */
563  if (Buffercmp8((uint8_t *)aStringToReceive, (uint8_t *)aRxBuffer, ubNbDataToReceive))
564  {
565  /* Processing Error */
566  LED_Blinking(LED_BLINK_ERROR);
