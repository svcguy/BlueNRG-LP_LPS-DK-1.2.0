551 void SPI_SLAVE_Rx_Callback(void)
552 {
553  /* Read character in Data register.
554  RXNE flag is cleared by reading data in DR register */
555 #if defined( CONFIG_DATASIZE_16BIT )
556  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData16(SPI_SLAVE);

558  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData8(SPI_SLAVE);
559 #endif

561  if(ubNbDataToReceive == ubReceiveIndex)
562  {
563  /* Disable RXNE Interrupt */
564  LL_SPI_DisableIT_RXNE(SPI_SLAVE);
565  }
