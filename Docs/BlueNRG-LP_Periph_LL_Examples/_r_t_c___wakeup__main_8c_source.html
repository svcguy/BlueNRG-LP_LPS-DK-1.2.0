315 void RTC_WakeupInit(void)
316 {
317  /* Enable Peripheral Clock */
318  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
319 
320  /* RTC peripheral reset */
321  LL_RCC_ClearFlag_RTCRSTREL();
322  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
323  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
324  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
325  LL_RCC_ClearFlag_RTCRSTREL();
326 
327  /* Disable the write protection for RTC registers */
328  LL_RTC_DisableWriteProtection(RTC);
329 
330  /* Init mode setup */
331  LL_RTC_EnableInitMode(RTC);
332 
333  /* Wait till the Init mode is active */
334  while(LL_RTC_IsActiveFlag_INIT(RTC) == RESET);
335 
336  /* Configure Hour Format */
337  LL_RTC_SetHourFormat(RTC, LL_RTC_HOURFORMAT_24HOUR);
338 
339  /* Output disabled */
340  LL_RTC_SetAlarmOutEvent(RTC, LL_RTC_ALARMOUT_DISABLE);
341 
342  /* Output polarity */
343  LL_RTC_SetOutputPolarity(RTC, LL_RTC_OUTPUTPOLARITY_PIN_HIGH);
344 
345  /* Set Asynchronous prescaler factor */
346  LL_RTC_SetAsynchPrescaler(RTC, 0x7F);
347 
348  /* Set Synchronous prescaler factor */
349  LL_RTC_SetSynchPrescaler(RTC, 0x00FF);
350 
351  /* Exit Initialization mode */
352  LL_RTC_DisableInitMode(RTC);
353 
354  /* Enable write protection */
355  LL_RTC_EnableWriteProtection(RTC);
356 }

358 void SetRTC_WakeupTimeout(uint32_t time)
359 {
360  /* Disable write protection */
361  LL_RTC_DisableWriteProtection(RTC);
362 
363  /* Disable Wake-up Timer */
364  LL_RTC_WAKEUP_Disable(RTC);
365 
366  /* In case of interrupt mode is used, the interrupt source must disabled */
367  LL_RTC_DisableIT_WUT(RTC);
368 
369  /* Wait till RTC WUTWF flag is set */
370  while(LL_RTC_IsActiveFlag_WUTW(RTC) == 0);
371 
372 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
373  /* Clear PWR wake up Flag */
374  LL_PWR_ClearWakeupSource(LL_PWR_EWS_INT);
375 #endif
376 
377  /* Clear RTC Wake Up timer Flag */
378  LL_RTC_ClearFlag_WUT(RTC);
379 
380  /* Configure the Wake-up Timer counter */
381  LL_RTC_WAKEUP_SetAutoReload(RTC, time);
382 
383  /* Configure the clock source */
384  LL_RTC_WAKEUP_SetClock(RTC, LL_RTC_WAKEUPCLOCK_CKSPRE);
385 
386  /* Configure the Interrupt in the RTC_CR register */
387  LL_RTC_EnableIT_WUT(RTC);
388 
389  /* Enable the Wake-up Timer */
390  LL_RTC_WAKEUP_Enable(RTC);
391 
392  /* Enable write protection */
393  LL_RTC_EnableWriteProtection(RTC);
394 
395  /* Configure NVIC for RTC */
396  NVIC_SetPriority(RTC_IRQn, IRQ_LOW_PRIORITY);
397  NVIC_EnableIRQ(RTC_IRQn);
398 
399  /* Clear RTC Wake Up timer Flag */
400  LL_RTC_ClearFlag_WUT(RTC);
401 }
