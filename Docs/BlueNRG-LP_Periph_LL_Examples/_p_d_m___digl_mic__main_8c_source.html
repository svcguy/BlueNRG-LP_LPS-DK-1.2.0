462 static void TC_IT_Callback(void)
463 {
464  uint16_t nBytes, nPCMs = 0;
465 
466  if(!discard_first_flow)
467  for ( nBytes = 0; nBytes< PCM_BUFFER_SIZE; nBytes++)
468  {
469  if((nBytes%2)==0) {
470  HP_Filter.Z = PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] * AudioVolume;
471  HP_Filter.oldOut = (0xFC * (HP_Filter.oldOut + HP_Filter.Z - HP_Filter.oldIn)) >> 8;
472  HP_Filter.oldIn = HP_Filter.Z;
473  PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] = SaturaLH(HP_Filter.oldOut, -32760, 32760);
474  nPCMs++;
475  }
476  }
477 
478  /* Send the PDM data through UART */
479  if(!discard_first_flow) {
480  start_sending_data = 1;
481  DMA_Rearm(LL_DMA_CHANNEL_1, (uint32_t)&(((uint8_t *)PCM_Buffer)[PCM_BUFFER_SIZE]), PCM_BUFFER_SIZE);
482  LL_DMA_EnableChannel(DMA1, LL_DMA_CHANNEL_1);
483  }
484  else
485  discard_first_flow--;
486 }
