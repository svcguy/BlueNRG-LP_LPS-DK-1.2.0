395 void PrintInfo(uint8_t *String, uint32_t Size)
396 {
397  uint32_t index = 0;
398  uint8_t *pchar = String;
399 
400  /* Send characters one per one, until last char to be sent */
401  for (index = 0; index < Size; index++)
402  {
403  /* Wait for TXE flag to be raised */
404  while (!LL_USART_IsActiveFlag_TXE(USART1))
405  {
406  }
407 
408  /* Write character in Transmit Data register.
409  TXE flag is cleared by writing data in TDR register */
410  LL_USART_TransmitData8(USART1, *pchar++);
411  }
412 
413  /* Wait for TC flag to be raised for last char */
414  while (!LL_USART_IsActiveFlag_TC(USART1))
415  {
416  }

448 void USART_CharReception_Callback(void)
449 {
450  uint8_t *ptemp;
451 
452  /* Read Received character. RXNE flag is cleared by reading of RDR register */
453  pBufferReadyForReception[uwNbReceivedChars++] = LL_USART_ReceiveData8(USART1);
454 
455  /* Checks if Buffer full indication has been set */
456  if (uwNbReceivedChars >= RX_BUFFER_SIZE)
457  {
458  /* Set Buffer swap indication */
459  uwBufferReadyIndication = 1;
460 
461  /* Swap buffers for next bytes to be received */
462  ptemp = pBufferReadyForUser;
463  pBufferReadyForUser = pBufferReadyForReception;
464  pBufferReadyForReception = ptemp;
465  uwNbReceivedChars = 0;
466  }
467 }

474 void Error_Callback(void)
475 {
476  __IO uint32_t isr_reg;
477 
478  /* Disable USARTx_IRQn */
479  NVIC_DisableIRQ(USART1_IRQn);
480 
481  /* Error handling example :
482  - Read USART ISR register to identify flag that leads to IT raising
483  - Perform corresponding error handling treatment according to flag
484  */
485  isr_reg = LL_USART_ReadReg(USART1, ISR);
486  if (isr_reg & LL_USART_ISR_NE)
487  {
488  /* case Noise Error flag is raised : Clear NF Flag */
489  LL_USART_ClearFlag_NE(USART1);
490  }
491  else
492  {
493  /* Unexpected IT source */
494  Error_Handler();
495  }
496 }
