544 void WaitAndCheckEndOfTransfer(void)
545 {
546  /* Wait end of transmission */
547  while (ubTransmissionComplete != 1)
548  {
549  }
550  ubTransmissionComplete = 0;
551 
552  /* Disable DMA1 Tx Channel */
553  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_7);
554 
555  /* Wait end of reception */
556  while (ubReceptionComplete != 1)
557  {
558  }
559  /* Disable DMA1 Rx Channel */
560  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_6);
561 
562  /* Compare received string to expected one */
563  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, ubNbDataToReceive))
564  {
565  /* Processing Error */
566  LED_Blinking(LED_BLINK_ERROR);
567  }
568  else
569  {
570  /* Turn On Led if data are well received */
571  LED_On();
572  }
573 }
