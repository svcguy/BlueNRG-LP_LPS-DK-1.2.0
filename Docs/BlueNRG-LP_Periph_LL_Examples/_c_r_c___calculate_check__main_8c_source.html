349 uint32_t Calculate_CRC(uint32_t BufferSize)
350 {
351  register uint32_t data = 0;
352  register uint32_t index = 0;
353 
354  /* Compute the CRC of Data Buffer array*/
355  for (index = 0; index < (BufferSize / 4); index++)
356  {
357  data = (uint32_t)((aDataBuffer[4 * index + 3] << 24) | (aDataBuffer[4 * index + 2] << 16) | (aDataBuffer[4 * index + 1] << 8) | aDataBuffer[4 * index]);
358  LL_CRC_FeedData32(CRC, data);
359  }
360 
361  /* Last bytes specific handling */
362  if ((BUFFER_SIZE % 4) != 0)
363  {
364  if (BUFFER_SIZE % 4 == 1)
365  {
366  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index]);
367  }
368  if (BUFFER_SIZE % 4 == 2)
369  {
370  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
371  }
372  if (BUFFER_SIZE % 4 == 3)
373  {
374  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
375  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index + 2]);
376  }
377  }
378  /* Return computed CRC value */
379  return(LL_CRC_ReadData32(CRC));
380 }
