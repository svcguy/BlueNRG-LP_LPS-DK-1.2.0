460 void WaitForUserButtonPress(void)
461 {
462  while(ubMessage<sizeof(aStringMessage))
463  {
464  /* Wait for TXE flag to be raised */
465  while (!LL_USART_IsActiveFlag_TXE(USART1))
466  {
467  }
468 
469  /* If last char to be sent, clear TC flag */
470  if (ubSend == (sizeof(aStringMessage) - 1))
471  {
472  LL_USART_ClearFlag_TC(USART1);
473  }
474  LL_USART_TransmitData8(USART1, aStringMessage[ubMessage++]);
475  }
476  while (ubButtonPress == 0)
477  {
478  LL_GPIO_TogglePin(LED2_GPIO_PORT, LED2_PIN);
479  LL_mDelay(LED_BLINK_FAST);
480  }
481  ubSend = 0;
482  ubMessage = 0;
483  ubButtonPress =0;
484 }

491 void BufferTransfer(void)
492 {
493  /* Send characters one per one, until last char to be sent */
494  while (ubSend < sizeof(aStringToSend))
495  {
496  /* Wait for TXE flag to be raised */
497  while (!LL_USART_IsActiveFlag_TXE(USART1))
498  {
499  }
500 
501  /* If last char to be sent, clear TC flag */
502  if (ubSend == (sizeof(aStringToSend) - 1))
503  {
504  LL_USART_ClearFlag_TC(USART1);
505  }
506 
507  /* Write character in Transmit Data register.
508  TXE flag is cleared by writing data in TDR register */
509  LL_USART_TransmitData8(USART1, aStringToSend[ubSend++]);
510  }
511 
512  /* Wait for TC flag to be raised for last char */
513  while (!LL_USART_IsActiveFlag_TC(USART1))
514  {
515  }
516 
517  /* Turn LED2 on at end of transfer : Tx sequence completed successfully */
518  LED_On();
519 }
