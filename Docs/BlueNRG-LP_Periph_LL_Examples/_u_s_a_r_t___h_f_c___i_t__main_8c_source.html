437 void USART_CharReception_Callback(void)
438 {
439  __IO uint32_t received_char;
440 
441  /* Read Received character. RXNE flag is cleared by reading of RDR register */
442  received_char = LL_USART_ReceiveData8(USART1);
443 
444  /* Check if received value is corresponding to specific one : S or s */
445  if ((received_char == 'S') || (received_char == 's'))
446  {
447  /* Turn LED2 on/off : Expected character has been received */
448  LED_Toggle();
449  }
450 
451  /* Echo received character on TX */
452  LL_USART_TransmitData8(USART1, received_char);
453 }

516 void Error_Callback(void)
517 {

519 
520  /* Disable USARTx_IRQn */

522 
523  /* Error handling example :

526  */
527  isr_reg = LL_USART_ReadReg(USART1, ISR);
528  if (isr_reg & LL_USART_ISR_NE)
529  {
530  /* case Noise Error flag is raised : ... */
531  LED_Blinking(LED_BLINK_FAST);
