314 static void MX_RTC_Init(void)
315 {
316  LL_RTC_InitTypeDef RTC_InitStruct = {0};
317  LL_RTC_TimeTypeDef RTC_TimeStruct = {0};
318  LL_RTC_DateTypeDef RTC_DateStruct = {0};
319 
320  /* Check if Data stored in BackUp register1: No Need to reconfigure RTC */
321  /* Read the Back Up Register 1 Data */
322  if (LL_RTC_BAK_GetRegister(RTC, LL_RTC_BKP_DR1) != RTC_BKP_DATE_TIME_UPDTATED)
323  {
324  printf("Check Data stored in BackUp register1: configuration RTC started\n\r");
325  /* Force RTC peripheral reset */
326  /* This action delete the calendar configuration in case of hardware reset */
327 
328  /* Enable Peripheral Clock */
329  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
330 
331  /* RTC peripheral reset */
332  LL_RCC_ClearFlag_RTCRSTREL();
333  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
334  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
335  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
336  LL_RCC_ClearFlag_RTCRSTREL();
337 
338  /* Disable the write protection for RTC registers */
339  LL_RTC_DisableWriteProtection(RTC);
340 
341  /* Init mode setup */
342  LL_RTC_EnableInitMode(RTC);
343 
344  /* Wait till the Init mode is active */
345  while(LL_RTC_IsActiveFlag_INIT(RTC) == RESET);
346 
347  /* Enter in initialization mode */
348  if (Enter_RTC_InitMode() != RTC_ERROR_NONE)
349  {
350  /* Initialization Error */
351  LED_Blinking(LED_BLINK_ERROR);
352  }
353 
354  /* Configure RTC */
355  RTC_InitStruct.HourFormat = LL_RTC_HOURFORMAT_24HOUR;
356  RTC_InitStruct.AsynchPrescaler = RTC_ASYNCH_PREDIV;
357  RTC_InitStruct.SynchPrescaler = RTC_SYNCH_PREDIV;
358  LL_RTC_Init(RTC, &RTC_InitStruct);
359 
360  /* Configure the Time */
361  RTC_TimeStruct.TimeFormat = LL_RTC_TIME_FORMAT_AM_OR_24;
362  RTC_TimeStruct.Hours = 0x23;
363  RTC_TimeStruct.Minutes = 0x59;
364  RTC_TimeStruct.Seconds = 0x50;
365  LL_RTC_TIME_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_TimeStruct);
366  /* Configure the Date */
367  RTC_DateStruct.WeekDay = LL_RTC_WEEKDAY_FRIDAY;
368  RTC_DateStruct.Month = LL_RTC_MONTH_DECEMBER;
369  RTC_DateStruct.Day = 0x31;
370  RTC_DateStruct.Year = 0x19;
371  LL_RTC_DATE_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_DateStruct);
372 
373  /* Exit of initialization mode */
374  if (Exit_RTC_InitMode() != RTC_ERROR_NONE)
375  {
376  /* Initialization Error */
377  LED_Blinking(LED_BLINK_ERROR);
378  }
379 
380  /* Enable RTC registers write protection */
381  LL_RTC_EnableWriteProtection(RTC);
382 
383  /* Writes a data in a RTC Backup data Register1 */
384  LL_RTC_BAK_SetRegister(RTC, LL_RTC_BKP_DR1, RTC_BKP_DATE_TIME_UPDTATED);
385  }
386  else
387  {
388  printf("Check Data stored in BackUp register1: No Need to reconfigure RTC\n\r");
389  printf("** Test successfully. ** \n\r\n\r");
390  }
391 }
