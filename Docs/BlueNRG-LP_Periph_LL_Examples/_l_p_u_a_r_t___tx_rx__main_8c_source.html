248 int main(void)
249 {
250  /* System initialization function */
251  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
252  {
253  /* Error during system clock configuration take appropriate action */
254  while(1);
255  }
256 
257  /* Set systick to 1ms using system clock frequency */
258  LL_Init1msTick(SystemCoreClock);
259 
260  /* Initialize LED2 */
261  LED_Init();
262 
263  /* LED off */
264  LED_Off();
265 
266  /* Configure LPUART1 (LPUART IP configuration and related GPIO initialization) */
267  Configure_LPUART1();
268 
269  /* Start main program loop :
270  - Wait for any character received on LPUART RX line
271  */
272 
273  /* Send Text Information on LPUART TX to PC Com port */
274  PrintInfo();
275 
276  /* Infinite loop */
277  while (1)
278  {
279  if(newCharReceived == 1)
280  {
281  /* Check if received value is corresponding to specific one : S or s */
282  if((ubReceivedChar == 'S') || (ubReceivedChar == 's'))
283  {
284  /* Turn LED2 on : Expected character has been received */
285  LED_On();
286  }
287  else
288  {
289  /* Turn LED2 off */
290  LED_Off();
291  }
292 
293  /* Echo received character on TX */
294  LL_LPUART_TransmitData8(LPUART1, ubReceivedChar);
295 
296  /* Wait for TC flag to be raised for last char */
297  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
298  {
299  }
300 
301  newCharReceived = 0;
302  }
303  }

320 void Configure_LPUART1(void)
321 {

323 
324  /* Enable the peripheral clock of GPIOB */
325  LL_EnableClock_LPUART();
326  LL_EnableClock_LPUART_TX_PORT();
327  LL_EnableClock_LPUART_RX_PORT();
328 
329  /* Configure TX Pin as : Alternate function, High Speed, PushPull, No-Pull */
330  LL_GPIO_SetPinMode(LPUART_TX_PORT, LPUART_TX_PIN, LL_GPIO_MODE_ALTERNATE);
331  LL_GPIO_SetAFPin_LPUART_TX();
332  LL_GPIO_SetPinSpeed(LPUART_TX_PORT, LPUART_TX_PIN, LL_GPIO_SPEED_FREQ_HIGH);
333  LL_GPIO_SetPinOutputType(LPUART_TX_PORT, LPUART_TX_PIN, LL_GPIO_OUTPUT_PUSHPULL);
334  LL_GPIO_SetPinPull(LPUART_TX_PORT, LPUART_TX_PIN, LL_GPIO_PULL_NO);

336  /* Configure RX Pin as : Alternate function, High Speed, PushPull, No-Pull */
337  LL_GPIO_SetPinMode(LPUART_RX_PORT, LPUART_RX_PIN, LL_GPIO_MODE_ALTERNATE);
338  LL_GPIO_SetAFPin_LPUART_RX();
339  LL_GPIO_SetPinSpeed(LPUART_RX_PORT, LPUART_RX_PIN, LL_GPIO_SPEED_FREQ_HIGH);
340  LL_GPIO_SetPinOutputType(LPUART_RX_PORT, LPUART_RX_PIN, LL_GPIO_OUTPUT_PUSHPULL);
341  LL_GPIO_SetPinPull(LPUART_RX_PORT, LPUART_RX_PIN, LL_GPIO_PULL_NO);
342 
343  /* (2) NVIC Configuration for LPUART1 interrupts */
344  /* - Set priority for LPUART1_IRQn */
345  /* - Enable LPUART1_IRQn */
346  NVIC_SetPriority(LPUART1_IRQn, IRQ_HIGH_PRIORITY);
347  NVIC_EnableIRQ(LPUART1_IRQn);

387 void PrintInfo(void)
388 {
389  uint32_t index = 0;
390 
391  /* Send characters one per one, until last char to be sent */
392  for (index = 0; index < sizeof(aTextInfo); index++)
393  {
394  /* Wait for TXE flag to be raised */
395  while (!LL_LPUART_IsActiveFlag_TXE(LPUART1))
396  {
397  }
398  /* Write character in Transmit Data register.
399  TXE flag is cleared by writing data in TDR register */
400  LL_LPUART_TransmitData8(LPUART1, aTextInfo[index]);
401  }
402 
403  /* Wait for TC flag to be raised for last char */
404  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
405  {
406  }
407 }
