382 void RTC_WakeupInit(void)
383 {
384  LL_RTC_InitTypeDef RTC_InitStruct = {0};
385  LL_RTC_TimeTypeDef RTC_TimeStruct = {0};
386  LL_RTC_DateTypeDef RTC_DateStruct = {0};
387 
388  /* Enable Peripheral Clock */
389  if (LL_RTC_BAK_GetRegister(RTC, LL_RTC_BKP_DR1) != RTC_BKP_DATE_TIME_UPDTATED)
390  {
391  /* Enable Peripheral Clock */
392  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
393 
394  /* RTC peripheral reset */
395  LL_RCC_ClearFlag_RTCRSTREL();
396  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
397  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
398  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
399  LL_RCC_ClearFlag_RTCRSTREL();
400 
401  /* Disable the write protection for RTC registers */
402  LL_RTC_DisableWriteProtection(RTC);
403 
404  /* Enter in initialization mode */
405  if (Enter_RTC_InitMode() != RTC_ERROR_NONE)
406  {
407  /* Initialization Error */
408  LED_Blinking(LED_BLINK_ERROR);
409  }
410  RTC_InitStruct.HourFormat = LL_RTC_HOURFORMAT_24HOUR;
411  RTC_InitStruct.AsynchPrescaler = RTC_ASYNCH_PREDIV;
412  RTC_InitStruct.SynchPrescaler = RTC_SYNCH_PREDIV;
413  LL_RTC_Init(RTC, &RTC_InitStruct);
414 
415  RTC_TimeStruct.TimeFormat = LL_RTC_TIME_FORMAT_AM_OR_24;
416  RTC_TimeStruct.Hours = 0x23;
417  RTC_TimeStruct.Minutes = 0x59;
418  RTC_TimeStruct.Seconds = 0x50;
419  LL_RTC_TIME_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_TimeStruct);
420  RTC_DateStruct.WeekDay = LL_RTC_WEEKDAY_FRIDAY;
421  RTC_DateStruct.Month = LL_RTC_MONTH_DECEMBER;
422  RTC_DateStruct.Day = 0x31;
423  RTC_DateStruct.Year = 0x19;
424 
425  LL_RTC_DATE_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_DateStruct);
426 
427  /* Output disabled */
428  LL_RTC_SetAlarmOutEvent(RTC, LL_RTC_ALARMOUT_DISABLE);
429 
430  /* Output polarity */
431  LL_RTC_SetOutputPolarity(RTC, LL_RTC_OUTPUTPOLARITY_PIN_HIGH);
432 
433  /* Exit of initialization mode */
434  if (Exit_RTC_InitMode() != RTC_ERROR_NONE)
435  {
436  /* Initialization Error */
437  LED_Blinking(LED_BLINK_ERROR);
438  }
439 
440  /* Enable RTC registers write protection */
441  LL_RTC_EnableWriteProtection(RTC);
442 
443  /* Writes a data in a RTC Backup data Register1 */
444  LL_RTC_BAK_SetRegister(RTC, LL_RTC_BKP_DR1, RTC_BKP_DATE_TIME_UPDTATED);
445  }
446 }

448 void SetRTC_WakeupTimeout(uint32_t time)
449 {
450  /* Disable write protection */
451  LL_RTC_DisableWriteProtection(RTC);
452 
453  /* Disable Wake-up Timer */
454  LL_RTC_WAKEUP_Disable(RTC);
455 
456  /* In case of interrupt mode is used, the interrupt source must disabled */
457  LL_RTC_DisableIT_WUT(RTC);
458 
459  /* Wait till RTC WUTWF flag is set */
460  while(LL_RTC_IsActiveFlag_WUTW(RTC) == 0);
461 
462 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
463  /* Clear PWR wake up Flag */
464  LL_PWR_ClearWakeupSource(LL_PWR_EWS_INT);
465 #endif
466 
467  /* Clear RTC Wake Up timer Flag */
468  LL_RTC_ClearFlag_WUT(RTC);
469 
470  /* Configure the Wake-up Timer counter */
471  LL_RTC_WAKEUP_SetAutoReload(RTC, time);
472 
473  /* Configure the clock source */
474  LL_RTC_WAKEUP_SetClock(RTC, LL_RTC_WAKEUPCLOCK_CKSPRE);
475 
476  /* Configure the Interrupt in the RTC_CR register */
477  LL_RTC_EnableIT_WUT(RTC);
478 
479  /* Enable the Wake-up Timer */
480  LL_RTC_WAKEUP_Enable(RTC);
481 
482  /* Enable write protection */
483  LL_RTC_EnableWriteProtection(RTC);
484 
485  /* Configure NVIC for RTC */
486  NVIC_SetPriority(RTC_IRQn, IRQ_LOW_PRIORITY);
487  NVIC_EnableIRQ(RTC_IRQn);
488 
489  /* Clear RTC Wake Up timer Flag */
490  LL_RTC_ClearFlag_WUT(RTC);
491 }

510 int main(void)

527 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)

531 
532  /* Initialization of COM port */
533  BSP_COM_Init(NULL);
534 
535  /* Init LED2 */
536  BSP_LED_Init(BSP_LED2);
537 
538  /* Init BUTTON 1 */
539  BSP_PB_Init(BSP_PUSH1, BUTTON_MODE_EXTI);
540 
541  /* RTC Wakeup Peripheral Init */
542  RTC_WakeupInit();
543 
544  while(1) {
545  /* Display the updated Time and Date */
546  Show_RTC_Calendar();
547 
548 #if defined(STEVAL_IDB011V1)
549  TogglePinsOutputDeepstop();
550 #endif /* STEVAL_IDB011V1 */
551 
552  /* POWER_SAVE_LEVEL_STOP_WITH_TIMER : wake on UART (PA8)/timeout 5 sec, (RTC)/button PUSH1 (PA10) */

554  /* Display time Format : hh:ms:ss */

559  printf("\n\r");

561  printf(" Date : ");
562  printf("%.2d-%.2d-%.2d", __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetMonth(RTC)),
563  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetDay(RTC)),
564  2000 + __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetYear(RTC)));
565  printf("\n\r");
566 
567  while(BSP_COM_UARTBusy());
568  wakeupIO.RTC_enable = 1;
569  wakeupIO.LPU_enable = 0;
570 #if defined(STEVAL_IDB011V1)
571  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
572  wakeupIO.IO_Mask_Low_polarity = WAKEUP_PA8;
573 #endif
574 #if defined(STEVAL_IDB012V1)
575  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
576  wakeupIO.IO_Mask_Low_polarity = WAKEUP_PB0;
577 #endif
578  SetRTC_WakeupTimeout(WAKEUP_TIMEOUT);
579  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
580 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
581  if (ret_val != SUCCESS)
582  printf("Error during clock config 0x%2x\r\n", ret_val);
583  PrintNegitatedLevel(stopLevel);
584  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
585  wakeupSources = HAL_PWR_MNGR_WakeupSource();
586  PrintWakeupSource(wakeupSources);
587  }
588 #endif
589 
590  }
591 }
