230 void Configure_TIMPWMOutput(void)
231 {
232  /*************************/
233  /* GPIO AF configuration */
234  /*************************/
235  /* Enable the peripheral clock of GPIOs */
236  LL_EnableClock_TIMx_CH4();
237 
238  /* GPIO TIMx_CH1 configuration */
239  LL_GPIO_SetPinMode(TIMx_CH4_PORT, TIMx_CH4_PIN, LL_GPIO_MODE_ALTERNATE);
240  LL_GPIO_SetPinPull(TIMx_CH4_PORT, TIMx_CH4_PIN, LL_GPIO_PULL_UP);
241  LL_GPIO_SetPinSpeed(TIMx_CH4_PORT, TIMx_CH4_PIN, LL_GPIO_SPEED_FREQ_HIGH);
242  LL_GPIO_SetAFPin_TIMx_CH4();
243 
244  /******************************/
245  /* Peripheral clocks enabling */
246  /******************************/
247  /* Enable the timer peripheral clock */
248  LL_EnableClock_TIMx();
249 
250  /* Set the pre-scaler value to have TIMx counter clock equal to 10 Hz */
251  LL_TIM_SetPrescaler(TIMx, __LL_TIM_CALC_PSC(LL_TIM_GetPeriphClock(TIMx), 10));
252 
253  /* Set the auto-reload value to have a counter frequency of 5 Hz */
254  /* TIMxCLK = LL_TIM_GetPeriphClock(TIMx) / (APB prescaler & multiplier) */
255  LL_TIM_SetAutoReload( TIMx, __LL_TIM_CALC_ARR(LL_TIM_GetPeriphClock(TIMx), LL_TIM_GetPrescaler(TIMx), 5));
256 
257  /* Set output mode */
258  LL_TIM_OC_SetMode(TIMx, LL_TIM_CHANNEL_CH4, LL_TIM_OCMODE_PWM1);
259 
260  /* Set output channel polarity */
261  /* Reset value is LL_TIM_OCPOLARITY_HIGH */
262  LL_TIM_OC_SetPolarity(TIMx, LL_TIM_CHANNEL_CH4, LL_TIM_OCPOLARITY_HIGH);
263 
264  /* Set compare value to half of the counter period (50% duty cycle ) */
265  LL_TIM_OC_SetCompareCH4(TIMx, ((LL_TIM_GetAutoReload(TIMx) + 1 ) * 0.50));
266 
267  /* Repetition counter to update the timer registers only after a given number of cycles of the counter */
268  LL_TIM_SetRepetitionCounter(TIMx, 0x2); // 3 cycles = 3 - 1 = 2 = TIMx_RCR
269 
270  /* Set the one pulse mode : Counter stops counting at the next update event */
271  LL_TIM_SetOnePulseMode(TIMx, LL_TIM_ONEPULSEMODE_SINGLE);
272 
273  /* To be sure that the pulse will be repeate N times an update event must occur only */
274  /* when the repetition counter is reloaded with the content of TIMx_RCR register */
275 
276  /* Enable output channel */
277  LL_TIM_CC_EnableChannel(TIMx, LL_TIM_CHANNEL_CH4);
278 
279  /* Enable the outputs (set the MOE bit in TIMx_BDTR register). */
280  LL_TIM_EnableAllOutputs(TIMx);
281 
282  /* Force update generation */
283  LL_TIM_GenerateEvent_UPDATE(TIMx);
284 
285  /* Enable counter */
286  LL_TIM_EnableCounter(TIMx);
287 }
