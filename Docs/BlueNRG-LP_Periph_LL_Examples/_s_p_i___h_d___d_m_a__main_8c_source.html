650 void WaitAndCheckEndOfTransfer(void)
651 {
652  /* 1 - Wait end of transmission */
653  while (ubTransmissionComplete != 1)
654  {
655  }
656  /* Disable DMA1 Tx Channel */
657  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_3);
658  /* 2 - Wait end of reception */
659  while (ubReceptionComplete != 1)
660  {
661  }
662  /* Disable DMA Rx Channel */
663  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_1);
664  /* 3 - Compare Transmit data to receive data */
665  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, sizeof(aTxBuffer)))
666  {
667  /* Processing Error */
668  printf("Processing Error.\n\r");
669  LED_Blinking(LED_BLINK_ERROR);
670  }
671  else
672  {
673  /* Turn On Led if data are well received */
674  printf("Data are well received.\n\r");
675  LED_On();
676  printf("** Test successfully. ** \n\r\n\r");
677  }
678 }
