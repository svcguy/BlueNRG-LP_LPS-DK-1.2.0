311 
312  /* Set the point P coordinate x */
313  LL_PKA_WriteOperand( 87, 8, (uint32_t*)&PKAStartPoint[0] );
314 
315  /* Set the point P coordinate y */

317 
318  while(LL_PKA_IsActiveFlag_BUSY(PKA));

321  LL_PKA_Start(PKA);
322
