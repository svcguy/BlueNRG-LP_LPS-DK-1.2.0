530 void WaitAndCheckEndOfTransfer(void)
531 {
532  /* Wait end of transmission */

534  {
535  }
536  ubTransmissionComplete = 0;
537 
538  /* Wait end of reception */

541  }
542  /* Disable DMA1 Rx Channel */
543  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_6);
544 
545  /* Compare received string to expected one */
546  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, ubNbDataToReceive))
