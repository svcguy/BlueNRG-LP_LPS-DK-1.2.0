372  NVIC_EnableIRQ(TIMx_IRQn);
373 
374  /******************************/
375  /* Peripheral clocks enabling */
376  /******************************/
377  /* Enable the timer peripheral clock */
378  LL_EnableClock_TIMx();
379 
380  /* Set the pre-scaler value */
381  tim_prescaler = __LL_TIM_CALC_PSC(LL_TIM_GetPeriphClock(TIMx), 10000);
382 
383  /* Set the auto-reload value to have a counter frequency of 10 Hz */
384  /* TIMxCLK = LL_TIM_GetPeriphClock(TIMx) / (APB prescaler & multiplier) */
385  tim_arr = __LL_TIM_CALC_ARR(LL_TIM_GetPeriphClock(TIMx), tim_prescaler, 10);
386 
387  /* Set output compare active/inactive delay to half of the auto-reload value */
388  tim_pulse_value = tim_arr / 2;
389 
390  TIM_InitStruct.Prescaler = tim_prescaler;
391  TIM_InitStruct.CounterMode = LL_TIM_COUNTERMODE_UP;
392  TIM_InitStruct.Autoreload = tim_arr;
393  TIM_InitStruct.ClockDivision = LL_TIM_CLOCKDIVISION_DIV1;
394  LL_TIM_Init(TIMx, &TIM_InitStruct);
395  LL_TIM_EnableARRPreload(TIMx);
396  TIM_OC_InitStruct.OCMode = LL_TIM_OCMODE_TOGGLE;
397  TIM_OC_InitStruct.OCState = LL_TIM_OCSTATE_DISABLE;
398  TIM_OC_InitStruct.OCNState = LL_TIM_OCSTATE_DISABLE;
399  TIM_OC_InitStruct.CompareValue = tim_pulse_value;
400  TIM_OC_InitStruct.OCPolarity = LL_TIM_OCPOLARITY_HIGH;

402  LL_TIM_OC_DisableFast(TIMx, LL_TIM_CHANNEL_CH1);

444 __STATIC_INLINE void Configure_OCMode(uint32_t OCMode)
445 {
446  /* Disable the counter */
447  LL_TIM_DisableCounter(TIMx);
448 
449  /* Reset the counter */
450  LL_TIM_SetCounter(TIMx, 0);
451 
452  /* Reset the compare match count */
453  uwCompareMatchCount = 0;
454 
455  /* Set the output level (active v.s. inactive) according to the new OC mode */
456  switch (OCMode)
457  {
458  case LL_TIM_OCMODE_TOGGLE:
459  /* Set the output channel to its toggles on compare match*/
460  printf("Set the output channel to its toggles on compare match.\n\r");
461  LL_TIM_OC_SetMode(TIMx, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_TOGGLE);
462  break;
463 
464  case LL_TIM_OCMODE_ACTIVE:
465  /* Set the output channel to its inactive level (LOW)*/
466  printf("Set the output channel to its inactive level (LOW).\n\r");
467  LL_TIM_OC_SetMode(TIMx, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_FORCED_INACTIVE);
468  break;
469 
470  case LL_TIM_OCMODE_INACTIVE:
471  /* Set the output channel to its active level (HIGH)*/
472  printf("Set the output channel to its active level (HIGH).\n\r");
473  LL_TIM_OC_SetMode(TIMx, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_FORCED_ACTIVE);
474  break;
475 
476  default:
477  break;
478  }
479 
480  /* Update the output channel mode */
481  LL_TIM_OC_SetMode(TIMx, LL_TIM_CHANNEL_CH1, OCMode);
482 
483  /* Re-enable the counter */
484  LL_TIM_EnableCounter(TIMx);
485 }
