246 int main(void)
247 {
248  float adcValue = 0.0;
249 
250  /* System initialization function */
251  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
252  {
253  /* Error during system clock configuration take appropriate action */
254  while(1);
255  }
256 
257 
258  /* BSP Init */
259  BSP_Init();
260 
261  /* Set systick to 1ms using system clock frequency */
262  LL_Init1msTick(SystemCoreClock);
263 
264  /* Initialize all configured peripherals */
265  APP_ADC_Init();
266 
267  /* Start ADC conversion */
268  LL_ADC_StartConversion(ADC);
269 
270  /* Infinite loop */
271  while (1) {
272 
273  /* Check the ADC flag End Of Down Sampler conversion */
274  if( LL_ADC_IsActiveFlag_EODS(ADC) == 1) {
275 
276  /* Get the raw value from the Down Sampler */
277  adcValue = (float)LL_ADC_GetADCConvertedValueSingle(ADC, LL_ADC_DSGetOutputData(ADC), LL_ADC_VIN_RANGE_3V6, USER_DATAWIDTH, offset_vinp0);
278  printf("ADC value %d.%03d mV\r\n", PRINT_INT(adcValue),PRINT_FLOAT(adcValue));
279 
280  /* Clear the ADC flag End Of Down Sampler conversion */
281  LL_ADC_ClearFlag_EODS(ADC);
282 
283  /* Toggle the conversion/activity LED */
284  BSP_LED_Toggle(BSP_LED1);
285 
286  /* Add 100 ms of delay between each measure */
287  LL_mDelay(100);
288 
289  /* Restart ADC conversion */
290  LL_ADC_StartConversion(ADC);
291  }
292 
293  if (LL_ADC_IsActiveFlag_AWD(ADC) == 1)
294  {
295  LL_ADC_ClearFlag_AWD(ADC);
296  /* Turn on the LED2 if AWD event occurs */
297  BSP_LED_On(BSP_LED2);
298  printf("AWD occurs\r\n");
299  }
300  else
301  {
302  LL_ADC_ClearFlag_AWD(ADC);
303  /* Turn off the LED2 if AWD event occurs */
304  BSP_LED_Off(BSP_LED2);
305  }
306  }
307 }

343 static void APP_ADC_Init(void)
344 {
345  /* Peripheral clock enable */
346  LL_APB1_EnableClock(LL_APB1_PERIPH_ADCDIG | LL_APB1_PERIPH_ADCANA);
347 
348  /* This function must not be called on QFN32 package */
349 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
350  LL_ADC_LDOEnable(ADC);
351 #endif
352 
353  /* Enable the ADC */
354  LL_ADC_Enable(ADC);
355 
356  /* Configure the sample rate */
357 #if defined(CONFIG_DEVICE_BLUENRG_LP)
358  LL_ADC_SetSampleRate(ADC, USER_SAMPLERATE);
359 #endif
360 #if defined(CONFIG_DEVICE_BLUENRG_LPS)
361  LL_ADC_SetSampleRate(ADC, USER_SAMPLERATE_MSB, USER_SAMPLERATE);
362 #endif
363 
364  /* Configure the Down Sampler data width and ratio */
365  LL_ADC_ConfigureDSDataOutput(ADC, USER_DATAWIDTH, USER_RATIO);
366 
367  /* Configure the operation mode as ADC mode (static/low frequency signal) */
368  LL_ADC_SetADCMode(ADC, LL_ADC_OP_MODE_ADC);
369 
370  /* Start the sampling at end of previous sample */
371  LL_ADC_InputSamplingMode(ADC, LL_ADC_SAMPLING_AT_END);
372 
373  /* Set the first (and only in this example) input of the conversion sequence as temperature sensor */
374  LL_ADC_SetChannelSeq0(ADC, LL_ADC_CH_VINP0_TO_SINGLE_POSITIVE_INPUT);
375 
376  /* Set the length of the conversion sequence as 1 */
377  LL_ADC_SetSequenceLength(ADC, LL_ADC_SEQ_LEN_01);
378 
379  LL_ADC_SetVoltageRangeSingleVinp0(ADC, LL_ADC_VIN_RANGE_3V6);
380 
381 
382  if(LL_ADC_GET_CALIB_GAIN_FOR_VINPX_3V6() != 0xFFF) {
383  LL_ADC_SetCalibPoint1Gain(ADC, LL_ADC_GET_CALIB_GAIN_FOR_VINPX_3V6() );
384 
385  offset_vinp0 = LL_ADC_GET_CALIB_OFFSET_FOR_VINPX_3V6();
386 #ifdef CONFIG_DEVICE_BLUENRG_LP
387  if(offset_vinp0 < -64 || offset_vinp0 > 63) {
388  LL_ADC_SetCalibPoint1Offset(ADC, 0);
389  }
390  else
391 #endif
392  {
393  LL_ADC_SetCalibPoint1Offset(ADC, offset_vinp0);
394  offset_vinp0 = 0;
395  }
396  }
397  else {
398  LL_ADC_SetCalibPoint1Gain(ADC, LL_ADC_DEFAULT_RANGE_VALUE_3V6);
399  }
400  LL_ADC_SetCalibPointForSinglePos3V6(ADC, LL_ADC_CALIB_POINT_1);
401 
402  /* Configure Analog watchdog */
403  LL_ADC_SetWDGInputChannels(ADC ,LL_ADC_WDG_CH_VINP0_TO_POS);
404  LL_ADC_ConfigureWDGThresholds(ADC, LL_ADC_ConvertVoltageToCodeForWDGThresholdSinglOrBatt(ADC_AWD_USER_LOW_THRESH, LL_ADC_VIN_RANGE_3V6),
405  LL_ADC_ConvertVoltageToCodeForWDGThresholdSinglOrBatt(ADC_AWD_USER_HIGH_THRESH, LL_ADC_VIN_RANGE_3V6));
406  LL_ADC_EnableIT_AWD(ADC);
407 
408 }
