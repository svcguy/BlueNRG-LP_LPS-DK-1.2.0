556 void GenerateRandomK16Word(uint32_t* buffer)
557 {
558  register uint8_t index = 0;
559 
560  /* Initialize random numbers generation */
561  LL_RNG_Enable(RNG);
562 
563  /* Generate Random 16bit Numbers */
564 
565 #if (USE_TIMEOUT == 1)
566  Timeout = RNG_GENERATION_TIMEOUT;
567 #endif /* USE_TIMEOUT */
568 
569  /* Wait for DRDY flag to be raised */
570  while (!LL_RNG_IsActiveFlag_RNGRDY(RNG))
571  {
572 #if (USE_TIMEOUT == 1)
573  /* Check Systick counter flag to decrement the time-out value */
574  if (LL_SYSTICK_IsActiveCounterFlag())
575  {
576  if(Timeout-- == 0)
577  {
578  /* Time-out occurred. Set LED to blinking mode */
579  LED_Blinking(LED_BLINK_SLOW);
580  }
581  }
582 #endif /* USE_TIMEOUT */
583  }
584 
585  /* Check if error occurs */
586  if ( LL_RNG_IsActiveFlag_FAULT(RNG) )
587  {
588  /* Clock or Seed Error detected. Set LED to blinking mode (Error type)*/
589  LED_Blinking(LED_BLINK_ERROR);
590  }
591 
592  /* Otherwise, no error detected : Value of generated random number could be retrieved
593  and stored in dedicated array */
594  buffer[index] = LL_RNG_ReadRandData16(RNG);
595 
596  /* Stop random numbers generation */
597  LL_RNG_Disable(RNG);
598 
599  /* Values of Generated Random numbers are now available in buffer array. */
600 }
