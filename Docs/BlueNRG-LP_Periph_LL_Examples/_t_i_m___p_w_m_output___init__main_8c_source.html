437 __STATIC_INLINE void Configure_DutyCycle(uint32_t D)
438 {
439  uint32_t P; /* Pulse duration */
440  uint32_t T; /* PWM signal period */
441 
442  /* PWM signal period is determined by the value of the auto-reload register */
443  T = LL_TIM_GetAutoReload(TIMx) + 1;
444 
445  /* Pulse duration is determined by the value of the compare register. */
446  /* Its value is calculated in order to match the requested duty cycle. */
447  P = (D*T)/100;
448  LL_TIM_OC_SetCompareCH1(TIMx, P);
449  printf("Changes the duty cycle of the PWM signal : %d \n\r", P);
450 }

472 void TimerCaptureCompare_Callback(void)
473 {
474  uint32_t CNT, ARR;
475  CNT = LL_TIM_GetCounter(TIMx);
476  ARR = LL_TIM_GetAutoReload(TIMx);
477 
478  if (LL_TIM_OC_GetCompareCH1(TIMx) > ARR )
479  {
480  /* If capture/compare setting is greater than autoreload, there is a counter overflow and counter restarts from 0.
481  Need to add full period to counter value (ARR+1) */
482  CNT = CNT + ARR + 1;
483  }
484  uwMeasuredDutyCycle = (CNT * 100) / ( ARR + 1 );
485 }
