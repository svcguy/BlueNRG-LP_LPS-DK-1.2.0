356 uint8_t Calculate_CRC(uint32_t BufferSize)
357 {
358  register uint32_t data = 0;
359  register uint32_t index = 0;
360 
361  /* Compute the CRC of Data Buffer array*/
362  for (index = 0; index < (BufferSize / 4); index++)
363  {
364  data = (uint32_t)((aDataBuffer[4 * index + 3] << 24) | (aDataBuffer[4 * index + 2] << 16) | (aDataBuffer[4 * index + 1] << 8) | aDataBuffer[4 * index]);
365  LL_CRC_FeedData32(CRC, data);
366  }
367 
368  /* Last bytes specific handling */
369  if ((BUFFER_SIZE % 4) != 0)
370  {
371  if (BUFFER_SIZE % 4 == 1)
372  {
373  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index]);
374  }
375  if (BUFFER_SIZE % 4 == 2)
376  {
377  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
378  }
379  if (BUFFER_SIZE % 4 == 3)
380  {
381  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
382  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index + 2]);
383  }
384  }
385 
386  /* Return computed CRC value */
387  return(LL_CRC_ReadData8(CRC));
388 }
