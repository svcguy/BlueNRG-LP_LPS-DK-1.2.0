245 int main(void)
246 {
247  /* System initialization function */
248  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
249  {
250  /* Error during system clock configuration take appropriate action */
251  while(1);
252  }
253 
254  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
255  LL_Init();
256 
257  /* Set systick to 1ms using system clock frequency */
258  LL_Init1msTick(SystemCoreClock);
259 
260 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
261  /* IO pull configuration with minimum power consumption */
262  BSP_IO_Init();
263 #endif
264 
265  /* Initialization of COM port */
266  BSP_COM_Init(Process_InputData);
267 
268  /* Initialize all configured peripherals */
269  MX_GPIO_Init();
270 
271  /* Reset LED_PIN */
272  LL_GPIO_ResetOutputPin(LED_GPIO_PORT, LED_PIN);
273 
274  /* Set the TIMx prescaler to get counter clock frequency at 10 MHz */
275  /* In this example TIMx input clock (TIMxCLK) is set to APB0 clock. */
276  /* => TIMxCLK = 64 MHz */
277  tim_swtrigger_prescaler = __LL_TIM_CALC_PSC(LL_TIM_GetPeriphClock(TIMx), 10000000);
278 
279  /* Set the autoreload register to get a pulse length of 50 us */
280  tim_swtrigger_period = __LL_TIM_CALC_PULSE(LL_TIM_GetPeriphClock(TIMx), tim_swtrigger_prescaler, 50, 50);
281 
282  /* Set the capture/compare register to get a pulse delay of 50 us */
283  tim_swtrigger_pulse = __LL_TIM_CALC_DELAY(LL_TIM_GetPeriphClock(TIMx), tim_swtrigger_prescaler, 50);
284 
285  MX_TIMx_Init();
286 
287  /**************************/
288  /* TIMx interrupts set-up */
289  /**************************/
290  /* Enable the capture/compare interrupt for channel 4 */
291  LL_TIM_EnableIT_CC4(TIMx);
292 
293  /**************************/
294  /* Start pulse generation */
295  /**************************/
296  /* Enable channel 4 */
297  LL_TIM_CC_EnableChannel(TIMx, LL_TIM_CHANNEL_CH4);
298 
299  /* Enable TIMx outputs */
300  LL_TIM_EnableAllOutputs(TIMx);
301 
302  /* Enable auto-reload register preload */
303  LL_TIM_EnableARRPreload(TIMx);
304 
305  // User push-button (PUSH1) or 'c'/'C' character is used to start TIMx counter.
306  printf("Wait for User push-button (PUSH1) press or enter 'c'/'C' character.\n\r");
307 
308  /* Infinite loop */
309  while (1)
310  {
311  if(updateMeasured == 1)
312  {
313  printf("uwMeasuredDelay = %d\n\r", uwMeasuredDelay);
314  printf("uwMeasuredPulseLength = %d\n\r", uwMeasuredPulseLength);
315  printf("\n\r");
316  updateMeasured =0;
317  LL_mDelay(10);
318  /* Reset LED_PIN */
319  LL_GPIO_ResetOutputPin(LED_GPIO_PORT, LED_PIN);
320  }
321  }
322 }

471 void TimerCaptureCompare_Callback(void)
472 {
473  uint32_t CNT;
474  uint32_t PSC;
475  uint32_t ARR;
476 
477  CNT = LL_TIM_GetCounter(TIMx);
478  PSC = LL_TIM_GetPrescaler(TIMx);
479  ARR = LL_TIM_GetAutoReload(TIMx);
480 
481  uwMeasuredDelay = (CNT * 1000000)/(LL_TIM_GetPeriphClock(TIMx)/(PSC + 1));
482  uwMeasuredPulseLength = ((ARR - CNT) * 1000000)/(LL_TIM_GetPeriphClock(TIMx)/(PSC + 1));
483  updateMeasured =1;
484 }
