365 void UserButton_Callback(void)
366 {
367  register uint32_t source = 0, prescaler = 0;
368 
369  /* Get the MCO config to apply */
370  source = aMCO_Config[bMCOIndex].Source;
371  prescaler = aMCO_Config[bMCOIndex].Prescaler;
372 
373  /* Toggle LED2 to indicate a button press*/
374  LL_GPIO_TogglePin(LED2_GPIO_PORT, LED2_PIN);
375 
376  /* Update User push-button (PUSH1) variable : to be checked in waiting loop in main program */
377  ubButtonPress = 1;
378 
379  /* Select MCO clock source and prescaler */
380  LL_RCC_ConfigMCO(source, prescaler);
381 
382  /* Set new MCO config Index */
383  bMCOIndex = (bMCOIndex + 1) % RCC_MCO_CONFIG_NB;
384 }
