538 void WaitAndCheckEndOfTransfer(void)
539 {
540  /* Wait end of transmission */
541  while (ubTransmissionComplete != 1)
542  {
543  }
544  ubTransmissionComplete = 0;
545  /* Disable DMA1 Tx Channel */
546  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_7);
547 
548  /* Wait end of reception */
549  while (ubReceptionComplete != 1)
550  {
551  }
552  /* Disable DMA1 Rx Channel */
553  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_6);
554 
555  // Release the CS signal after RX is completed to have same behavior of the SPI
556  LL_GPIO_SetOutputPin(GPIO_PORT_MASTER_CS, GPIO_PIN_SPI_MASTER_CS);
557 
558  /* Compare received string to expected one */
559  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, ubNbDataToReceive))
560  {
561  /* Processing Error */
562  Error_Handler();
563  }
564  else
565  {
566  /* Turn On Led if data are well received */
567  BSP_LED_On(BSP_LED3);
568  }
569 }
