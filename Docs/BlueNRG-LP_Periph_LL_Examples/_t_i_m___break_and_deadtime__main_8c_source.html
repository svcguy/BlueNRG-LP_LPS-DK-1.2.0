265  /* Set the TIMx prescaler to get counter clock frequency at 10 MHz */
266  tim_prescaler = __LL_TIM_CALC_PSC(LL_TIM_GetPeriphClock(TIMx), 10000000);
267 
268  /* Set the TIMx auto-reload register to get a PWM frequency at 10 KHz */
269  /* Note that in macro call below, targeted PWM frequency must be multiplied */
270  /* by 2 when the counter operates in center-aligned mode (due to the */
271  /* symmetry of the pattern). */
272  tim_period = __LL_TIM_CALC_ARR(LL_TIM_GetPeriphClock(TIMx), tim_prescaler, 10000*2);
273 
274  /* Set PWM output channel 1 duty cycle to 50% */
275  pulse_value_1 = (tim_period * 500) / 1000;
276 
277  /* Set PWM output channel 2 duty cycle to 25% */
278  pulse_value_2 = (tim_period * 250) / 1000;
279 
280  /* Set dead time to 0,1 us (100 ns) */
281  tim_deadtime = __LL_TIM_CALC_DEADTIME(LL_TIM_GetPeriphClock(TIMx), 0, 100);
282 
283  /* Initialize all configured peripherals */
284  MX_GPIO_Init();
285  MX_TIMx_Init();
286 
287  /**********************************/
288  /* Start output signal generation */
289  /**********************************/
290  /* Enable outputs OC1, OC1N, OC2 */
291  LL_TIM_CC_EnableChannel(TIMx, LL_TIM_CHANNEL_CH1 |
292  LL_TIM_CHANNEL_CH1N );
293 
294 #ifdef LL_TIM_CHANNEL_CH2
295  /* Enable outputs OC2 */
296  LL_TIM_CC_EnableChannel(TIMx, LL_TIM_CHANNEL_CH2 );
297 #endif /* LL_TIM_CHANNEL_CH2 */
298 
299 #ifdef LL_TIM_CHANNEL_CH2N
300  /* Enable outputs OC2N */

302 #endif /* LL_TIM_CHANNEL_CH2N */
303 
304  /* Enable TIMx outputs */
305  LL_TIM_EnableAllOutputs(TIMx);
306 
307  /* Enable counter */

309
