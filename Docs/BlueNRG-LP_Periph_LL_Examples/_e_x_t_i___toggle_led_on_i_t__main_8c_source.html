270 void Configure_EXTI()
271 {
272  /* Enable the BUTTON Clock */
273  USER_BUTTON_GPIO_CLK_ENABLE();
274  USER_BUTTON_SYSCFG_CLK_ENABLE();
275 
276  /* Configure GPIO for BUTTON */
277  LL_GPIO_SetPinMode(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_MODE_INPUT);
278  LL_GPIO_SetPinPull(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_PULL_NO);
279 
280  /* Enable a rising trigger External line 10 Interrupt */
281  USER_BUTTON_EXTI_LINE_ENABLE();
282  USER_BUTTON_EXTI_RISING_TRIG_ENABLE();
283 
284  /* Clear the event occurred on the interrupt line 10 port A. */
285  if (LL_EXTI_IsInterruptPending(USER_BUTTON_EXTI_LINE) != RESET)
286  {
287  LL_EXTI_ClearInterrupt(USER_BUTTON_EXTI_LINE);
288  }
289 
290  /* Configure NVIC for USER_BUTTON_EXTI_IRQn */
291  NVIC_SetPriority(USER_BUTTON_EXTI_IRQn, IRQ_HIGH_PRIORITY);
292  NVIC_EnableIRQ(USER_BUTTON_EXTI_IRQn);
293 
294  /* Configure NVIC for SysTick_IRQn */
295  NVIC_SetPriority(SysTick_IRQn, IRQ_LOW_PRIORITY);
296 }

335 void LED_Init(void)

339

342  /* Reset value is LL_GPIO_OUTPUT_PUSHPULL */
343  LL_GPIO_SetPinOutputType(LED2_GPIO_PORT, LED2_PIN, LL_GPIO_OUTPUT_PUSHPULL);
344  /* Reset value is LL_GPIO_SPEED_FREQ_LOW */
345  LL_GPIO_SetPinSpeed(LED2_GPIO_PORT, LED2_PIN, LL_GPIO_SPEED_FREQ_HIGH);
346  /* Reset value is LL_GPIO_PULL_NO */
347  LL_GPIO_SetPinPull(LED2_GPIO_PORT, LED2_PIN, LL_GPIO_PULL_NO);
