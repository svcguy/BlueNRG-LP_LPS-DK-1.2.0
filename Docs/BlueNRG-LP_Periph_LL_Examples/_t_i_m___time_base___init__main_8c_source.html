252 int main(void)
253 {
254  /* System initialization function */
255  if (SystemInit(SYSCLK_64M, RADIO_SYSCLK_NONE) != SUCCESS)
256  {
257  /* Error during system clock configuration take appropriate action */
258  while(1);
259  }

262  LL_Init();

265  LL_Init1msTick(SystemCoreClock);

279  /* Clear the update flag */
280  LL_TIM_ClearFlag_UPDATE(TIMx);
281 
282  /* Enable the update interrupt */
283  LL_TIM_EnableIT_UPDATE(TIMx);
284 
285  /* Enable counter */
286  LL_TIM_EnableCounter(TIMx);
287 
288  /* Force update generation */
289  LL_TIM_GenerateEvent_UPDATE(TIMx);
290 
291  printf("The update event period can be changed. \n\r");
292  printf("Wait for User push-button (PUSH1) press or enter 'c'/'C' character.\n\r");
293 
294  /* Infinite loop */
295  while (1)
296  {
297  if( ubButtonPress == 1)
298  {
299  /* Change the update event period by modifying the autoreload value. */
300  /* In up-counting update event is generated at each counter overflow (when */
301  /* the counter reaches the auto-reload value). */
302  /* Update event period is calculated as follows: */
303  /* Update_event = TIMxCLK /((PSC + 1)*(ARR + 1)*(RCR + 1)) */
304  /* where TIMxCLK is 64 MHz */
305  AutoreloadMult = AutoreloadMult % TIM_BASE_FREQ_NB;
306  new_tim_period = tim_period * (AutoreloadMult +1);
307  LL_TIM_SetAutoReload(TIMx, new_tim_period);
308  printf("timebase period : %dms\n\r", new_tim_period/10);
309 
310  /* Force update generation */
311  LL_TIM_GenerateEvent_UPDATE(TIMx);
312 
313  AutoreloadMult = AutoreloadMult + 1;
314  ubButtonPress = 0;
315  printf("Wait for User push-button (PUSH1) press or enter 'c'/'C' character.\n\r");
316  }
317  }
318 }

332 static void MX_TIMx_Init(void)
333 {
334  LL_TIM_InitTypeDef TIM_InitStruct = {0};
335 
336  /* Set the pre-scaler value to have TIMx counter clock equal to 10 kHz */
337  tim_prescaler = __LL_TIM_CALC_PSC(LL_TIM_GetPeriphClock(TIMx), 10000);
338 
339  /* - Set the auto-reload value to have a counter frequency of 10 Hz */
340  tim_period = __LL_TIM_CALC_ARR(LL_TIM_GetPeriphClock(TIMx), tim_prescaler, 10);
341 
342  /* Peripheral clock enable */
343  LL_EnableClock(LL_PERIPH_TIMx);
344 
345  TIM_InitStruct.Prescaler = tim_prescaler;
346  TIM_InitStruct.CounterMode = LL_TIM_COUNTERMODE_UP;
347  TIM_InitStruct.Autoreload = tim_period;
348  TIM_InitStruct.ClockDivision = LL_TIM_CLOCKDIVISION_DIV1;
349  TIM_InitStruct.RepetitionCounter = 0;
350  LL_TIM_Init(TIMx, &TIM_InitStruct);
351 
352  /* TIMx interrupt Init */
353  NVIC_SetPriority(TIMx_IRQn, IRQ_MED_PRIORITY);
354  NVIC_EnableIRQ(TIMx_IRQn);
355 }

380  /* Enable the BUTTON Clock */
381  USER_BUTTON_GPIO_CLK_ENABLE();
382  USER_BUTTON_SYSCFG_CLK_ENABLE();
383 
384  /* Configure GPIO for BUTTON */
385  LL_GPIO_SetPinMode(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_MODE_INPUT);
386  LL_GPIO_SetPinPull(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_PULL_NO);
387 
388  /* Enable a rising trigger External line 10 Interrupt */
389  USER_BUTTON_EXTI_LINE_ENABLE();
390  USER_BUTTON_EXTI_RISING_TRIG_ENABLE();
391 
392  /* Clear the event occurred on the interrupt line 10 port A. */
393  if (LL_EXTI_IsInterruptPending(USER_BUTTON_EXTI_LINE) != RESET)
394  {
395  LL_EXTI_ClearInterrupt(USER_BUTTON_EXTI_LINE);
396  }
397

399  NVIC_SetPriority(USER_BUTTON_EXTI_IRQn, IRQ_HIGH_PRIORITY);
400  NVIC_EnableIRQ(USER_BUTTON_EXTI_IRQn);
