537 void WaitAndCheckEndOfTransfer(void)
538 {
539  /* 1 - Wait end of transmission */
540  while (ubTransmissionComplete != 1)
541  {
542  }
543  /* Disable DMA1 Tx Channel */
544  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_1);
545  /* 2 - Wait end of reception */
546  while (ubReceptionComplete != 1)
547  {
548  }
549  /* Disable DMA1 Rx Channel */
550  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_3);
551  /* 3 - Compare Transmit data to receive data */
552  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, ubNbDataToTransmit))
553  {
554  /* Processing Error */
555  printf("Processing Error.\n\r");
556  LED_Blinking(LED_BLINK_ERROR);
557  }
558  else
559  {
560  /* Turn On Led if data are well received */
561  LED_On();
562  printf("Data are well received.\n\r");
563  printf("** Test successfully. ** \n\r\n\r");
564  }
565 }
