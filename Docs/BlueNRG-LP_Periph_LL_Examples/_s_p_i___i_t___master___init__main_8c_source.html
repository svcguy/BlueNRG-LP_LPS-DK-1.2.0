555 void WaitAndCheckEndOfTransfer(void)
556 {
557  /* 1 - Wait end of transmission */
558  while (ubTransmitIndex < (ubNbDataToTransmit-1))

560  }
561 
562  /* 2 - Wait end of reception */

613 void SPI_MASTER_Rx_Callback(void)
614 {
615  /* Read character in Data register.
616  RXNE flag is cleared by reading data in DR register */
617 #if defined( CONFIG_DATASIZE_16BIT )
618  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData16(SPI_MASTER);
619 #elif defined( CONFIG_DATASIZE_8BIT )
620  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData8(SPI_MASTER);
621 #endif
622 
623  /* Check the end of reception */
624  if(ubNbDataToReceive == ubReceiveIndex)
625  {
626  /* Disable RXNE Interrupt */
627  LL_SPI_DisableIT_RXNE(SPI_MASTER);
628  while(LL_SPI_GetTxFIFOLevel(SPI_MASTER)!=LL_SPI_TX_FIFO_EMPTY);
629  while(LL_SPI_IsActiveFlag_BSY(SPI_MASTER));
630  LL_SPI_Disable(SPI_MASTER);
631  LL_GPIO_SetOutputPin(GPIO_PORT_MASTER_CS, GPIO_PIN_SPI_MASTER_CS);
632  }
633 }
