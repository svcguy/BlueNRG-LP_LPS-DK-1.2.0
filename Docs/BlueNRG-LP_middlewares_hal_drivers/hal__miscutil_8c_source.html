81 void HAL_CrashHandler(uint32_t msp, uint32_t signature)
82 {
83  volatile uint32_t * crash_info = (volatile uint32_t *)&CrashInfoRam;
84  register uint32_t reg_content;
85  /* Init to zero the crash_info RAM locations */
86  for (reg_content=0; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
87  crash_info[reg_content] = 0;
88  }
89  /* Store Crash Signature */
90  CrashInfoRam.signature = signature;
91  /* Store SP register */
92  CrashInfoRam.SP = msp;
93  for (reg_content=2; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
94  uint32_t *ptr = ((uint32_t *)msp)+(reg_content-2);
95  if ((ptr >= ((uint32_t *) _MEMORY_RAM_BEGIN_)) &&
96  (ptr <= ((uint32_t *) _MEMORY_RAM_END_)))
97  crash_info[reg_content] = *ptr;
98  }
99  NVIC_SystemReset();
100 }
