926 static NVMDB_status_t LoadDBToRAM(NVMDB_HandleType *handle_p, uint8_t *buff, uint16_t buff_size, uint16_t *num_read_bytes_p, ReadStateType *read_state)
927 {
928  NVMDB_status_t status;
929  uint16_t free_buffer_size;
930  NVMDB_RecordSizeType record_size, remaining_record_bytes;
931  NVMDB_RecordHeaderType header;
932  uint16_t index = 0;
933 
934  memset(buff, 0xFF, buff_size);
935 
936  header.valid_flag = VALID_RECORD;
937 
938  while(1)
939  {
940 
941  free_buffer_size = buff_size - index - RECORD_HEADER_SIZE;
942 
943  // Read the record
944  if(read_state->move_to_next_record)
945  {
946  /* Passing read_state->record_length to the function because we do not want
947  the function to retrieve the info from the header of the current record,
948  since it may not be anymore in the same position in flash. */
949  status = ReadNextRecordNoLock(handle_p, ALL_TYPES, 0, buff + index + RECORD_HEADER_SIZE, free_buffer_size, &record_size, read_state->record_length, &header.record_id);
950  read_state->record_length = record_size;
951 
952  if(status != NVMDB_STATUS_OK)
953  {
954  *num_read_bytes_p = index;
955  return status;
956  }
957 
958  // Prepare the header and copy it to buffer
959  header.length = record_size;
960  memcpy(buff + index, &header, RECORD_HEADER_SIZE);
961  index += ROUND4_R(header.length + RECORD_HEADER_SIZE);
962  }
963  else
964  {
965 
966  status = ReadCurrentRecordNoLock(handle_p, read_state->record_offset, buff + index, free_buffer_size, &record_size, read_state->record_length); // It will not read record in cache because this operation is not allowed if there are any.
967 
968  if(status != NVMDB_STATUS_OK)
969  {
970  // This should not happen
971  return status;
972  }
973  index = ROUND4_R(record_size - read_state->record_offset);
974  }
975 
976  // The number of remaining bytes for the current record that were to be read before calling last NVMDB_ReadNextRecord().
977  // Rounded to the next multiple of 4 to find out if buffer can now be written inside the page or not.
978  remaining_record_bytes = ROUND4_R(record_size - read_state->record_offset);
979 
980  if(remaining_record_bytes > free_buffer_size)
981  {
982  // Not able to read the entire record.
983  read_state->move_to_next_record = FALSE;
984  read_state->record_offset += free_buffer_size;
985  *num_read_bytes_p = buff_size;
986  return NVMDB_STATUS_OK;
987  }
988  else if(remaining_record_bytes == free_buffer_size)
989  {
990  read_state->move_to_next_record = TRUE;
991  read_state->record_offset = 0;
992  *num_read_bytes_p = buff_size;
993  return NVMDB_STATUS_OK;
994  }
995  else if(remaining_record_bytes < free_buffer_size)
996  {
997  read_state->move_to_next_record = TRUE;
998  read_state->record_offset = 0;
999  }
1000  }
1001 }
