391  /* Disable Wake-up Timer */

394  /* In case of interrupt mode is used, the interrupt source must disabled */
395  LL_RTC_DisableIT_WUT(RTC);
396 
397  /* Wait till RTC WUTWF flag is set */
398  while(LL_RTC_IsActiveFlag_WUTW(RTC) == 0);
399 
400  /* Clear PWR wake up Flag */
401  LL_PWR_ClearWakeupSource(LL_PWR_EWS_INT);
402 
403  /* Clear RTC Wake Up timer Flag */
404  LL_RTC_ClearFlag_WUT(RTC);
405 
406  /* Configure the Wake-up Timer counter */
407  LL_RTC_WAKEUP_SetAutoReload(RTC, time);
408

411 
412  /* Configure the Interrupt in the RTC_CR register */
413  LL_RTC_EnableIT_WUT(RTC);

415  /* Enable the Wake-up Timer */

520 int main(void)
521 {
522  uint8_t data, ret_val, tmp;
523  PartInfoType partInfo;
524  WakeupSourceConfig_TypeDef wakeupIO;
525  uint32_t wakeupSources;
526  PowerSaveLevels stopLevel;
527  HAL_VTIMER_InitType VTIMER_InitStruct;
528 
529  /* System initialization function */
530  if (SystemInit(SYSCLK_DIRECT_HSE, BLE_SYSCLK_16M) != SUCCESS) {
531  /* Error during system clock configuration take appropriate action */
532  while(1);
533  }
534 
535  /* Configure IOs for pwer save modes */
536  BSP_IO_Init();
537 
538  /* Init the UART peripheral */
539  BSP_COM_Init(BSP_COM_RxDataUserCb);
540 
541  /* Init LED1 */
542  BSP_LED_Init(BSP_LED1);
543 
544  /* Init BUTTON 1 */
545  BSP_PB_Init(BSP_PUSH1, BUTTON_MODE_EXTI);
546 
547  /* VTimer module Init */
548  VTIMER_InitStruct.XTAL_StartupTime = HS_STARTUP_TIME;
549  VTIMER_InitStruct.EnableInitialCalibration = INITIAL_CALIBRATION;
550  VTIMER_InitStruct. PeriodicCalibrationInterval = CALIBRATION_INTERVAL;
551  HAL_VTIMER_Init(&VTIMER_InitStruct);
552  timerHandle.callback = TimeoutCallback;
553 
554  /* RTC Wakeup Peripheral Init */
555  RTC_WakeupInit();
556 
557  printf("Power Manager FW demo!\r\nDigit ? for help command\r\n");
558 
559  while(1) {
560  /* To run the VTIMER state machine */
561  HAL_VTIMER_Tick();
562 
563  if (BSP_COM_Read(&data)) {
564  switch(data)
565  {
566  case 's':
567  {
568  /* SHUTDOWN LEVEL : the only wakeup source is a low pulse on the RSTN pad */
569  printf("Enable Power Save Request : SHUTDOWN\r\n");
570  while(BSP_COM_UARTBusy());
571  if (HAL_PWR_MNGR_ShutdownRequest(TRUE) != SUCCESS)
572  printf("ERORR during the SHUTDOWN Request!\r\n");
573  }
574  break;
575  case 't':
576  {
577  /* POWER_SAVE_LEVEL_STOP_WITH_TIMER : wake on UART RX pin/timeout 5 sec (VTIMER)/button PUSH1 (PA10) */
578  printf("Enable Power Save Request : STOP_WITH_TIMER (VTIMER)\r\n");
579  while(BSP_COM_UARTBusy());
580  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
581  wakeupIO.IO_Mask_Low_polarity = WAKEUP_UART_RX_PIN;
582  wakeupIO.RTC_enable = 0;
583  wakeupIO.LPU_enable = 0;
584  ret_val = HAL_VTIMER_StartTimerMs(&timerHandle, WAKEUP_TIMEOUT);
585  if (ret_val != SUCCESS) {
586  printf("HAL_VTIMER_StartTimerMs() error 0x%02x\r\n", ret_val);
587  while(1);
588  }
589  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
590  if (ret_val != SUCCESS)
591  printf("Error during clock config 0x%2x\r\n", ret_val);
592  PrintNegotiatedLevel(stopLevel);
593  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
594  wakeupSources = HAL_PWR_MNGR_WakeupSource();
595  PrintWakeupSource(wakeupSources);
596  }
597  }
598  break;
599  case 'z':
600  {
601  /* POWER_SAVE_LEVEL_STOP_WITH_TIMER : wake on UART RX pin/timeout 5 sec (RTC)/button PUSH1 (PA10) */
602  printf("Enable Power Save Request : STOP_WITH_TIMER (RTC)\r\n");
603  while(BSP_COM_UARTBusy());
604  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
605  wakeupIO.IO_Mask_Low_polarity = WAKEUP_UART_RX_PIN;
606  wakeupIO.RTC_enable = 1;
607  wakeupIO.LPU_enable = 0;
608  SetRTC_WakeupTimeout(WAKEUP_TIMEOUT);
609  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
610  if (ret_val != SUCCESS)
611  printf("Error during clock config 0x%2x\r\n", ret_val);
612  PrintNegotiatedLevel(stopLevel);
613  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
614  wakeupSources = HAL_PWR_MNGR_WakeupSource();
615  PrintWakeupSource(wakeupSources);
616  }
617  DisableRTC_WakeupTimeout();
618  }
619  break;
620  case 'u':
621  {
622  /* POWER_SAVE_LEVEL_STOP_WITH_TIMER : wake on LPUART/button PUSH1 (PA10) */
623 #if defined CONFIG_DEVICE_BLUENRG_LPS
624  Configure_LPUART();
625  printf("Enable Power Save Request : STOP_WITH_TIMER (LPUART)\r\n");
626  while(BSP_COM_UARTBusy());
627  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
628  wakeupIO.IO_Mask_Low_polarity = 0;
629  wakeupIO.RTC_enable = 0;
630  wakeupIO.LPU_enable = 1;
631  LL_PWR_EnablePDB(LL_PWR_PUPD_IO0);
632  LL_PWR_EnablePUA(LL_PWR_PUPD_IO1);
633  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
634  if (ret_val != SUCCESS)
635  printf("Error during clock config 0x%2x\r\n", ret_val);
636  PrintNegotiatedLevel(stopLevel);
637  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
638  wakeupSources = HAL_PWR_MNGR_WakeupSource();
639  PrintWakeupSource(wakeupSources);
640  }
641  Disable_LPUART();
642  if (wakeupSources == WAKEUP_LPU)
643  printf("Wakeup Character: %c\r\n", receivedChar);
644 #else
645  printf("Feature not supported!!!\r\n");
646 #endif
647  }
648  break;
649  case 'n':
650  {
651  /* POWER_SAVE_LEVEL_NOTIMER : wake on UART RX pin/button PUSH1 (PA10) */
652  printf("Enable Power Save Request : STOP_NOTIMER\r\n");
653  while(BSP_COM_UARTBusy());
654  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
655  wakeupIO.IO_Mask_Low_polarity = WAKEUP_UART_RX_PIN;
656  wakeupIO.RTC_enable = 0;
657  wakeupIO.LPU_enable = 0;
658  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_NOTIMER, wakeupIO, &stopLevel);
659  if (ret_val != SUCCESS)
660  printf("Error during clock config 0x%2x\r\n", ret_val);
661  PrintNegotiatedLevel(stopLevel);
662  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
663  wakeupSources = HAL_PWR_MNGR_WakeupSource();
664  PrintWakeupSource(wakeupSources);
665  }
666  }
667  break;
668  case 'c':
669  {
670  /* POWER_SAVE_LEVEL_CPU_HALT : wake on button PUSH1 (PA10) */
671  printf("Enable Power Save Request : CPU_HALT\r\n");
672  while(BSP_COM_UARTBusy());
673  __WFI();
674  printf("Exit from WFI cortex instruction with interrupt\r\n");
675  /* Empty the USART buffer */
676  while(BSP_COM_Read(&tmp));
677  }
678  break;
679  case 'l':
680  {
681  /* Toggle led LED1 */
682  BSP_LED_Toggle(BSP_LED1);
683  }
684  break;
685  case 'p':
686  {
687  /* Print Hello World */
688  HAL_GetPartInfo(&partInfo);
689 #if defined CONFIG_DEVICE_BLUENRG_LPS
690  printf("Hello World: BlueNRG-LPS (%d.%d) is here!\r\n",
691  partInfo.die_major,
692  partInfo.die_cut);
693 #endif
694 #if defined CONFIG_DEVICE_BLUENRG_LP
695  printf("Hello World: BlueNRG-LP (%d.%d) is here!\r\n",
696  partInfo.die_major,
697  partInfo.die_cut);
698 #endif
699  }
700  break;
701  case 'r':
702  {
703  /* Reset BlueNRG-LP */
704  NVIC_SystemReset();
705  }
706  break;
707  case '?':
708  {
709  /* Help command */
710  help();
711  }
712  break;
713  default:
714  printf("UNKNWON COMMAND! Press ? for command list\r\n");
715  }
716  }
717  }
