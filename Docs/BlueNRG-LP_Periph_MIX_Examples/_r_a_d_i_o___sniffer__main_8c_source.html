254 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
255 {
256  if( (p->status & BLUE_INTERRUPT1REG_DONE) != 0) {
257 
258  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
259 
260  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
261  /* Check if there is enough space in the FIFO to store the incoming data */
262  if(MAX_PACKET_LENGTH*2 - fifo_size(&blueRec_fifo) > (p->data[1]+ 4 + 4 + 4)){
263  fifo_put_var_len_item(&blueRec_fifo, p->data[1]+2, p->data,0,NULL);
264  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->rssi));
265  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->timestamp_receive));
266  }
267  HAL_RADIO_ReceivePacket(channel, delay, receivedData, timeOut, MAX_LL_PACKET_LENGTH, RxCallback);
268  }
269  else if((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
270  HAL_RADIO_ReceivePacket(channel,delay, receivedData, timeOut, MAX_LL_PACKET_LENGTH, RxCallback);
271  }
272  else if((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) {
273  HAL_RADIO_ReceivePacket(channel, delay, receivedData, timeOut, MAX_LL_PACKET_LENGTH, RxCallback);
274  }
275  }
276  }
277  return TRUE;
278 }

280 int main(void)
281 {
282  HAL_VTIMER_InitType VTIMER_InitStruct = {HS_STARTUP_TIME, INITIAL_CALIBRATION, CALIBRATION_INTERVAL};
283  uint8_t packet[MAX_PACKET_LENGTH];
284  uint16_t length;
285 
286  /* System initialization function */
287  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS) {
288  /* Error during system clock configuration take appropriate action */
289  while(1);
290  }
291 
292  /* Enable UART: 115200-8bit-No Parity-1 Stop bit */
293  BSP_COM_Init(NULL);
294 
295  /* Create the blueRec FIFO */
296  fifo_init(&blueRec_fifo,MAX_PACKET_LENGTH*2, blueRec_buffer, 1);
297 
298  RADIO_Init();
299  HAL_VTIMER_Init(&VTIMER_InitStruct);
300 
301  /* Set the Network ID */
302  HAL_RADIO_SetNetworkID(BLE_ADV_ACCESS_ADDRESS); /* 0x8E89BED6 */
303 
304  printf("\r\nSniffing channel: %d, Network_ID: 0x%08X\r\n", channel, BLE_ADV_ACCESS_ADDRESS);
305 
306 #ifdef PACKET_ENCRYPTION
307  /* Set Encryption key and Initial vector */
308  RADIO_SetEncryptionAttributes(STATE_MACHINE_0, enc_iv, enc_key);
309 
310  RADIO_SetEncryptionCount(STATE_MACHINE_0, &count_tx[0], &count_rcv[0]);
311 
312  /* Enable encryption:1 for Enable, 0 for disable */
313  RADIO_SetEncryptFlags(STATE_MACHINE_0, ENABLE, ENABLE);
314 #endif
315 
316  /* Receives a packet on the desired channel. */
317  HAL_RADIO_ReceivePacket(channel, delay, receivedData, timeOut, MAX_LL_PACKET_LENGTH, RxCallback);
318  /* Infinite loop */
319  while(1) {
320  HAL_VTIMER_Tick();
321 
322  /* Print the packets sniffed */
323  if(fifo_size(&blueRec_fifo) !=0) {
324 
325  /* Get the length of the packet and the packet itself */
326  fifo_get_var_len_item(&blueRec_fifo, &length, packet);
327 
328  /* Get the RSSI information */
329  fifo_get(&blueRec_fifo, 4, (uint8_t*)&rssi_val);
330 
331  /* Get the timestamp */
332  fifo_get(&blueRec_fifo, 4, (uint8_t*)&timestamp);
333 
334  printf("\r\nchannel: %d, RSSI: %d dBm\r\n", channel, rssi_val);
335  //printf("Timestamp: %.3f ms\r\n", ((float)timestamp)*625/256000.0);
336  printf("Timestamp: %d.%03d ms\r\n", PRINT_INT((timestamp*625/256000.0)), PRINT_FLOAT((timestamp*625/256000.0)));
337 
338 
339  printf("Frame: ");
340  for(uint16_t i= 0; i<(length-MIC_FIELD_LEN); i++) {
341  printf("%02x:", packet[i]);
342  }
343 #ifdef PACKET_ENCRYPTION
344  printf("\r\nMIC: ");
345  for(uint16_t i= (length-MIC_FIELD_LEN); i<(length); i++) {
346  printf("%02x:", packet[i]);
347  }

349  }
350  }
