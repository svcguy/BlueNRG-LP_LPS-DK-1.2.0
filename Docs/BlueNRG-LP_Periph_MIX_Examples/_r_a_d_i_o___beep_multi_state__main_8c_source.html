242 void multi_state_schedular(multi_state_t * multi_state)
243 {
244  uint32_t next_wakeup_time_min = multi_state->wakeup_time[0];
245  uint8_t i = 0;
246  uint8_t index = 0;
247 
248  /* Find the minimum wakeup time among the active state machine */
249  for(i = 1;i<STATEMACHINE_COUNT;i++) {
250  if((next_wakeup_time_min > multi_state->wakeup_time[i]) && ((multi_state->activity_flag >> i)&0x01)) {
251  next_wakeup_time_min = multi_state->wakeup_time[i];
252  index = i;
253  }
254  }
255  RADIO_SetChannel(actPacket[index].StateMachineNo, channel[index], 0);
256  RADIO_MakeActionPacketPending(&actPacket[index]);
257  multi_state->wakeup_time[index] +=(period_usec[index] >> 10);
258 }

291 void beep_init(uint8_t StateMachineNo)
292 {
293  sendData[StateMachineNo][0] = 0x02;
294  sendData[StateMachineNo][1] = DATA_LEN; /* Length position is fixed */
295  sendData[StateMachineNo][2] = StateMachineNo;
296 
297  for(uint8_t i = 1; i < DATA_LEN; i++) {
298  sendData[StateMachineNo][2+i] = i;
299  }
300 
301  if(encryption[StateMachineNo]) {
302  sendData[StateMachineNo][1] += MIC_FIELD_LENGTH;
303  }
304 
305  actPacket[StateMachineNo].StateMachineNo = StateMachineNo;
306  actPacket[StateMachineNo].ActionTag = RELATIVE | TIMER_WAKEUP | TXRX | PLL_TRIG ;
307  actPacket[StateMachineNo].WakeupTime = period_usec[StateMachineNo];
308  actPacket[StateMachineNo].MaxReceiveLength = 0;/* not apllied for Tx*/
309  actPacket[StateMachineNo].data = (uint8_t *)&sendData[StateMachineNo][0];
310  actPacket[StateMachineNo].next_true = NULL; /* -> points to AP[0] */
311  actPacket[StateMachineNo].next_false = NULL; /* -> points to AP[0] */
312  actPacket[StateMachineNo].condRoutine = conditionRoutine;
313  actPacket[StateMachineNo].dataRoutine = dataRoutine;
314  multi_state.activity_flag = multi_state.activity_flag | 1<<StateMachineNo;
315  multi_state.wakeup_time[StateMachineNo] = actPacket[StateMachineNo].WakeupTime;
316  RADIO_SetChannelMap(StateMachineNo, &channel_map[0]);
317  RADIO_SetChannel(StateMachineNo, channel[StateMachineNo],0);
318  RADIO_SetTxAttributes(StateMachineNo, network_id[StateMachineNo] , 0x555555);
319  RADIO_SetTxPower(MAX_OUTPUT_RF_POWER);
320 
321  if(encryption[StateMachineNo]) {
322  RADIO_SetEncryptionCount(StateMachineNo, &count_tx[0], &count_rcv[0]);
323  RADIO_SetEncryptionAttributes(StateMachineNo, enc_iv, enc_key);
324  RADIO_SetEncryptFlags(StateMachineNo,(FunctionalState)encryption[StateMachineNo],(FunctionalState)encryption[StateMachineNo]);
325  }
326 
327  RADIO_SetReservedArea(&actPacket[StateMachineNo]);
328 }
