372  hdma_memtomem_dma1_channel1.Instance = DMA1_Channel1;
373  hdma_memtomem_dma1_channel1.Init.Request = DMA_REQUEST_MEM2MEM;
374  hdma_memtomem_dma1_channel1.Init.Direction = DMA_MEMORY_TO_MEMORY;

378  hdma_memtomem_dma1_channel1.Init.MemDataAlignment = DMA_MDATAALIGN_WORD;
379  hdma_memtomem_dma1_channel1.Init.Mode = DMA_NORMAL;
