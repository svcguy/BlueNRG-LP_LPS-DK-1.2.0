354 static void MX_CRC_Init(void)
355 {
356  /*-1- Configure the CRC peripheral */
357  hcrc.Instance = CRC;
358  /* The default polynomial is not used. It is required to defined it in hcrc.Init.GeneratingPolynomial*/
359  hcrc.Init.DefaultPolynomialUse = DEFAULT_POLYNOMIAL_DISABLE;
360  /* Set the value of the polynomial */
361  hcrc.Init.GeneratingPolynomial = CRC_POLYNOMIAL_8B;
362  /* The user-defined generating polynomial generates a 8-bit long CRC */
363  hcrc.Init.CRCLength = CRC_POLYLENGTH_8B;
364  /* The default init value is used */
365  hcrc.Init.DefaultInitValueUse = DEFAULT_INIT_VALUE_ENABLE;
366  /* The input data are not inverted */
367  hcrc.Init.InputDataInversionMode = CRC_INPUTDATA_INVERSION_NONE;
368  /* The output data are not inverted */
369  hcrc.Init.OutputDataInversionMode = CRC_OUTPUTDATA_INVERSION_DISABLE;
370  /* The input data are 32- bit long */
371  hcrc.InputDataFormat = CRC_INPUTDATA_FORMAT_WORDS;
372  if (HAL_CRC_Init(&hcrc) != HAL_OK)
373  {
374  /* Initialization Error */
375  Error_Handler();
376  }
377 }
