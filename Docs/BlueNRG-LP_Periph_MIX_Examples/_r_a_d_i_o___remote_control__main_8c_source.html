240 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
241 {
242  /* received a packet */
243  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
244 
245  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
246  BSP_LED_Toggle(BSP_LED1);
247  if(flag_SendingPacket == FALSE) {
248  HAL_RADIO_ReceivePacket(FREQUENCY_CHANNEL, RX_RELATIVETIME, receivedData, RX_RECEIVE_TIMEOUT, MAX_LL_PACKET_LENGTH, RxCallback);
249  }
250  else {
251  HAL_RADIO_SendPacket(FREQUENCY_CHANNEL, TX_RELATIVETIME, sendData, TxCallback);
252  flag_SendingPacket = FALSE;
253  }
254  }
255  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
256  if(flag_SendingPacket == FALSE) {
257  HAL_RADIO_ReceivePacket(FREQUENCY_CHANNEL, RX_RELATIVETIME, receivedData, RX_RECEIVE_TIMEOUT, MAX_LL_PACKET_LENGTH, RxCallback);
258  }
259  else {
260  HAL_RADIO_SendPacket(FREQUENCY_CHANNEL, TX_RELATIVETIME, sendData, TxCallback);
261  flag_SendingPacket = FALSE;
262  }
263  }
264 }
265  return TRUE;
266 }
