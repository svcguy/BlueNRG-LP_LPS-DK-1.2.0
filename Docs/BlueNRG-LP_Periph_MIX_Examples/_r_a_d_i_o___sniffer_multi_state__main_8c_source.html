266 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
267 {
268  static uint8_t count_t = 0;
269  static uint8_t StateMachine_index = 0;
270 
271  /* received a packet */
272  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
273 
274  /* Number of packet capture per State Machine */
275  if(count_t < No_Packet_Per_StateMachineNo) {
276  count_t++;
277 
278  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
279  fifo_put_var_len_item(&blueRec_fifo, p->data[1]+2, p->data,0,NULL);
280  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->rssi) );
281  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->timestamp_receive));
282  fifo_put(&blueRec_fifo, 1, (uint8_t*)(&channel[p->StateMachineNo]));
283  RADIO_SetChannel(StateMachine_index,(uint8_t)channel[StateMachine_index], 0);
284  RADIO_MakeActionPacketPending(&aPacket[StateMachine_index]);
285  }
286  else if((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
287 // printf("Timeout:%dus\r\n",rx_timeout);
288  printf("%d",count_t);
289  RADIO_SetChannel(StateMachine_index,(uint8_t)channel[StateMachine_index], 0);
290  RADIO_MakeActionPacketPending(&aPacket[StateMachine_index]);
291  }
292  else if((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) {
293  printf("CRC error\r\n");
294  printf("%d",count_t);
295  RADIO_SetChannel(StateMachine_index,(uint8_t)channel[StateMachine_index], 0);
296  RADIO_MakeActionPacketPending(&aPacket[StateMachine_index]);
297  }
298  }
299  else {
300  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
301  fifo_put_var_len_item(&blueRec_fifo, p->data[1]+2, p->data,0,NULL);
302  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->rssi) );
303  fifo_put(&blueRec_fifo, 4, (uint8_t*)(&p->timestamp_receive));
304  fifo_put(&blueRec_fifo, 1, (uint8_t*)(&channel[p->StateMachineNo]));
305  }
306  count_t = 0;
307  StateMachine_index++;
308  StateMachine_index = StateMachine_index%N_STATE_MACHINES; /* Update the State Machine */
309 
310  RADIO_SetChannel(StateMachine_index,(uint8_t)channel[StateMachine_index], 0);
311  RADIO_MakeActionPacketPending(&aPacket[StateMachine_index]);
312  // printf("\r\n >> Sniffing channel..%d StateMachineNo:%d Encryption:%d Network_ID:0x%X\r\n",(uint8_t)channel[StateMachine_index],StateMachine_index,encryption[StateMachine_index],network_id[StateMachine_index]);
313  printf("\r\n >> Sniffing channel..%d\r\n", (uint8_t)channel[StateMachine_index]);
314  }
315  }
316  return TRUE;
317 }

321 {
322  RADIO_SetChannelMap(StateMachineNo, channel_map);
323  RADIO_SetChannel(StateMachineNo, (uint8_t)channel[StateMachineNo], 0);
324  RADIO_SetTxAttributes(StateMachineNo, network_id[StateMachineNo], 0x555555);
325 
326  aPacket[StateMachineNo].StateMachineNo = StateMachineNo;
327  aPacket[StateMachineNo].ActionTag = PLL_TRIG | RELATIVE | TIMER_WAKEUP;
328  aPacket[StateMachineNo].WakeupTime = delay;
329  aPacket[StateMachineNo].MaxReceiveLength = MAX_LL_PACKET_LENGTH;
330  aPacket[StateMachineNo].data = receivedData;
331  aPacket[StateMachineNo].next_true = NULL_0;
332  aPacket[StateMachineNo].next_false = NULL_0;
333  aPacket[StateMachineNo].condRoutine = nullfunction;
334  aPacket[StateMachineNo].dataRoutine = RxCallback;
335 
336  RADIO_SetEncryptionAttributes(StateMachineNo, enc_iv, enc_key);
337  RADIO_SetEncryptionCount(StateMachineNo, &count_tx[0], &count_rcv[0]);
338  RADIO_SetEncryptFlags(StateMachineNo,(FunctionalState)encryption[StateMachineNo],(FunctionalState)encryption[StateMachineNo]);
339  RADIO_SetReservedArea(&aPacket[StateMachineNo]);
340 
341  printf(" >> Sniffing channel..%d StateMachineNo:%d Encryption:%d Network_ID:0x%X\r\n", (uint8_t)channel[StateMachineNo], StateMachineNo, encryption[StateMachineNo], network_id[StateMachineNo]);
342  return 0;
