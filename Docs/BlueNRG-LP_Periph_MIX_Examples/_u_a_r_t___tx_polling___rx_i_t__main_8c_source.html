280  /*-2- Configure UART peripheral for reception process (using LL) */

285  pBufferReadyForReception = aRXBufferA;

294  /*-3- Start the transmission process (using HAL Polling mode) */

296  As soon as RX buffer is detected as full, received bytes are echoed on TX line to PC com port */
297 
298  /* Infinite loop */

384 void UART_CharReception_Callback(void)
385 {
386  uint8_t *ptemp;
387 
388  /* Read Received character. RXNE flag is cleared by reading of RDR register */
389  pBufferReadyForReception[uwNbReceivedChars++] = LL_USART_ReceiveData8(USARTx);
390 
391  /* Checks if Buffer full indication has to be set */
392  if (uwNbReceivedChars >= RX_BUFFER_SIZE)
393  {
394  /* Set Buffer swap indication */
395  uwBufferReadyIndication = 1;
396 
397  /* Swap buffers for next bytes to be received */
398  ptemp = pBufferReadyForUser;
399  pBufferReadyForUser = pBufferReadyForReception;
400  pBufferReadyForReception = ptemp;
401  uwNbReceivedChars = 0;
402  }
403 }
