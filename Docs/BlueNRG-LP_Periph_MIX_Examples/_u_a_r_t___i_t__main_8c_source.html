259 int main(void)
260 {
261  /* System initialization function */

264  /* Error during system clock configuration take appropriate action */

271  /* Configure leds */
272  BSP_LED_Init(BSP_LED1);
273  BSP_LED_Init(BSP_LED2);
274  BSP_LED_Init(BSP_LED3);

276  /* Initialize all configured peripherals */
277  MX_GPIO_Init();
278  MX_USART_Init();

280  /*-2- Configure UART peripheral for reception process (using LL) */
281  /* Any data received will be stored "aRxBuffer" buffer : the number max of
282  data received is RXBUFFERSIZE */
283  /* Enable RXNE and Error interrupts */
284  LL_USART_EnableIT_RXNE(USARTx);
285  LL_USART_EnableIT_ERROR(USARTx);
286 
287  /*-3- Start the transmission process (using LL) **/
288  /* While the UART in reception process, user can transmit data from
289  "aTxStartMessage" buffer */
290  /* Start USART transmission : Will initiate TXE interrupt after TDR register is empty */
291  LL_USART_TransmitData8(USARTx, aTxStartMessage[uwTxIndex++]);
292 
293  BSP_LED_On(BSP_LED1);
294 
295  /* Enable TXE interrupt */
296  LL_USART_EnableIT_TXE(USARTx);
297 
298  /*-4- Wait for the end of the transfer */
299  /* USART IRQ handler is not anymore routed to HAL_UART_IRQHandler() function
300  and is now based on LL API functions use.
301  Therefore, use of HAL IT based services is no more possible. */
302  /* Once TX transfer is completed, LED1 will turn On.
303  Then, when RX transfer is completed, LED2 will turn On. */
304  while (ubTxComplete == 0)
305  {
306  }
307 
308  while (ubRxComplete == 0)
309  {
310  }
311 
312  BSP_LED_Off(BSP_LED1);
313  BSP_LED_On(BSP_LED2);
314 
315  /*-5- Send the received Buffer */
316  /* Even if use of HAL IT based services is no more possible, use of HAL Polling based services
317  (as Transmit in polling mode) is still possible. */
318  if(HAL_UART_Transmit(&huartx, (uint8_t*)aRxBuffer, RXBUFFERSIZE, 1000)!= HAL_OK)
319  {
320  /* Transfer error in transmission process */
321  Error_Handler();
322  }
323 
324  /*-6- Send the End Message */
325  if(HAL_UART_Transmit(&huartx, (uint8_t*)aTxEndMessage, TXENDMESSAGESIZE, 1000)!= HAL_OK)
326  {
327  /* Transfer error in transmission process */
328  Error_Handler();
329  }
330 
331  /* Infinite loop */
332  while (1)
333  {
334  }
335 }
