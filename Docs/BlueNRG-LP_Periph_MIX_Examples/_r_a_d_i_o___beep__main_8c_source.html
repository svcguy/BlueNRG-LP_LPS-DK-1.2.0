232 uint8_t conditionRoutine(ActionPacket* p)
233 {
234  if( (p->status & BLUE_INTERRUPT1REG_DONE) != 0) {
235  if(p->status & BLUE_STATUSREG_PREVTRANSMIT){
236  BSP_LED_Toggle(BSP_LED1);
237  channel ++;
238  if(channel == (END_CHANNEL+1)) {
239  channel = STARTING_CHANNEL;
240  }
241  RADIO_SetChannel(STATE_MACHINE_0, channel, 0);
242  }
243  }
244  return TRUE;
