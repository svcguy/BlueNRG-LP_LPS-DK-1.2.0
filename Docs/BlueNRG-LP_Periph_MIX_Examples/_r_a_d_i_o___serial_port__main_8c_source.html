354 void IOProcessInputData(uint8_t* data_buffer, uint16_t Nb_bytes)
355 {
356  static uint8_t cUartDataSize = 0;
357  uint8_t i;
358 
359  for (i = 0; i < Nb_bytes; i++) {
360  uart_buffer[cUartDataSize] = data_buffer[i];
361  BSP_COM_Write(&data_buffer[i],1);
362  cUartDataSize++;
363 
364  if((uart_buffer[cUartDataSize-1] == '\r') || (cUartDataSize == (MAX_CHAT_PACKET_LEN - MIC_FIELD_LEN))) {
365  xUartDataReady = SET;
366  Osal_MemCpy(&sendData[2], uart_buffer, cUartDataSize);
367  sendData[1] = cUartDataSize + MIC_FIELD_LEN;
368  cUartDataSize = 0;
369  }
370  }
