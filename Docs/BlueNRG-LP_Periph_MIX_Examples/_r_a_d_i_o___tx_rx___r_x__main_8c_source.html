216 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
217 {
218  uint8_t ret;
219  /* received a packet */
220  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
221 
222  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
223  BSP_LED_Toggle(BSP_LED1);
224  rx_done = TRUE;
225  }
226  else if( ((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) ){
227  BSP_LED_Toggle(BSP_LED2);
228  ret = HAL_RADIO_ReceivePacketWithAck(channel, RX_WAKEUP_TIME, receivedData, sendAckData, RX_TIMEOUT, MAX_LL_PACKET_LENGTH, RxCallback);
229  if(ret != SUCCESS_0) {
230  printf("ERROR %d (%d)\r\n",ret, packet_counter);
231  }
232  }
233  }
234  else if((p->status & BLUE_INTERRUPT1REG_DONE) != 0) {
235  ret = HAL_RADIO_ReceivePacketWithAck(channel, RX_WAKEUP_TIME, receivedData, sendAckData, RX_TIMEOUT, MAX_LL_PACKET_LENGTH, RxCallback);
236  if(ret != SUCCESS_0) {
237  printf("ERROR %d (%d)\r\n",ret, packet_counter);
238  }
239  }
240  return TRUE;
241 }
