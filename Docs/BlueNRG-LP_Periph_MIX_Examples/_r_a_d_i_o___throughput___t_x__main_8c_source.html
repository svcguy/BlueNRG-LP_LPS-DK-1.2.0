322 uint8_t TxCallback(ActionPacket* p, ActionPacket* next)
323 {
324  /* received a packet */
325  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
326 
327  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
328  time2 = LL_TIM_GetCounter(TIMx);
329  timer_reload_store = timer_reload;
330  packet_counter++;
331  }
332  else if((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
333  }
334  else if((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) {
335  }
336 
337  if(packet_counter != MAX_NUM_PACKET) {
338  sendNewPacket = TRUE;
339  }
340  else
341  {
342  diff = (0xFFFF - time2) + timer_reload_store*0xFFFF;
343  time_cumulate += diff;
344  }
345 }
346  /* Transmit complete */
347  else if((p->status & BLUE_INTERRUPT1REG_DONE) != 0){
348  if(packet_counter != MAX_NUM_PACKET) {
349  sendNewPacket = TRUE;
350  }
351  }
352  return TRUE;
353 }

476  uint8_t ret;

494 
495  if(DATA_PACKET_LEN > (MAX_PACKET_LENGTH-HEADER_LENGTH)) {
496  printf("DATA_PACKET_LEN too big %d\r\n", DATA_PACKET_LEN);
497  while(1);
498  }
499 
500  /* Build packet */
501  sendData[0] = 0x02;
502  sendData[1] = DATA_PACKET_LEN; /* Length position is fixed */
503  for(volatile uint16_t j=0; j<DATA_PACKET_LEN; j++) {
504  sendData[2+j] = 0xAE;
505  }
506 
507  /* Configures the transmit power level */
508  RADIO_SetTxPower(MAX_OUTPUT_RF_POWER);
509 
510  printf("BlueNRG-LP Radio Driver Throughput Application\r\n");
511 
512  LL_TIM_ClearFlag_UPDATE(TIMx);
513 
514  /* Enable the update interrupt */
515  LL_TIM_EnableIT_UPDATE(TIMx);
516 
517  /* Enable counter */
518  LL_TIM_EnableCounter(TIMx);
519 
520  ret = HAL_RADIO_SendPacket(channel, TX_WAKEUP_TIME, sendData, TxCallback);
521  if(ret != SUCCESS_0) {
522  printf("ERROR %d (%d)\r\n",ret, packet_counter);
523  }
524 
525  RADIO_SetBackToBackTime(100);
526  /* Infinite loop */
527  while(1) {
528  HAL_VTIMER_Tick();
529 
530  if(packet_counter == MAX_NUM_PACKET) {
531  if(time_cumulate!=0) {
532  //printf("%d TX packets. %d PCKT LEN. Average time: %.2f ms. Data throughput: %.1f kbps.\r\n", packet_counter, DATA_PACKET_LEN, (((float)time_cumulate)/1000.0/packet_counter), (((float)packet_counter*DATA_PACKET_LEN*8)*1000.0)/time_cumulate);
533  printf("%d TX packets. %d PCKT LEN. Average time: %d.%02d ms. Data throughput: %d.%01d kbps.\r\n", packet_counter, DATA_PACKET_LEN,PRINT_INT(time_cumulate/1000.0/packet_counter),PRINT_FLOAT(time_cumulate/1000.0/packet_counter),PRINT_INT((packet_counter*DATA_PACKET_LEN*8)*1000.0/time_cumulate),PRINT_FLOAT((packet_counter*DATA_PACKET_LEN*8)*1000.0/time_cumulate));
534  }
535  packet_counter = 0;
536  timeout_error_counter = 0;
537  crc_error_counter = 0;
538  sendData[6] = 0;
539  sendNewPacket = TRUE;
540  time_cumulate = 0;
541  for(volatile uint32_t i = 0; i<0xFFFFF; i++);
542  }
543 
544  if(sendNewPacket == TRUE) {
545  sendNewPacket = FALSE;
546  if(packet_counter != 0){
547  diff = (0xFFFF - time2) + timer_reload_store*0xFFFF;
548  }
549  time_cumulate += diff;
550  timer_reload = 0;
551  LL_TIM_SetCounter(TIMx, 0xFFFF);
552 #ifdef UNIDIRECTIONAL_TEST
553  ret = UNIDIRECTIONAL_Sequence(channel, sendData, UniTxCallback);
554 #else
555  ret = BIDIRECTIONAL_TX_Sequence(channel, sendData, receivedData, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, TxCallback);
556 #endif
557  if(ret != SUCCESS_0) {
558  printf("ERROR %d (%d)\r\n",ret, packet_counter);
559  }
560 
561  }
562  }
