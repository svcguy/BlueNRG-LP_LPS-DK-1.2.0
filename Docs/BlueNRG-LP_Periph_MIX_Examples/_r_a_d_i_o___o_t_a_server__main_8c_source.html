428 uint8_t OTA_DataRequestCallback(ActionPacket* p, ActionPacket* next)
429 {
430  static volatile uint8_t rx_valid = 0, retry = 0;
431 
432  /* received a packet */
433  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
434  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
435  if(p->data[0] == HEADER_DATAREQ) {
436  if(p->data[1] == 2) {
437  rx_valid = 1;
438  seq_num = (((uint16_t)p->data[2])<<8) | ((uint16_t)p->data[3]);
439  }
440  else {
441  PRINTF(">> OTA_DataRequestCallback != %d LENGTH WRONG\r\n", p->data[1]);
442  BSP_LED_Off(BSP_LED2);
443  while(1);
444  }
445  }
446  /* It have received the ACK packet from GETDATA state */
447  else if(p->data[0] == HEADER_SENDATA) {
448  PRINTF(">> OTA_DataRequestCallback get HEADER_SENDATA\r\n");
449  }
450  else {
451  PRINTF(">> OTA_DataRequestCallback != %02x\r\n", p->data[0]);
452  while(1);
453  }
454  }
455  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
456  ota_state_machine_g = OTA_DATAREQ;
457  retry++;
458  }
459  }
460  else { /* ACK transmission completed */
461  if(rx_valid) {
462  retry = 0;
463  rx_valid = 0;
464  ota_state_machine_g = OTA_SENDATA;
465  }
466  else { /* Received a HEADER_SENDATA */
467  ota_state_machine_g = OTA_DATAREQ;
468  }
469  }
470  if(retry > MAX_RETRY) { /* create a reset function */
471  retry = 0;
472  // TODO: handle the crash of the connection, once the Ymodem is opened (send abort on ymodem channel */
473  PRINTF(">> OTA_DataRequestCallback MAX RETRY\r\n");
474  // ota_state_machine_g = OTA_CONNECTION;
475  // seq_num = 0;
476  }
477  return TRUE;

565 uint8_t OTA_Tick()
566 {
567  static uint32_t ota_pkt_idx = 0;
568  static uint32_t ota_pkt_size = 0;
569  uint8_t ret;
570 
571  /* Manage the token between the two state machines */
572  if(app_token == TOKEN_TAKEN_YMODEM)
573  return 0;
574  else
575  app_token = TOKEN_TAKEN_OTA;
576 
577  if(ota_state_machine_g == OTA_CONNECTION) {
578  PRINTF("OTA_CONNECTION\r\n");
579 
580  /* state setting */
581  last_frame = 0;
582  seq_num = 0;
583  seq_num_max = image_size / BYTE_IN_FRAME;
584  if(image_size%BYTE_IN_FRAME) {
585  seq_num_max++;
586  }
587 
588  /* frame setting */
589  tx_ota_buffer[0] = HEADER_CONNECTION;
590  tx_ota_buffer[1] = 0;
591 
592  debugger[3] = ota_state_machine_g;
593  ota_state_machine_g = OTA_ONGOING;
594  RADIO_SetBackToBackTime(150);
595 
596  BSP_LED_On(BSP_LED2);
597  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, tx_ota_buffer, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_ConnectionCallback);
598  if(ret != SUCCESS_0) {
599  PRINTF("ERROR %d\r\n",ret);
600  }
601  }
602 
603  else if(ota_state_machine_g == OTA_SIZE) {
604  PRINTF("OTA_SIZE\r\n");
605 
606  /* frame setting */
607  tx_ota_buffer[0] = HEADER_SIZE;
608  tx_ota_buffer[1] = 4;
609  tx_ota_buffer[2] = (uint8_t)(image_size>>24);
610  tx_ota_buffer[3] = (uint8_t)(image_size>>16);
611  tx_ota_buffer[4] = (uint8_t)(image_size>>8);
612  tx_ota_buffer[5] = (uint8_t)image_size;
613 
614  debugger[3] = ota_state_machine_g;
615  ota_state_machine_g = OTA_ONGOING;
616  RADIO_SetBackToBackTime(150);
617  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, tx_ota_buffer, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_SizeCallback);
618  if(ret != SUCCESS_0) {
619  PRINTF("ERROR %d\r\n",ret);
620  }
621  }
622 
623  else if(ota_state_machine_g == OTA_START) {
624  PRINTF("OTA_START\r\n");
625 
626  /* frame setting */
627  tx_ota_buffer[0] = HEADER_START;
628  tx_ota_buffer[1] = 0;
629 
630  debugger[3] = ota_state_machine_g;
631  ota_state_machine_g = OTA_ONGOING;
632  RADIO_SetBackToBackTime(300);
633  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_ota_buffer, tx_ota_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_StartCallback);
634  if(ret != SUCCESS_0) {
635  PRINTF("ERROR %d\r\n",ret);
636  }
637  }
638 
639  else if(ota_state_machine_g == OTA_DATAREQ) {
640  PRINTF("OTA_DATAREQ %d\r\n", seq_num);
641 
642  /* Manage the token between the two state machines */
643  if(ymodem_state_machine_g == YMODEM_LOAD) {
644  app_token = TOKEN_RELEASE;
645  return 0;
646  }
647 
648  /* frame setting */
649  tx_ota_buffer[0] = HEADER_DATAREQ;
650  tx_ota_buffer[1] = 0;
651  debugger[3] = ota_state_machine_g;
652  ota_state_machine_g = OTA_ONGOING;
653  RADIO_SetBackToBackTime(300);
654  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_ota_buffer, tx_ota_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_DataRequestCallback);
655  if(ret != SUCCESS_0) {
656  PRINTF("ERROR %d\r\n",ret);
657  }
658  }
659 
660  else if(ota_state_machine_g == OTA_SENDATA) {
661  PRINTF("OTA_SENDATA %d\r\n", seq_num);
662 
663  /* pointer of buffer */
664  ota_pkt_idx = (seq_num * BYTE_IN_FRAME);
665 
666  /* size of buffer */
667  if((ota_pkt_idx + BYTE_IN_FRAME) <= data_len) {
668  debugger[0]++;
669  ota_pkt_size = BYTE_IN_FRAME + 2; /* +2 because the sequence number is added */
670  }
671  else if((ota_pkt_idx + 1) > data_len) {
672  debugger[1]++;
673  /* last frame already sent */
674  ota_state_machine_g = OTA_DATAREQ;
675  if(ymodem_state_machine_g == YMODEM_CLOSE) {
676  while(1);
677  }
678  app_token = TOKEN_RELEASE;
679  return 0;
680  }
681  /* still a few byte to send, it is the last frame */
682  else if((ota_pkt_idx + 1) <= data_len) {
683  debugger[2]++;
684  ota_pkt_size = (data_len - ota_pkt_idx) + 2; /* +2 because the sequence number is added */
685  if(ymodem_state_machine_g == YMODEM_CLOSE)
686  last_frame = 1;
687  }
688  else {
689  while(1); /* not possible ? */
690  }
691 
692  /* copy the buffer */
693  for(uint8_t i = 0; i < ota_pkt_size; i++) {
694  image_tmp[4+i] = image[ota_pkt_idx%1024 + i];
695  }
696 
697  image_tmp[0] = HEADER_SENDATA;
698  image_tmp[1] = ota_pkt_size;
699  image_tmp[2] = (uint8_t)(seq_num>>8);
700  image_tmp[3] = (uint8_t)seq_num;
701 
702  debugger[3] = ota_state_machine_g;
703  ota_state_machine_g = OTA_ONGOING;
704  RADIO_SetBackToBackTime(150);
705  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, (uint8_t*)image_tmp, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_SendDataCallback);
706  if(ret != SUCCESS_0) {
707  PRINTF("ERROR %d\r\n",ret);
708  }
709  }
710 
711  else if(ota_state_machine_g == OTA_COMPLETE) {
712  PRINTF("OTA_COMPLETE\r\n");
713 
714  PRINTF("image %d\r\n",image_size);
715  ota_state_machine_g = OTA_CONNECTION;
716  last_frame = 0;
717  seq_num = 0;
718  app_token = TOKEN_RELEASE;
719  }
720 
721  if (bootloadingCompleted) {
722  bootloadingCompleted = 0;
723  }
724 
725  return (bootloadingCompleted_end);
726 }

735 int32_t OTA_ymodem_tick(void)
736 {
737  int32_t ret = 0;
738  static volatile uint32_t packet = 0;
739  uint32_t ymodem_data_len = 0;
740  volatile uint32_t tmp_size;
741  uint8_t tmp_buff[1024+8];
742 
743  if(app_token == TOKEN_TAKEN_OTA)
744  return 0;
745  app_token = TOKEN_TAKEN_YMODEM;
746 
747  /* Get file name and size */
748  if(ymodem_state_machine_g == YMODEM_SIZE) {
749  ret = Ymodem_Receive ((uint8_t *)tmp_buff, 0, (uint32_t *) &image_size, packet++);
750  if (ret != YMODEM_CONTINUE) {
751  ymodem_state_machine_g = YMODEM_ABORT;
752  }
753  else {
754  ymodem_state_machine_g = YMODEM_LOAD;
755  app_token = TOKEN_RELEASE;
756  }
757  }
758 
759  else if(ymodem_state_machine_g == YMODEM_LOAD) {
760 
761  /* Answer to the previous Receive */
762 
763  Ymodem_SendAck();
764 
765  ret = Ymodem_Receive (image, 0, (uint32_t *) &ymodem_data_len, packet++);
766  if (ret != YMODEM_CONTINUE) {
767  ymodem_state_machine_g = YMODEM_ABORT;
768  return ret;
769  }
770  else {
771  app_token = TOKEN_RELEASE;
772  }
773 
774  if((ymodem_data_len+data_len) > image_size) {
775  ymodem_state_machine_g = YMODEM_CLOSE;
776 
777  /* Answer to the previous Receive */
778  Ymodem_SendAck();
779  data_len = image_size;
780  }
781  else {
782  data_len += ymodem_data_len;
783  ymodem_state_machine_g = YMODEM_WAIT;
784  }
785  }
786 
787  else if(ymodem_state_machine_g == YMODEM_WAIT) {
788  /* Empty state waiting for OTA action. Still data with ymodem. */
789  ymodem_state_machine_g = YMODEM_LOAD;
790  }
791 
792  else if(ymodem_state_machine_g == YMODEM_CLOSE) {
793  ymodem_state_machine_g = YMODEM_COMPLETE;
794 
795  /* Close Ymodem session */
796  ret = Ymodem_Receive (tmp_buff, 0, (uint32_t *) &tmp_size, packet++);
797  if (ret != YMODEM_DONE) {
798  Ymodem_Abort();
799  }
800  Ymodem_SendAck();
801  }
802 
803  else if(ymodem_state_machine_g == YMODEM_COMPLETE) {
804 
805  }
806 
807  else if(ymodem_state_machine_g == YMODEM_ABORT) {
808  Ymodem_Abort();
809  /* Reset all the states and variables or reset the system */
810  }
811  return 0;
812 }
