318  /*-1- Start the Half Duplex Communication process */

321 
322  /* Enable SPI before start transmission */
323  LL_SPI_Enable(hspiMaster.Instance);
324 
325  while(ubNbDataToTransmit > 0)

333 #elif defined( CONFIG_DATASIZE_8BIT )
334  /* Transmit 8bit Data */

338  }
339  }
340 
341  /* Wait End Of Transmission: TXE set and Tx Fifo empty */

343  while((LL_SPI_IsActiveFlag_TXE(hspiMaster.Instance) != 1));
