221 uint8_t RxCallback(ActionPacket* p, ActionPacket* next)
222 {
223  /* received a packet */
224  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
225 
226  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
227  rx_done = TRUE;
228  rx_timeout = RX_TIMEOUT_OK;
229  }
230  else if( ((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) ){
231  rx_timeout = RX_TIMEOUT_NOTOK;
232  schedule_rx = TRUE;
233  }
234  }
235  /* Transmit complete */
236  else {
237  schedule_rx = TRUE;
238  }
239  return TRUE;
240 }
