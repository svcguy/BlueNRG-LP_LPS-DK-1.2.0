239 uint8_t TxCallback(ActionPacket* p, ActionPacket* next)
240 {
241  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
242 
243  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
244  }
245  else if((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
246  BSP_LED_Toggle(BSP_LED2);
247  timeout_error_counter++;
248  }
249  else if((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) {
250  BSP_LED_Toggle(BSP_LED3);
251  crc_error_counter++;
252  }
253  sendData[6] = (uint8_t)(++packet_counter);
254 
255  if(packet_counter == MAX_NUM_PACKET) {
256  print_res = 1;
257  packet_counter_res = packet_counter;
258  crc_error_counter_res = crc_error_counter;
259  timeout_error_counter_res = timeout_error_counter;
260  packet_counter = 0;
261  timeout_error_counter = 0;
262  crc_error_counter = 0;
263  sendData[6] = 0;
264  }
265  BSP_LED_Toggle(BSP_LED1);
266  sendNewPacket = TRUE;
267  }
268  return TRUE;
269 }
