234 uint8_t TxCallback(ActionPacket* p, ActionPacket* next)
235 {
236  if((p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0){
237 
238  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
239  }
240  else if((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
241  timeout_error_counter++;
242  }
243  else if((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) {
244  crc_error_counter++;
245  }
246  sendData[6] = (uint8_t)(++packet_counter);
247 
248  if(packet_counter == MAX_NUM_PACKET) {
249  print_res = 1;
250  packet_counter_res = packet_counter;
251  crc_error_counter_res = crc_error_counter;
252  timeout_error_counter_res = timeout_error_counter;
253  packet_counter = 0;
254  timeout_error_counter = 0;
255  crc_error_counter = 0;
256  sendData[6] = 0;
257  }
258  sendNewPacket = TRUE;
259  }
260  return TRUE;
261 }
