317 uint8_t OTA_ConnectionCallback(ActionPacket* p, ActionPacket* next)
318 {
319  static volatile uint8_t rx_valid = 0;
320 
321  /* received a packet */
322  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
323  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
324  /* check the reception of a connection header otherwise do not go to app_size state */
325  if(p->data[0] == HEADER_CONNECTION) {
326  rx_valid = 1;
327  }
328  }
329  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
330  ota_state_machine_g = OTA_CONNECTION;
331  }
332  }
333  else { /* Transmit complete */
334  if(rx_valid) {
335  rx_valid = 0;
336  ota_state_machine_g = OTA_SIZE;
337  }
338  else {
339  ota_state_machine_g = OTA_CONNECTION;
340  }
341  }
342  return TRUE;

350 uint8_t OTA_SizeCallback(ActionPacket* p, ActionPacket* next)
351 {
352  static volatile uint8_t rx_valid = 0, retry = 0;
353 
354  /* received a packet */
355  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
356  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
357  if(p->data[0] == HEADER_SIZE) {
358  if(p->data[1] == 4) {
359  rx_valid = 1;

363  }
364  }
365  }
366  else {
367  retry++;
368  }
369  }
370  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
371  ota_state_machine_g = OTA_SIZE;
372  retry++;

377  retry = 0;

500 uint8_t OTA_GetDataCallback(ActionPacket* p, ActionPacket* next)
501 {
502  static volatile uint8_t rx_valid = 0, retry = 0;
503  volatile uint16_t sequence_number_received = 0;
504 
505  /* received a packet */
506  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
507  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
508  if(p->data[0] == HEADER_GETDATA) {
509  rx_valid = 1;
510  if(p->data[1] > 2) {
511  sequence_number_received = (((uint16_t)p->data[2])<<8) | ((uint16_t)p->data[3]);
512  if(seq_num == sequence_number_received) {
513  for(uint8_t i=0; i< (p->data[1]-2); i++) {
514  page_buffer[page_size+i] = p->data[4+i];
515  }
516  page_size += p->data[1] - 2;
517  seq_num++;
518  }
519  }
520  }
521  }
522  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
523  ota_state_machine_g = OTA_GETDATA;
524  retry++;
525  }
526  }
527  else { /* Transmit complete */
528  if(rx_valid) {
529  retry = 0;
530  rx_valid = 0;
531  ota_state_machine_g = OTA_FLASHDATA;
532  }
533  else {
534  ota_state_machine_g = OTA_GETDATA;
535  }
536  }
537  if(retry > MAX_RETRY) {
538  retry = 0;
539  ota_state_machine_g = OTA_DATAREQ;
540  }
541  return TRUE;
542 }

613  }
614

619  ota_state_machine_g = OTA_ONGOING;
620  RADIO_SetBackToBackTime(300);
621  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_buffer, tx_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_SizeCallback);
622  if(ret != SUCCESS_0) {
623  PRINTF("ERROR %d\r\n",ret);
624  }
625  }
626 
627  else if(ota_state_machine_g == OTA_START) {
628  if(app_size<=SM_APP_SIZE) { /* Check if the new application fit the Flash memory */
629  //PRINTF("OK for 0x%08X %.2f KB max %.2f KB\r\n", app_size, ((float)app_size)/1024.0, ((float)SM_APP_SIZE)/1024.0);

631 
632  PRINTF("OTA_START\r\n");
633  tx_buffer[0] = HEADER_START;
634  tx_buffer[1] = 0;
635 
636  seq_num_max = app_size / BYTE_IN_FRAME;
637  if(app_size%BYTE_IN_FRAME) {
638  seq_num_max++;
639  }
640 
641  ota_state_machine_g = OTA_ONGOING;
642  RADIO_SetBackToBackTime(150);
643  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, tx_buffer, rx_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_StartCallback);
644  if(ret != SUCCESS_0) {
645  PRINTF("ERROR %d\r\n",ret);
646  }
647  }
648  /* TO BE DONE */
649  else { /* If the new image does not fit the Flash memory */
650  //PRINTF("NOT ok for 0x%08X %.2f KB max %.2f KB\r\n", app_size, ((float)app_size)/1024.0, ((float)SM_APP_SIZE)/1024.0);

662  }

676  PRINTF("ERROR %d\r\n",ret);
677  }
678  }
679  else {
680  ota_state_machine_g = OTA_FLASHDATA;
681 // ota_state_machine_g = OTA_COMPLETE;
682  }
683  }
684 
685  else if(ota_state_machine_g == OTA_GETDATA) {
686 // PRINTF("OTA_GETDATA %d page_size %d\r\n", seq_num, page_size);
687  tx_buffer[0] = HEADER_GETDATA;
688  tx_buffer[1] = 0;
689  ota_state_machine_g = OTA_ONGOING;
690  RADIO_SetBackToBackTime(300);
691  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_buffer, tx_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_GetDataCallback);
692  if(ret != SUCCESS_0) {
693  PRINTF("ERROR %d\r\n",ret);
694  }
695 
696  }
697 
698  else if(ota_state_machine_g == OTA_FLASHDATA) {
699 // PRINTF("OTA_FLASHDATA\r\n");
700 
701  /* New data available */
702  if((page_size == _MEMORY_BYTES_PER_PAGE_) || ((app_size - page_size) == 0)) {
703 
704  /* FLASH LOAD _MEMORY_BYTES_PER_PAGE_ */
705 
706  /* in which address ? must be calculated my free address plus size after write op */
707 
708  int app_size_tmp = app_size - page_size;
709  if(app_size_tmp < 0) {
710  while(1);
711  }
712 
713  uint16_t i;
714 
715  i=0;
716  while(i<page_size) {
717 
718  if( (i+16) <= page_size) {
719  if(page_currently_written==0 && i==16) {
720  page_buffer[16] = 0xFF;
721  page_buffer[17] = 0xFF;
722  page_buffer[18] = 0xFF;
723  page_buffer[19] = 0xFF;
724  }
725  LL_FLASH_ProgramBurst(FLASH,APP_WITH_OTA_SERVICE_ADDRESS + page_currently_written*2048 + i, (uint32_t *)&page_buffer[i]);
726  i += 16;
727  }
728  else {
729  LL_FLASH_Program(FLASH,APP_WITH_OTA_SERVICE_ADDRESS + page_currently_written*2048 + i, *(uint32_t *)&page_buffer[i]);
730  i += 4;
731  }
732  }
733  page_currently_written++;
734 
735  /* erase data buffer */
736  for(uint16_t i=0; i<page_size; i++) {
737  page_buffer[i] = 0;
738  }
739 
740  if(app_size_tmp >= 0) {
741  app_size -= page_size;
742  }
743  page_size = 0;
744  //PRINTF("app_size 0x%08X %.2f KB\r\n", app_size, ((float)app_size)/1024.0);
745  PRINTF("app_size 0x%08X %d.%02d KB\r\n", app_size, PRINT_INT(app_size/1024.0),PRINT_FLOAT(app_size/1024.0));
746  }
747  if(app_size) {
748  ota_state_machine_g = OTA_DATAREQ;
749  }
750  else {
751  ota_state_machine_g = OTA_COMPLETE;
752  /* If the new image does not fit the Flash range */
753  PRINTF("OTA_COMPLETE\r\n");
754 
755  bootloadingCompleted = 1;
756  }
757  }
758 
759  if (bootloadingCompleted)
760  {
761  /* Set valid tag x the new application just successfully upgraded through OTA */
762  OTA_Set_Application_Tag_Value(APP_WITH_OTA_SERVICE_ADDRESS, OTA_VALID_TAG);
763 
764  bootloadingCompleted = 0;
765  bootloadingCompleted_end = 1;
766  PRINTF("** Over The Air BLE FW upgrade completed with success! *****************\r\n");
767  PRINTF("** Application is JUMPING to new base address: 0x%08X *********************\r\n",(unsigned int)imageBase);
768  }
769 
770  return (bootloadingCompleted_end);
771 }
