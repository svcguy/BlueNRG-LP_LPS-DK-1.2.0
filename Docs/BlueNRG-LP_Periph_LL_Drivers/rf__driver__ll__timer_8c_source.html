155 static void _get_calibration_data(TIMER_ContextType *context)
156 {
157  int32_t period,freq;
158  period = TIMER_GET_SLOW_PERIOD;
159 
160  while(period != TIMER_GET_SLOW_PERIOD || period == 0)
161  {
162  period = TIMER_GET_SLOW_PERIOD;
163  }
164 
165  int32_t mul1 = 0x8BCF6 ;
166  int32_t b1 = period >> 8 ;
167  int32_t b2 = period & 0xff ;
168  context->period1 = ((mul1 * b1) + ((b2 * mul1)>>8) + 16 )>>5 ;
169  context->period = period;
170 
171  int32_t mult = 0x753 ;
172  freq = TIMER_GET_SLOW_FREQEUNCY;
173 
174  while(freq != TIMER_GET_SLOW_FREQEUNCY || freq == 0)
175  {
176  freq = TIMER_GET_SLOW_FREQEUNCY;
177  }
178  int32_t a1 = freq >> 6 ;
179  int32_t a2 = a1 * mult ;
180  context->freq1 = (a2 + 128) >> 8 ;
181  context->freq = freq;
182 }

246 /* This function update the system time after a calibration.
247  * If the user calls too often this function, you could have rounding issues in the integer maths.

249 static void update_system_time(TIMER_ContextType *context)
250 {
251  uint32_t current_machine_time;
252  current_machine_time = WAKEUP->ABSOLUTE_TIME;
253  uint32_t period = context->last_period1;
254  context->cumulative_time = context->last_calibration_time + \
255  blue_unit_conversion(TIME_ABSDIFF(current_machine_time,
256  context->last_calibration_machine_time),
257  period, MULT64_THR_PERIOD);
258 
259  if ((TIMER_Context.periodic_calibration == 0) && (TIME_ABSDIFF(current_machine_time, context->last_calibration_machine_time) < context->calibration_machine_interval)) {
260  context->cumulative_time += blue_unit_conversion(TIMER_MAX_VALUE, period, MULT64_THR_PERIOD);
261  }
262  context->last_machine_time = current_machine_time;
263  context->last_calibration_machine_time = current_machine_time;
264  context->last_calibration_time = context->cumulative_time;
265  context->last_period1 = context->period1;
266 }

305

683 uint8_t TIMER_SetRadioTimerValue(uint32_t timeout, BOOL event_type, BOOL cal_req)
684 {
685  uint32_t current_time, delay, radio_init_delay, device_delay, relTimeout;
686  /*choose the 2nd init duration. Check the event_type and cal. request*/
687  if(event_type == TX)
688  {
689  if(cal_req)
690  {
691  radio_init_delay = TIMER_Context.tx_cal_delay;
692  device_delay = TIMER_Context.tx_cal_delay_st;
693  }
694  else
695  {
696  radio_init_delay = TIMER_Context.tx_no_cal_delay;
697  device_delay = TIMER_Context.tx_cal_delay_st;
698  }
699  }
700  else
701  {
702  if(cal_req)
703  {
704  radio_init_delay = TIMER_Context.rx_cal_delay;
705  device_delay = TIMER_Context.tx_cal_delay_st;
706  }
707  else
708  {
709  radio_init_delay = TIMER_Context.rx_no_cal_delay;
710  device_delay = TIMER_Context.tx_cal_delay_st;
711  }
712  }
713 
714  /* At this point, it is care of the upper layers to guarantee that the timeout represents an absolute time in the future */
715  relTimeout = timeout - (uint32_t)get_system_time_and_machine(&TIMER_Context,&current_time);
716  /*Check if the timeout is beyond the wakeup time offset. Then program either the WakeUp timer or the Timer1*/
717  if (relTimeout > (device_delay + TIMER_Context.hs_startup_time + MARGIN_EXT))
718  {
719  /*The timeout is after the wakeup_time_offset, So it is ok to program the wakeup timer*/
720  delay = blue_unit_conversion(relTimeout, TIMER_Context.freq1, MULT64_THR_FREQ) \
721  - blueglob->WAKEUPINITDELAY \
722  - radio_init_delay;
723  WAKEUP->BLUE_WAKEUP_TIME = ((current_time + delay) & TIMER_MAX_VALUE);
724  TIMER_BLUE_SET_REQ_MODE(0);
725  TIMER_DISABLE_TIMER12;
726  TIMER_ENABLE_BLUE_SLEEP_REQ;
727  radio_init_delay += blueglob->WAKEUPINITDELAY;
728  }
729  else
730  {
731  delay = blue_unit_conversion(relTimeout , TIMER_Context.freq1, MULT64_THR_FREQ) \
732  - TIMER_Context.tim12_delay_mt \
733  - radio_init_delay;
734  TIMER_SET_TIMER1((current_time + delay) & TIMER_MAX_VALUE);
735  TIMER_DISABLE_WAKEUP_TIMER;
736  radio_init_delay += TIMER_Context.tim12_delay_mt;
737  }
738 
739 #if HOST_WAKEUP_FIX_ENABLE
740  blueglob->BYTE4 |= 1<<7;
741  blueglob->BYTE22 = 0xF0;
742  blueglob->BYTE23 = 0xFF;
743 #endif
744 
745  /*Basic low level check with an extra margin of machine units */
746  if ((delay + radio_init_delay) < (radio_init_delay + 5))
747  {
748  TIMER_DISABLE_RADIO_TIMERS;
749  return 1;
750  }
751  TIMER_Context.last_setup_time = blue_unit_conversion(radio_init_delay, TIMER_Context.period1, MULT64_THR_PERIOD);
752  return 0;
753 }
