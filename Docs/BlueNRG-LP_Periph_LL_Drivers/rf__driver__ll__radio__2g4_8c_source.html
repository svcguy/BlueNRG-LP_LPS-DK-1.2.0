133 int8_t RADIO_ReadRSSI(void)
134 {
135  int32_t rssi_dbm;
136  uint32_t rssi0 = RRM->RSSI0_DIG_OUT;
137  uint32_t rssi1 = RRM->RSSI1_DIG_OUT;
138 
139  uint32_t rssi_int16 = ((rssi1&0xFF)<<8)|(rssi0&0xFF);
140  uint32_t reg_agc = RRM->AGC_DIG_OUT;
141 
142  if ((rssi_int16 == 0U) || (reg_agc > 0xbU))
143  {
144  rssi_dbm = 127 ;
145  }
146  else
147  {
148  rssi_dbm = (int32_t)reg_agc * 6 - RSSI_OFFSET;//127 ;
149  while (rssi_int16 > 30U)
150  {
151  rssi_dbm = rssi_dbm + 6 ;
152  rssi_int16 = (rssi_int16 >> 1) ;
153  }
154  rssi_dbm = rssi_dbm + (int32_t)(uint32_t)((417U*rssi_int16 + 18080U)>>10) ;
155  }
156  return (int8_t)rssi_dbm;
157 }

172 void RADIO_IRQHandler(void)
173 {
174  uint32_t int_value = BLUE->INTERRUPT1REG;
175 
176  /* clear interrupts */
177  BLUE->INTERRUPT1REG = int_value;
178 
179  if ( (int_value & BLUE_INTERRUPT1REG_DONE) != 0) {
180 
181  ActionPacket* next, *actionPacketBackup;
182  BlueTransStruct *p;
183  uint32_t time;
184 
185  /* Copy status in order for callback to access it. */
186  globalParameters.current_action_packet->status = int_value | \
187  (BLUE->STATUSREG & BLUE_STATUSREG_PREVTRANSMIT_Msk);
188 
189  if((globalParameters.current_action_packet->condRoutine(globalParameters.current_action_packet) ) == TRUE) {
190  next = globalParameters.current_action_packet->next_true;
191  }
192  else {
193  next = globalParameters.current_action_packet->next_false;
194  }
195  /*The radio event is started. So here a check on the next packet of the event is made*/
196  if(next == NULL_0) {
197  /* timer2 off */
198  TIMER_DISABLE_RADIO_TIMERS;
199  MODIFY_REG(blueglob->BYTE4,GLOBAL_BYTE4_ACTIVE_Msk,BLUE_IDLE_0);
200  }
201  else {
202  blueglob->BYTE4 = (next->StateMachineNo | GLOBAL_BYTE4_ACTIVE_Msk);
203  p= &next->trans_packet;
204  MODIFY_REG((bluedata + next->StateMachineNo)->BYTE0,STATEMACH_BYTE0_TXMODE_Msk,next->trans_config);/* Transmission or reception*/
205  (bluedata + next->StateMachineNo)->TXPOINT = BLUE_STRUCT_PTR_CAST(p);
206  (bluedata + next->StateMachineNo)->RCVPOINT = BLUE_STRUCT_PTR_CAST(p);
207  (bluedata + next->StateMachineNo)->MAXRECEIVEDLENGTH = next->MaxReceiveLength;
208 
209  /* Packet will execute on time specified by WakeupTime */
210  if((next->ActionTag & TIMER_WAKEUP ) !=0){
211  /* timer2 off */
212  TIMER_DISABLE_TIMER12;
213 
214  /* program timer at next->wakeuptime */
215  if((next->ActionTag & RELATIVE ) !=0) {
216  time = TIMER_GetCurrentSysTime() + TIMER_UsToSystime(next->WakeupTime);
217  HAL_VTIMER_SetRadioTimerValue(time,(next->trans_config == STATEMACH_BYTE0_TXMODE_Msk),(next->ActionTag & PLL_TRIG));
218  }
219  else {
220  HAL_VTIMER_SetRadioTimerValue(next->WakeupTime,(next->trans_config == STATEMACH_BYTE0_TXMODE_Msk),(next->ActionTag & PLL_TRIG));
221  }
222  }
223  else {
224  /* back to back */
225  }
226  }
227 
228  /* Accept the packet even with CRC Error */
229  if( ( (int_value & BLUE_INTERRUPT1REG_RCVOK) != 0) || ( (int_value & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) ) {
230  /* read RSSI */
231  globalParameters.current_action_packet->rssi = RADIO_ReadRSSI();
232 
233  /* bluedata->config = bluedata->config ^ 0x80 ; toggle NESN bit
234  bluedata->config = bluedata->config & 0x7F ; //reset NESN bit */
235 
236  /* read time stamp */
237  globalParameters.current_action_packet->timestamp_receive = TIMER_GetAnchorPoint();
238  }
239  else if( (int_value & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) {
240  /* read RSSI even if a timeout happens */
241  globalParameters.current_action_packet->rssi = RADIO_ReadRSSI();
242  }
243 
244  actionPacketBackup = globalParameters.current_action_packet;
245  globalParameters.current_action_packet = next;
246  actionPacketBackup->dataRoutine(actionPacketBackup, next);
247  }
248 
249  /* If the device is configured with
250  System clock = 64 MHz and BLE clock = 16 MHz
251  a register read is necessary to end fine
252  the clear interrupt register operation,
253  due the AHB down converter latency */
254  int_value = BLUE->INTERRUPT1REG;
255 
256  return ;
257 }

846  /* [EM:] Calculate the synt divide factor for 16 MHz quarts and +250 kHz offset wrt the channel center frequency
847  * Algorithmically MAK = F_rf*(2^20)/32 = F_rf*(2^15)
848  * With F_rf = (2402+2*RF_Channel)+0.25 MHz
849  * K corresponds to b19-b0 of MAK
850  * A corresponds to b22-b20 of MAK
851  * M corresponds to b27-b23 of MAK
852  */
853  uint32_t kHz_250_scaled = 8192; // = 0.250*2^20/32
854  uint32_t MAK = ((2402UL+2UL*RF_Channel)<<15);
855 
856  if (freq_offset == 1U)
857  {
858  MAK += kHz_250_scaled;
859  }
860  else if (freq_offset == 2U)
861  {
862  MAK -= kHz_250_scaled;
863  }
864  else
865  {
866  }
867 
868  uint8_t M = ((uint8_t)(MAK>>23)&0x1FU);
869  uint8_t A = ((uint8_t)(MAK>>20)&0x07U);
870  uint32_t K = (MAK&0x000FFFFFUL)+1UL;
871 
872  uint32_t MOD3_DIG_TEST = (M<<3) | (A & 0x7U);
873  uint32_t MOD2_DIG_TEST = (K>>12) & 0xFFU;
874  uint32_t MOD1_DIG_TEST = (K>> 4) & 0xFFU;
875  uint32_t MOD0_DIG_TEST = ((K&0x0FU)<<4) | 0x09U;
876  uint32_t RADIO_FSM_USR = ((powerLevel)<<3) | 0x06;
877 
878  RRM->RADIO_FSM_USR = RADIO_FSM_USR;
879  RRM->MOD3_DIG_TST = MOD3_DIG_TEST;
880  RRM->MOD2_DIG_TST = MOD2_DIG_TEST;
881  RRM->MOD1_DIG_TST = MOD1_DIG_TEST;
882  RRM->MOD0_DIG_TST = MOD0_DIG_TEST;
883 
884  /* Take control of the radio FSM through the test bus */
885  RRM->DTB5_DIG_ENG = 0x09;
886  uint32_t* rrm_udra_test = ((uint32_t *)&RRM->RRM_CTRL) + 0x01;
887  *rrm_udra_test = 0x03;
888  RRM->DTB5_DIG_ENG = 0x39;
889  while (RRM->FSM_STATUS_DIG_OUT != 0x04);
890  RRM->DTB5_DIG_ENG = 0x3B;
891  retVal = 0;
892  }
893  else
894  {
895  retVal = 1;
896  }
897  }
898  }
899  return retVal;
900 }
