112  assert_param(IS_LL_EXTI_TYPE(EXTI_InitStruct->Type));
113 
114  /* ENABLE LineCommand */
115  if (EXTI_InitStruct->LineCommand != DISABLE)
116  {
117  assert_param(IS_LL_EXTI_TRIGGER(EXTI_InitStruct->Trigger));
118 
119  /* Configure EXTI Lines in range from PA0 to PB15 */
120  if (EXTI_InitStruct->Line != LL_EXTI_LINE_NONE)
121  {
122  /* Set the IT Type on provided Lines */
123  LL_EXTI_SetType(EXTI_InitStruct->Type, EXTI_InitStruct->Line);
124 
125  if (EXTI_InitStruct->Trigger != LL_EXTI_TRIGGER_NONE)
126  {
127  /* Set the IT Trigger on provided Lines */
128  LL_EXTI_SetTrigger(EXTI_InitStruct->Trigger, EXTI_InitStruct->Line);
129  }
130  LL_EXTI_EnableIT(EXTI_InitStruct->Line);
131  LL_EXTI_ClearInterrupt(EXTI_InitStruct->Line);
132  }
133  }
134  /* DISABLE LineCommand */

137  LL_EXTI_DisableIT(EXTI_InitStruct->Line);
