118 ErrorStatus LL_DMA_DeInit(DMA_TypeDef *DMAx, uint32_t Channel)
119 {
120  DMA_Channel_TypeDef *tmp;
121  ErrorStatus status = SUCCESS;
122 
123  /* Check the DMA Instance DMAx and Channel parameters*/
124  assert_param(IS_LL_DMA_ALL_CHANNEL_INSTANCE(DMAx, Channel) || (Channel == LL_DMA_CHANNEL_ALL));
125 
126  if (Channel == LL_DMA_CHANNEL_ALL)
127  {
128  if (DMAx == DMA1)
129  {
130  /* Force reset of DMA clock */
131  LL_AHB_ForceReset(LL_AHB_PERIPH_DMA);
132 
133  /* Release reset of DMA clock */
134  LL_AHB_ReleaseReset(LL_AHB_PERIPH_DMA);
135  }
136  else
137  {
138  status = ERROR;
139  }
140  }
141  else
142  {
143  tmp = (DMA_Channel_TypeDef *)(__LL_DMA_GET_CHANNEL_INSTANCE(DMAx, Channel));
144 
145  /* Reset DMAx_Channely control register */
146  WRITE_REG(tmp->CCR, 0U);
147 
148  /* Reset DMAx_Channely remaining bytes register */
149  WRITE_REG(tmp->CNDTR, 0U);
150 
151  /* Reset DMAx_Channely peripheral address register */
152  WRITE_REG(tmp->CPAR, 0U);
153 
154  /* Reset DMAx_Channely memory address register */
155  WRITE_REG(tmp->CMAR, 0U);
156 
157  /* Reset Request register field for DMAx Channel */
158  LL_DMA_SetPeriphRequest(DMAx, Channel, LL_DMAMUX_REQ_MEM2MEM);
159 
160  if (Channel == LL_DMA_CHANNEL_1)
161  {
162  /* Reset interrupt pending bits for DMAx Channel1 */
163  LL_DMA_ClearFlag_GI1(DMAx);
164  }
165  else if (Channel == LL_DMA_CHANNEL_2)
166  {
167  /* Reset interrupt pending bits for DMAx Channel2 */
168  LL_DMA_ClearFlag_GI2(DMAx);
169  }
170  else if (Channel == LL_DMA_CHANNEL_3)
171  {
172  /* Reset interrupt pending bits for DMAx Channel3 */
173  LL_DMA_ClearFlag_GI3(DMAx);
174  }
175  else if (Channel == LL_DMA_CHANNEL_4)
176  {
177  /* Reset interrupt pending bits for DMAx Channel4 */
178  LL_DMA_ClearFlag_GI4(DMAx);
179  }
180  else if (Channel == LL_DMA_CHANNEL_5)
181  {
182  /* Reset interrupt pending bits for DMAx Channel5 */
183  LL_DMA_ClearFlag_GI5(DMAx);
184  }
185  else if (Channel == LL_DMA_CHANNEL_6)
186  {
187  /* Reset interrupt pending bits for DMAx Channel6 */
188  LL_DMA_ClearFlag_GI6(DMAx);
189  }
190  else if (Channel == LL_DMA_CHANNEL_7)
191  {
192  /* Reset interrupt pending bits for DMAx Channel7 */
193  LL_DMA_ClearFlag_GI7(DMAx);
194  }
195  else if (Channel == LL_DMA_CHANNEL_8)
196  {
197  /* Reset interrupt pending bits for DMAx Channel8 */
198  LL_DMA_ClearFlag_GI8(DMAx);
199  }
200  else
201  {
202  status = ERROR;
203  }
204  }
205 
206  return status;
207 }

229 ErrorStatus LL_DMA_Init(DMA_TypeDef *DMAx, uint32_t Channel, LL_DMA_InitTypeDef *DMA_InitStruct)
230 {
231  /* Check the DMA Instance DMAx and Channel parameters*/
232  assert_param(IS_LL_DMA_ALL_CHANNEL_INSTANCE(DMAx, Channel));
233 
234  /* Check the DMA parameters from DMA_InitStruct */
235  assert_param(IS_LL_DMA_DIRECTION(DMA_InitStruct->Direction));
236  assert_param(IS_LL_DMA_MODE(DMA_InitStruct->Mode));
237  assert_param(IS_LL_DMA_PERIPHINCMODE(DMA_InitStruct->PeriphOrM2MSrcIncMode));
238  assert_param(IS_LL_DMA_MEMORYINCMODE(DMA_InitStruct->MemoryOrM2MDstIncMode));
239  assert_param(IS_LL_DMA_PERIPHDATASIZE(DMA_InitStruct->PeriphOrM2MSrcDataSize));
240  assert_param(IS_LL_DMA_MEMORYDATASIZE(DMA_InitStruct->MemoryOrM2MDstDataSize));
241  assert_param(IS_LL_DMA_NBDATA(DMA_InitStruct->NbData));
242  assert_param(IS_LL_DMA_PERIPHREQUEST(DMA_InitStruct->PeriphRequest));
243  assert_param(IS_LL_DMA_PRIORITY(DMA_InitStruct->Priority));
244 
245  /*---------------------------- DMAx CCR Configuration ------------------------
246  * Configure DMAx_Channely: data transfer direction, data transfer mode,
247  * peripheral and memory increment mode,
248  * data size alignment and priority level with parameters :
249  * - Direction: DMA_CCR_DIR and DMA_CCR_MEM2MEM bits
250  * - Mode: DMA_CCR_CIRC bit
251  * - PeriphOrM2MSrcIncMode: DMA_CCR_PINC bit
252  * - MemoryOrM2MDstIncMode: DMA_CCR_MINC bit
253  * - PeriphOrM2MSrcDataSize: DMA_CCR_PSIZE[1:0] bits
254  * - MemoryOrM2MDstDataSize: DMA_CCR_MSIZE[1:0] bits
255  * - Priority: DMA_CCR_PL[1:0] bits
256  */
257  LL_DMA_ConfigTransfer(DMAx, Channel, DMA_InitStruct->Direction | \
258  DMA_InitStruct->Mode | \
259  DMA_InitStruct->PeriphOrM2MSrcIncMode | \
260  DMA_InitStruct->MemoryOrM2MDstIncMode | \
261  DMA_InitStruct->PeriphOrM2MSrcDataSize | \
262  DMA_InitStruct->MemoryOrM2MDstDataSize | \
263  DMA_InitStruct->Priority);
264 
265  /*-------------------------- DMAx CMAR Configuration -------------------------
266  * Configure the memory or destination base address with parameter :
267  * - MemoryOrM2MDstAddress: DMA_CMAR_MA[31:0] bits
268  */
269  LL_DMA_SetMemoryAddress(DMAx, Channel, DMA_InitStruct->MemoryOrM2MDstAddress);
270 
271  /*-------------------------- DMAx CPAR Configuration -------------------------
272  * Configure the peripheral or source base address with parameter :
273  * - PeriphOrM2MSrcAddress: DMA_CPAR_PA[31:0] bits
274  */
275  LL_DMA_SetPeriphAddress(DMAx, Channel, DMA_InitStruct->PeriphOrM2MSrcAddress);
276 
277  /*--------------------------- DMAx CNDTR Configuration -----------------------
278  * Configure the peripheral base address with parameter :
279  * - NbData: DMA_CNDTR_NDT[15:0] bits
280  */
281  LL_DMA_SetDataLength(DMAx, Channel, DMA_InitStruct->NbData);
282 
283  /*--------------------------- DMAMUXx CCR Configuration ----------------------
284  * Configure the DMA request for DMA Channels on DMAMUX Channel x with parameter :
285  * - PeriphRequest: DMA_CxCR[7:0] bits
286  */
287  LL_DMA_SetPeriphRequest(DMAx, Channel, DMA_InitStruct->PeriphRequest);
288 
289  return SUCCESS;
290 }
