495 __STATIC_INLINE void LL_GPIO_SetPinPull(GPIO_TypeDef *GPIOx, uint32_t Pin, uint32_t Pull)
496 {
497  /* To use the GPIO register configuration for PULL-UP PULL-DOWN and NO-PULL define USE_GPIO_PULL_REGISTER,
498  * otherwise the IO PULL will be managed from the PWR peripheral block
499  */
500 #ifdef USE_GPIO_PULL_REGISTER
501  MODIFY_REG(GPIOx->PUPDR, ((Pin * Pin) * GPIO_PUPDR_PUPD0), ((Pin * Pin) * Pull));
502 #else
503  if (GPIOx == GPIOA)
504  {
505  switch (Pull)
506  {
507  case LL_GPIO_PULL_NO:
508  LL_PWR_SetNoPullA(Pin);
509  break;
510  case LL_GPIO_PULL_UP:
511  LL_PWR_EnablePUA(Pin);
512  break;
513  case LL_GPIO_PULL_DOWN:
514  LL_PWR_EnablePDA(Pin);
515  break;
516  }
517  } else {
518  switch (Pull)
519  {
520  case LL_GPIO_PULL_NO:
521  LL_PWR_SetNoPullB(Pin);
522  break;
523  case LL_GPIO_PULL_UP:
524  LL_PWR_EnablePUB(Pin);
525  break;
526  case LL_GPIO_PULL_DOWN:
527  LL_PWR_EnablePDB(Pin);
528  break;
529  }
530  }
531 #endif
532 }

561 __STATIC_INLINE uint32_t LL_GPIO_GetPinPull(GPIO_TypeDef *GPIOx, uint32_t Pin)
562 {
563  /* To use the GPIO register configuration for PULL-UP PULL-DOWN and NO-PULL define USE_GPIO_PULL_REGISTER,
564  * otherwise the IO PULL will be managed from the PWR peripheral block
565  */
566 #ifdef USE_GPIO_PULL_REGISTER
567  return (uint32_t)(READ_BIT(GPIOx->PUPDR, ((Pin * Pin) * GPIO_PUPDR_PUPD0)) / (Pin * Pin));
568 #else
569  uint32_t pu, pd;
570 
571  if (GPIOx == GPIOA)
572  {
573  pu = LL_PWR_IsEnabledPUA(Pin);
574  pd = LL_PWR_IsEnabledPDA(Pin);
575  }
576  else
577  {
578  pu = LL_PWR_IsEnabledPUB(Pin);
579  pd = LL_PWR_IsEnabledPDB(Pin);
580  }
581 
582  if ((pu & Pin) == Pin)
583  {
584  return LL_GPIO_PULL_UP;
585  }
586 
587  if ((pd & Pin) == Pin)
588  {
589  return LL_GPIO_PULL_DOWN;
590  }
591 
592  return LL_GPIO_PULL_NO;
593 #endif
594 }
