577 __STATIC_INLINE uint8_t LL_EXTI_GetTrigger(uint32_t ExtiLine)
578 {
579  uint32_t level_setup, level_edge, both_edge;
580 
581  level_setup = READ_BIT(SYSCFG->IO_DTR, ExtiLine);
582  level_edge = READ_BIT(SYSCFG->IO_IEVR, ExtiLine);
583  both_edge = READ_BIT(SYSCFG->IO_IBER, ExtiLine);
584 
585  if (level_setup) {
586  if (level_edge) {
587  return LL_EXTI_TRIGGER_HIGH_LEVEL;
588  } else {
589  return LL_EXTI_TRIGGER_LOW_LEVEL;
590  }
591  } else {
592  if (both_edge) {
593  return LL_EXTI_TRIGGER_BOTH_EDGE;
594  }
595  if (level_edge) {
596  return LL_EXTI_TRIGGER_RISING_EDGE;
597  }
598  }
599  return LL_EXTI_TRIGGER_FALLING_EDGE;
600 }
