142  assert_param(IS_LL_I2C_TYPE_ACKNOWLEDGE(I2C_InitStruct->TypeAcknowledge));
143  assert_param(IS_LL_I2C_OWN_ADDRSIZE(I2C_InitStruct->OwnAddrSize));
144 
145  /* Disable the selected I2Cx Peripheral */
146  LL_I2C_Disable(I2Cx);
147 
148  /*---------------------------- I2Cx CR1 Configuration ------------------------
149  * Configure the analog and digital noise filters with parameters :
150  * - AnalogFilter: I2C_CR1_ANFOFF bit
151  * - DigitalFilter: I2C_CR1_DNF[3:0] bits
152  */
153  LL_I2C_ConfigFilters(I2Cx, I2C_InitStruct->AnalogFilter, I2C_InitStruct->DigitalFilter);
154 
155  /*---------------------------- I2Cx TIMINGR Configuration --------------------
156  * Configure the SDA setup, hold time and the SCL high, low period with parameter :
157  * - Timing: I2C_TIMINGR_PRESC[3:0], I2C_TIMINGR_SCLDEL[3:0], I2C_TIMINGR_SDADEL[3:0],
158  * I2C_TIMINGR_SCLH[7:0] and I2C_TIMINGR_SCLL[7:0] bits
159  */
160  LL_I2C_SetTiming(I2Cx, I2C_InitStruct->Timing);
161 
162  /* Enable the selected I2Cx Peripheral */
163  LL_I2C_Enable(I2Cx);
164 
165  /*---------------------------- I2Cx OAR1 Configuration -----------------------
166  * Disable, Configure and Enable I2Cx device own address 1 with parameters :
167  * - OwnAddress1: I2C_OAR1_OA1[9:0] bits
168  * - OwnAddrSize: I2C_OAR1_OA1MODE bit
169  */
170  LL_I2C_DisableOwnAddress1(I2Cx);
171  LL_I2C_SetOwnAddress1(I2Cx, I2C_InitStruct->OwnAddress1, I2C_InitStruct->OwnAddrSize);
172 
173  /* OwnAdress1 == 0 is reserved for General Call address */
174  if (I2C_InitStruct->OwnAddress1 != 0U)
175  {
176  LL_I2C_EnableOwnAddress1(I2Cx);
177  }
178 
179  /*---------------------------- I2Cx MODE Configuration -----------------------
180  * Configure I2Cx peripheral mode with parameter :
181  * - PeripheralMode: I2C_CR1_SMBDEN and I2C_CR1_SMBHEN bits
182  */
183  LL_I2C_SetMode(I2Cx, I2C_InitStruct->PeripheralMode);
184 
185  /*---------------------------- I2Cx CR2 Configuration ------------------------
186  * Configure the ACKnowledge or Non ACKnowledge condition
187  * after the address receive match code or next received byte with parameter :
188  * - TypeAcknowledge: I2C_CR2_NACK bit
189  */
190  LL_I2C_AcknowledgeNextData(I2Cx, I2C_InitStruct->TypeAcknowledge);
191 
192  return SUCCESS;
193 }
