693  CLEAR_BIT(SYSCFG->BLERXTX_IBER, Signal);

698  CLEAR_BIT(SYSCFG->BLERXTX_IBER, Signal);

721 __STATIC_INLINE uint8_t LL_SYSCFG_BLERXTX_GetTrigger(uint32_t Signal)
722 {
723  uint32_t level_setup, level_edge, both_edge;
724 
725  level_setup = READ_BIT(SYSCFG->BLERXTX_DTR, Signal);
726  level_edge = READ_BIT(SYSCFG->BLERXTX_IEVR, Signal);
727  both_edge = READ_BIT(SYSCFG->BLERXTX_IBER, Signal);
728 
729  if (level_setup) {
730  if (level_edge) {
731  return LL_SYSCFG_BLERXTX_TRIGGER_HIGH_LEVEL;
732  } else {
733  return LL_SYSCFG_BLERXTX_TRIGGER_LOW_LEVEL;
734  }
735  } else {
736  if (both_edge) {
737  return LL_SYSCFG_BLERXTX_TRIGGER_BOTH_EDGE;
738  }
739  if (level_edge) {
740  return LL_SYSCFG_BLERXTX_TRIGGER_RISING_EDGE;
741  }
742  }
743  return LL_SYSCFG_BLERXTX_TRIGGER_FALLING_EDGE;
