190 ErrorStatus LL_SPI_Init(SPI_TypeDef *SPIx, LL_SPI_InitTypeDef *SPI_InitStruct)
191 {
192  ErrorStatus status = ERROR;
193 
194  /* Check the SPI Instance SPIx*/
195  assert_param(IS_SPI_ALL_INSTANCE(SPIx));
196 
197  /* Check the SPI parameters from SPI_InitStruct*/
198  assert_param(IS_LL_SPI_TRANSFER_DIRECTION(SPI_InitStruct->TransferDirection));
199  assert_param(IS_LL_SPI_MODE(SPI_InitStruct->Mode));
200  assert_param(IS_LL_SPI_DATAWIDTH(SPI_InitStruct->DataWidth));
201  assert_param(IS_LL_SPI_POLARITY(SPI_InitStruct->ClockPolarity));
202  assert_param(IS_LL_SPI_PHASE(SPI_InitStruct->ClockPhase));
203  assert_param(IS_LL_SPI_NSS(SPI_InitStruct->NSS));
204  assert_param(IS_LL_SPI_BAUDRATE(SPI_InitStruct->BaudRate));
205  assert_param(IS_LL_SPI_BITORDER(SPI_InitStruct->BitOrder));
206  assert_param(IS_LL_SPI_CRCCALCULATION(SPI_InitStruct->CRCCalculation));
207 
208  if (LL_SPI_IsEnabled(SPIx) == 0x00000000U)
209  {
210  /*---------------------------- SPIx CR1 Configuration ------------------------
211  * Configure SPIx CR1 with parameters:
212  * - TransferDirection: SPI_CR1_BIDIMODE, SPI_CR1_BIDIOE and SPI_CR1_RXONLY bits
213  * - Master/Slave Mode: SPI_CR1_MSTR bit
214  * - ClockPolarity: SPI_CR1_CPOL bit
215  * - ClockPhase: SPI_CR1_CPHA bit
216  * - NSS management: SPI_CR1_SSM bit
217  * - BaudRate prescaler: SPI_CR1_BR[2:0] bits
218  * - BitOrder: SPI_CR1_LSBFIRST bit
219  * - CRCCalculation: SPI_CR1_CRCEN bit
220  */
221  MODIFY_REG(SPIx->CR1,
222  SPI_CR1_CLEAR_MASK,
223  SPI_InitStruct->TransferDirection | SPI_InitStruct->Mode |
224  SPI_InitStruct->ClockPolarity | SPI_InitStruct->ClockPhase |
225  SPI_InitStruct->NSS | SPI_InitStruct->BaudRate |
226  SPI_InitStruct->BitOrder | SPI_InitStruct->CRCCalculation);
227 
228  /*---------------------------- SPIx CR2 Configuration ------------------------
229  * Configure SPIx CR2 with parameters:
230  * - DataWidth: DS[3:0] bits
231  * - NSS management: SSOE bit
232  */
233  MODIFY_REG(SPIx->CR2,
234  SPI_CR2_DS | SPI_CR2_SSOE,
235  SPI_InitStruct->DataWidth | (SPI_InitStruct->NSS >> 16U));
236 
237  /*---------------------------- SPIx CRCPR Configuration ----------------------
238  * Configure SPIx CRCPR with parameters:
239  * - CRCPoly: CRCPOLY[15:0] bits
240  */
241  if (SPI_InitStruct->CRCCalculation == LL_SPI_CRCCALCULATION_ENABLE)
242  {
243  assert_param(IS_LL_SPI_CRC_POLYNOMIAL(SPI_InitStruct->CRCPoly));
244  LL_SPI_SetCRCPolynomial(SPIx, SPI_InitStruct->CRCPoly);
245  }
246  status = SUCCESS;
247  }
248 
249  /* Activate the SPI mode (Reset I2SMOD bit in I2SCFGR register) */
250  CLEAR_BIT(SPIx->I2SCFGR, SPI_I2SCFGR_I2SMOD);
251 
252 
253  return status;
254 }

380 ErrorStatus LL_I2S_Init(SPI_TypeDef *SPIx, LL_I2S_InitTypeDef *I2S_InitStruct)
381 {
382  uint32_t i2sdiv = 2U, i2sodd = 0U, packetlength = 1U;
383  uint32_t tmp;
384  uint32_t sourceclock, rcc_clock;
385  ErrorStatus status = ERROR;
386 
387  /* Check the I2S parameters */
388  assert_param(IS_I2S_ALL_INSTANCE(SPIx));
389  assert_param(IS_LL_I2S_MODE(I2S_InitStruct->Mode));
390  assert_param(IS_LL_I2S_STANDARD(I2S_InitStruct->Standard));
391  assert_param(IS_LL_I2S_DATAFORMAT(I2S_InitStruct->DataFormat));
392  assert_param(IS_LL_I2S_MCLK_OUTPUT(I2S_InitStruct->MCLKOutput));
393  assert_param(IS_LL_I2S_AUDIO_FREQ(I2S_InitStruct->AudioFreq));
394  assert_param(IS_LL_I2S_CPOL(I2S_InitStruct->ClockPolarity));
395 
396  if (LL_I2S_IsEnabled(SPIx) == 0x00000000U)
397  {
398  /*---------------------------- SPIx I2SCFGR Configuration --------------------
399  * Configure SPIx I2SCFGR with parameters:
400  * - Mode: SPI_I2SCFGR_I2SCFG[1:0] bit
401  * - Standard: SPI_I2SCFGR_I2SSTD[1:0] and SPI_I2SCFGR_PCMSYNC bits
402  * - DataFormat: SPI_I2SCFGR_CHLEN and SPI_I2SCFGR_DATLEN bits
403  * - ClockPolarity: SPI_I2SCFGR_CKPOL bit
404  */
405 
406  /* Write to SPIx I2SCFGR */
407  MODIFY_REG(SPIx->I2SCFGR,
408  I2S_I2SCFGR_CLEAR_MASK,
409  I2S_InitStruct->Mode | I2S_InitStruct->Standard |
410  I2S_InitStruct->DataFormat | I2S_InitStruct->ClockPolarity |
411  SPI_I2SCFGR_I2SMOD);
412 
413  /*---------------------------- SPIx I2SPR Configuration ----------------------
414  * Configure SPIx I2SPR with parameters:
415  * - MCLKOutput: SPI_I2SPR_MCKOE bit
416  * - AudioFreq: SPI_I2SPR_I2SDIV[7:0] and SPI_I2SPR_ODD bits
417  */
418 
419  /* If the requested audio frequency is not the default, compute the prescaler (i2sodd, i2sdiv)
420  * else, default values are used: i2sodd = 0U, i2sdiv = 2U.
421  */
422  if (I2S_InitStruct->AudioFreq != LL_I2S_AUDIOFREQ_DEFAULT)
423  {
424  /* Check the frame length (For the Prescaler computing)
425  * Default value: LL_I2S_DATAFORMAT_16B (packetlength = 1U).
426  */
427  if (I2S_InitStruct->DataFormat != LL_I2S_DATAFORMAT_16B)
428  {
429  /* Packet length is 32 bits */
430  packetlength = 2U;
431  }
432 
433  /* Get the source clock value: based on System Clock value */
434 #if defined(SPI2)
435  if (SPIx == SPI2) {
436  rcc_clock = LL_RCC_GetSPI2I2SClockSource();
437  if (rcc_clock == LL_RCC_SPI2_I2S_CLK16M) {
438  sourceclock = 16000000;
439  } else {
440  sourceclock = 32000000;
441  }
442  }
443 #endif
444 #if defined(SPI3)
445  if (SPIx == SPI3) {
446  rcc_clock = LL_RCC_GetSPI3I2SClockSource();
447  if (rcc_clock == LL_RCC_SPI3_I2S_CLK16M) {
448  sourceclock = 16000000;
449  } else {
450  sourceclock = 32000000;
451  }
452  }
453 #endif
454 
455  /* Compute the Real divider depending on the MCLK output state with a floating point */
456  if (I2S_InitStruct->MCLKOutput == LL_I2S_MCLK_OUTPUT_ENABLE)
457  {
458  /* MCLK output is enabled */
459  tmp = (((((sourceclock / 256U) * 10U) / I2S_InitStruct->AudioFreq)) + 5U);
460  }
461  else
462  {
463  /* MCLK output is disabled */
464  tmp = (((((sourceclock / (32U * packetlength)) * 10U) / I2S_InitStruct->AudioFreq)) + 5U);
465  }
466 
467  /* Remove the floating point */
468  tmp = tmp / 10U;
469 
470  /* Check the parity of the divider */
471  i2sodd = (tmp & (uint16_t)0x0001U);
472 
473  /* Compute the i2sdiv prescaler */
474  i2sdiv = ((tmp - i2sodd) / 2U);
475 
476  /* Get the Mask for the Odd bit (SPI_I2SPR[8]) register */
477  i2sodd = (i2sodd << 8U);
478  }
479 
480  /* Test if the divider is 1 or 0 or greater than 0xFF */
481  if ((i2sdiv < 2U) || (i2sdiv > 0xFFU))
482  {
483  /* Set the default values */
484  i2sdiv = 2U;
485  i2sodd = 0U;
486  }
487 
488  /* Write to SPIx I2SPR register the computed value */
489  WRITE_REG(SPIx->I2SPR, i2sdiv | i2sodd | I2S_InitStruct->MCLKOutput);
490 
491  status = SUCCESS;
492  }
493  return status;
494 }
