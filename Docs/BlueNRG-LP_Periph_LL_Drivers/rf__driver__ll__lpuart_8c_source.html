166 ErrorStatus LL_LPUART_Init(USART_TypeDef *LPUARTx, LL_LPUART_InitTypeDef *LPUART_InitStruct)
167 {
168  ErrorStatus status = ERROR;
169 
170  /* Check the parameters */
171  assert_param(IS_LPUART_INSTANCE(LPUARTx));
172  assert_param(IS_LL_LPUART_PRESCALER(LPUART_InitStruct->PrescalerValue));
173  assert_param(IS_LL_LPUART_BAUDRATE(LPUART_InitStruct->BaudRate));
174  assert_param(IS_LL_LPUART_DATAWIDTH(LPUART_InitStruct->DataWidth));
175  assert_param(IS_LL_LPUART_STOPBITS(LPUART_InitStruct->StopBits));
176  assert_param(IS_LL_LPUART_PARITY(LPUART_InitStruct->Parity));
177  assert_param(IS_LL_LPUART_DIRECTION(LPUART_InitStruct->TransferDirection));
178  assert_param(IS_LL_LPUART_HWCONTROL(LPUART_InitStruct->HardwareFlowControl));
179 
180  /* LPUART needs to be in disabled state, in order to be able to configure some bits in
181  CRx registers. Otherwise (LPUART not in Disabled state) => return ERROR */
182  if (LL_LPUART_IsEnabled(LPUARTx) == 0U)
183  {
184  /*---------------------------- LPUART CR1 Configuration -----------------------
185  * Configure LPUARTx CR1 (LPUART Word Length, Parity and Transfer Direction bits) with parameters:
186  * - DataWidth: USART_CR1_M bits according to LPUART_InitStruct->DataWidth value
187  * - Parity: USART_CR1_PCE, USART_CR1_PS bits according to LPUART_InitStruct->Parity value
188  * - TransferDirection: USART_CR1_TE, USART_CR1_RE bits according to LPUART_InitStruct->TransferDirection value
189  */
190  MODIFY_REG(LPUARTx->CR1,
191  (USART_CR1_M | USART_CR1_PCE | USART_CR1_PS | USART_CR1_TE | USART_CR1_RE),
192  (LPUART_InitStruct->DataWidth | LPUART_InitStruct->Parity | LPUART_InitStruct->TransferDirection));
193 
194  /*---------------------------- LPUART CR2 Configuration -----------------------
195  * Configure LPUARTx CR2 (Stop bits) with parameters:
196  * - Stop Bits: USART_CR2_STOP bits according to LPUART_InitStruct->StopBits value.
197  */
198  LL_LPUART_SetStopBitsLength(LPUARTx, LPUART_InitStruct->StopBits);
199 
200  /*---------------------------- LPUART CR3 Configuration -----------------------
201  * Configure LPUARTx CR3 (Hardware Flow Control) with parameters:
202  * - HardwareFlowControl: USART_CR3_RTSE, USART_CR3_CTSE bits according to LPUART_InitStruct->HardwareFlowControl value.
203  */
204  LL_LPUART_SetHWFlowCtrl(LPUARTx, LPUART_InitStruct->HardwareFlowControl);
205 
206  /*---------------------------- LPUART BRR Configuration -----------------------
207  * Configure the LPUART Baud Rate :
208  * - prescaler value is required
209  * - valid baud rate value (different from 0) is required
210  */
211  if (LPUART_InitStruct->BaudRate != 0U)
212  {
213  status = SUCCESS;
214  LL_LPUART_SetBaudRate(LPUARTx,
215  LPUART_InitStruct->PrescalerValue,
216  LPUART_InitStruct->BaudRate);
217 
218  /* Check BRR is greater than or equal to 0x300 */
219  assert_param(IS_LL_LPUART_BRR_MIN(LPUARTx->BRR));
220 
221  /* Check BRR is lower than or equal to 0xFFFFF */
222  assert_param(IS_LL_LPUART_BRR_MAX(LPUARTx->BRR));
223  }
224 
225  /*---------------------------- LPUART PRESC Configuration -----------------------
226  * Configure LPUARTx PRESC (Prescaler) with parameters:
227  * - PrescalerValue: LPUART_PRESC_PRESCALER bits according to LPUART_InitStruct->PrescalerValue value.
228  */
229  LL_LPUART_SetPrescaler(LPUARTx, LPUART_InitStruct->PrescalerValue);
230  }
231 
232  return (status);
233 }
