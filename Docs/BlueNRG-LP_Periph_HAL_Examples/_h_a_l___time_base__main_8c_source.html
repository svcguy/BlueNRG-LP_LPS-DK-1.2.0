287 HAL_StatusTypeDef HAL_InitTick (uint32_t TickPriority)
288 {
289  RCC_ClkInitTypeDef clkconfig;
290  uint32_t uwPrescalerValue = 0;
291  uint32_t pFLatency;
292 
293  /*Configure the TIMx IRQ priority */
294  HAL_NVIC_SetPriority(TIMx_IRQn, TickPriority);
295 
296  /* Enable the TIMx global Interrupt */
297  HAL_NVIC_EnableIRQ(TIMx_IRQn);
298 
299  /* Enable TIMx clock */
300  EnableClock_TIMx();
301 
302  /* Get clock configuration */
303  HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
304 
305  /* Compute the prescaler value to have TIMx counter clock equal to 1MHz */
306  uwPrescalerValue = (uint32_t) ((HAL_TIM_GetPeriphClock(TIMx) / 1000000) - 1);
307 
308  /* Initialize TIMx peripheral as follow:
309  + Prescaler = (HAL_TIM_GetPeriphClock(TIMx)/1000000 - 1) to have a 1MHz TIMx counter clock.
310  + Period = (TIMx counter clock / TIMx output clock) - 1. to have a (1/1000) s time base.
311  + ClockDivision = 0
312  + Counter direction = Up
313  */
314  htimx.Instance = TIMx;
315  htimx.Init.Period = (1000000 / 1000) - 1;
316  htimx.Init.Prescaler = uwPrescalerValue;
317  htimx.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
318  htimx.Init.CounterMode = TIM_COUNTERMODE_UP;
319  if(HAL_TIM_Base_Init(&htimx) != HAL_OK)
320  {
321  /* Initialization Error */
322  Error_Handler();
323  }
324 
325  /* Start the TIM time Base generation in interrupt mode */
326  if(HAL_TIM_Base_Start_IT(&htimx) != HAL_OK)
327  {
328  /* Starting Error */
329  Error_Handler();
330  }
331 
332  /* Return function status */
333  return HAL_OK;
334 }
