449 void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
450 {
451  if (htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1)
452  {
453  if(uhCaptureIndex == 0)
454  {
455  /* Get the 1st Input Capture value */
456  uwIC2Value1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
457  uhCaptureIndex = 1;
458  }
459  else if(uhCaptureIndex == 1)
460  {
461  /* Get the 2nd Input Capture value */
462  uwIC2Value2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
463 
464  /* Capture computation */
465  if (uwIC2Value2 > uwIC2Value1)
466  {
467  uwDiffCapture = (uwIC2Value2 - uwIC2Value1);
468  }
469  else if (uwIC2Value2 < uwIC2Value1)
470  {
471  /* 0xFFFF is max TIMx_CCRx value */
472  uwDiffCapture = ((0xFFFF - uwIC2Value1) + uwIC2Value2) + 1;
473  }
474  else
475  {
476  /* If capture values are equal, we have reached the limit of frequency
477  measures */
478  Error_Handler();
479  }
480 
481  /* Frequency computation: for this example TIMx (TIMx) is clocked by
482  64 MHz Clk */
483  inputFrequency = HAL_TIM_GetPeriphClock(htim1.Instance) / uwDiffCapture;
484  uhCaptureIndex = 0;
485  }
486  }
487 }

514 void CentralizedCalcForAsynchSynchPrescaler(uint32_t inputFrequency){
515 
516 #if USE_SOLUTION_NUMBER == 1
517  AsynchPrescaler = (inputFrequency / uwCkApreFreq ) - 1;
518  // AsynchPrediv has only 7bit of range
519  if(AsynchPrescaler > 0x7F)
520  {
521  AsynchPrescaler = 0x7F;
522  }
523  SynchPrescaler = (inputFrequency / (AsynchPrescaler + 1)) - 1;
524  // SynchPrediv has only 15bit of range
525  if(SynchPrescaler > 0x7FFF)
526  {
527  SynchPrescaler = 0x7FFF;
528  printf("wrong PREDIV_S value\n\r");
529  }
530 #endif
531 
532 #if USE_SOLUTION_NUMBER == 2
533  AsynchPrescaler = 1;
534  SynchPrescaler = (inputFrequency / (AsynchPrescaler + 1)) - 1;
535  // SynchPrediv has only 15bit of range
536  if(SynchPrescaler > 0x7FFF)
537  {
538  SynchPrescaler = 0x7FFF;
539  printf("wrong PREDIV_S value\n\r");
540  }
541 #endif
542 
543 #if USE_SOLUTION_NUMBER == 3
544  // algorith to find the best couple of a_prediv s_prediv to have 1Hz output
545  __IO int32_t temp = 0;
546  __IO int32_t gap = (int32_t) inputFrequency; // gap value could be from 0 (the best) to f_rtc_ck (the wrost)
547 
548  for(__IO int8_t prediv_a = 0x7F; prediv_a>0x0; prediv_a--)
549  {
550  for(__IO int16_t prediv_s = 0x7FF; prediv_s>0x0; prediv_s--)
551  {
552  temp = ABS( ((prediv_a + 1) * (prediv_s + 1)) - ((int32_t)inputFrequency) );
553  if(temp<gap)
554  {
555  gap = temp;
556  AsynchPrescaler = prediv_a;
557  SynchPrescaler = prediv_s;
558  }
559  }
560  }
561 #endif
562 
563 }
