741 {
742  if(AddrMatchCode == I2C_ADDRESS)
743  {
744  uwTransferInitiated = 1;

746 
747  /* First of all, check the transfer direction to call the correct Slave Interface */
748  if(uwTransferDirection == I2C_DIRECTION_TRANSMIT)
749  {
750  if(HAL_I2C_Slave_Seq_Receive_IT(&hi2cx, &aSlaveReceiveBuffer[ubSlaveReceiveIndex], 1, I2C_FIRST_FRAME) != HAL_OK)
751  {
752  Error_Handler();

754  ubSlaveReceiveIndex++;
755  }
756  else
757  {
758  pSlaveTransmitBuffer = (uint8_t*)(aSlaveInfo[ubSlaveInfoIndex]);

761  if(HAL_I2C_Slave_Seq_Transmit_IT(&hi2cx, pSlaveTransmitBuffer, ubSlaveNbDataToTransmit, I2C_LAST_FRAME) != HAL_OK)
762  {
763  Error_Handler();
764  }
765  }
766  }
