258 static void EXTI10_IRQHandler_Config(void)

265  __HAL_RCC_GPIOA_CLK_ENABLE();
266  __HAL_RCC_SYSCFG_CLK_ENABLE();
267 
268  /* Configure PA.10 pin as input floating */
269  GPIO_InitStructure.Mode = GPIO_MODE_INPUT;
270  GPIO_InitStructure.Pull = GPIO_NOPULL;
271  GPIO_InitStructure.Pin = GPIO_PIN_10;
272  HAL_GPIO_Init(GPIOA, &GPIO_InitStructure);
273 
274  EXTI_Config_InitStructure.Line = EXTI_LINE_PA10;
