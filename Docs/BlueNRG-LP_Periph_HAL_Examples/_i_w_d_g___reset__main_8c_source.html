309 static void MX_IWDG_Init(void)
310 {
311  /* WDG clock enable */
312  __HAL_RCC_WDG_CLK_ENABLE();
313 
314  /* Force WDG peripheral reset */
315  __HAL_RCC_WDG_FORCE_RESET();
316  __HAL_RCC_WDG_RELEASE_RESET();
317 
318  /* Check if WDG Reset Release flag interrupt occurred or not */
319  while(__HAL_RCC_GET_IT(RCC_IT_WDGRSTRELRDY) == 0)
320  {
321  }
322 
323  hiwdg.Instance = IWDG;
324  hiwdg.Init.Prescaler = IWDG_PRESCALER;
325  hiwdg.Init.Window = IWDG_WINDOW;
326  hiwdg.Init.Reload = IWDG_RELOAD;
327  if (HAL_IWDG_Init(&hiwdg) != HAL_OK)
328  {
329  Error_Handler();
330  }
331 }
