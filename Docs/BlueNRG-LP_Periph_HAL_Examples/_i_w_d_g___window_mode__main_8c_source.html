324 static void MX_IWDG_Init(void)
325 {
326  /* WDG clock enable */
327  __HAL_RCC_WDG_CLK_ENABLE();
328 
329  /* Force WDG peripheral reset */
330  __HAL_RCC_WDG_FORCE_RESET();
331  __HAL_RCC_WDG_RELEASE_RESET();
332 
333  /* Check if WDG Reset Release flag interrupt occurred or not */
334  while(__HAL_RCC_GET_IT(RCC_IT_WDGRSTRELRDY) == 0)
335  {
336  }

338  hiwdg.Instance = IWDG;
339  hiwdg.Init.Prescaler = IWDG_PRESCALER_16;
340  hiwdg.Init.Window = IWDG_WINDOW;
341  hiwdg.Init.Reload = IWDG_RELOAD;
342  if (HAL_IWDG_Init(&hiwdg) != HAL_OK)
