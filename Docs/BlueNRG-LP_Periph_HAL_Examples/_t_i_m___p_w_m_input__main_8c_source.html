381 void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
382 {
383  if (htim->Channel == HAL_TIM_ACTIVE_CHANNEL_2)
384  {
385  /* Get the Input Capture value */
386  uwIC2Value = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_2);
387 
388  if (uwIC2Value != 0)
389  {
390  /* LED2 toggle during the Input Capture value */
391  BSP_LED_Toggle(BSP_LED2);
392 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
393  /* Duty cycle computation */
394  uwDutyCycle = ((HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1)) * 100) / uwIC2Value;
395 #endif
396  /* uwFrequency computation */
397  uwFrequency = (uint32_t) ( HAL_TIM_GetPeriphClock(htimx.Instance) / uwIC2Value );
398  }
399  else
400  {
401 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
402  uwDutyCycle = 0;
403 #endif
404  uwFrequency = 0;
405  }
406  }
407 }
