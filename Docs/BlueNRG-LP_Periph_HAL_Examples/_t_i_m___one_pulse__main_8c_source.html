245  /* 2 Compute the prescaler value, to have TIM1Freq = 1000000 Hz
246  * TIM1CLK = HAL_TIM_GetPeriphClock(TIMx)
247  *
248  * Prescaler = (TIM1CLK /TIM1 counter clock) - 1
249  *
250  * The prescaler value is computed in order to have TIM1 counter clock

252  */
253  uwPrescalerValue = (uint32_t)((HAL_TIM_GetPeriphClock(TIMx)) / 1000000) - 1;

255  /* 3 Configure the TIM peripheral
256  *
257  *-The external signal is connected to TIM1_CH2 ,
258  * and a falling edge on this input is used to trigger the Timer.
259  *
260  *-The One Pulse signal is output on TIM1_CH1.
261  *
262  * The delay value is fixed to:
263  * - Delay = CCR1/TIM1 counter clock
264  * = 16383 / 1000000 [sec]
265  *
266  * The pulse value is fixed to :
267  * - Pulse value = (TIM_Period - TIM_Pulse)/TIM1 counter clock
268  * = (65535 - 16383) / 1000000 [sec]
269  *
270  **/
271 
272  TimHandle.Instance = TIMx;
273  TimHandle.Init.Period = 0xFFFF;
274  TimHandle.Init.Prescaler = uwPrescalerValue;
275  TimHandle.Init.ClockDivision = 0;
276  TimHandle.Init.CounterMode = TIM_COUNTERMODE_UP;
277  TimHandle.Init.RepetitionCounter = 0;
278  TimHandle.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
279

291  sConfig.ICSelection = TIM_ICSELECTION_DIRECTTI;
