294 static void MX_I2S_Init(void)
295 {
296  /* Disable I2S peripheral to allow access to I2S internal registers */
297  __HAL_I2S_DISABLE(&hi2s);
298 
299  /* Configure I2S */
300  hi2s.Instance = I2S_INSTANCE;
301  hi2s.Init.AudioFreq = I2S_AUDIOFREQ_32K;
302  hi2s.Init.CPOL = I2S_CPOL_LOW;
303  hi2s.Init.DataFormat = I2S_DATAFORMAT_16B;
304  // uncomment to use MCK - MCK enbable
305 // hi2s.Init.MCLKOutput = I2S_MCLKOUTPUT_ENABLE;
306  // comment to use MCK - MCK enbable
307  hi2s.Init.MCLKOutput = I2S_MCLKOUTPUT_DISABLE;
308  hi2s.Init.Mode = I2S_MODE_SLAVE_TX;
309  hi2s.Init.Standard = I2S_STANDARD_PHILIPS;
310 
311  I2S_CLK_ENABLE();
312 
313  /* Configure SPIxI2S clock source */
314  I2S_CLK_CONFIG();
315 
316  /* Initialization I2S function */
317  if(HAL_I2S_Init(&hi2s) != HAL_OK)
318  {
319  printf("Initializes the I2S ended with error.\n\r");
320  Error_Handler();
321  }
322 
323  /* Disable I2S peripheral */
324  __HAL_I2S_ENABLE(&hi2s);
325 }
