373 void StartReception(void)

377  */
378  pBufferReadyForReception = aRXBufferA;

384 
385 
386  if (HAL_OK != HAL_UART_Receive_DMA(&huart1, aRXBufferUser, RX_BUFFER_SIZE))
387  {
