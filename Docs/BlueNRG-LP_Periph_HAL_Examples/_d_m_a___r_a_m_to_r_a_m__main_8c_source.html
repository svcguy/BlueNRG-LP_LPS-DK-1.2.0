257 int main(void)
258 {
259  /* System initialization function */
260  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
261  {
262  /* Error during system clock configuration take appropriate action */
263  while(1);
264  }
265 
266  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
267  HAL_Init();
268 
269 #if defined(CONFIG_DEVICE_BLUENRG_LP) || defined(CONFIG_DEVICE_BLUENRG_LPS)
270  /* IO pull configuration with minimum power consumption */
271  BSP_IO_Init();
272 #endif
273 
274  /* Initialization of COM port */
275  BSP_COM_Init(NULL);
276 
277  printf("Initial buffer data:\n\r");
278  PrintBuffer((uint32_t*)aSRC_Buffer, BUFFER_SIZE);
279  PrintBuffer((uint32_t*)aDST_Buffer, BUFFER_SIZE);
280 
281  /* Initialize all configured peripherals */
282  MX_DMA_Init();
283 
284  /* Initialize LEDs */
285  BSP_LED_Init(BSP_LED2);
286  BSP_LED_Init(BSP_LED1);
287  BSP_LED_Init(BSP_LED3);
288 
289  /* Infinite loop */
290  /* Reset transferErrorDetected to 0, it will be set to 1 if a transfer error is detected */
291  transferErrorDetected = 0;
292  /* Reset transferCompleteDetected to 0, it will be set to 1 if a transfer is correctly completed */
293  transferCompleteDetected = 0;
294 
295  /* Select Callbacks functions called after Transfer complete and Transfer error */
296  HAL_DMA_RegisterCallback(&hdma_memtomem_dma1_channel1, HAL_DMA_XFER_CPLT_CB_ID, TransferComplete);
297  HAL_DMA_RegisterCallback(&hdma_memtomem_dma1_channel1, HAL_DMA_XFER_ERROR_CB_ID, TransferError);
298 
299  /* Configure the source, destination and buffer size DMA fields and Start DMA Channel/Stream transfer */
300  /* Enable All the DMA interrupts */
301  if (HAL_DMA_Start_IT(&hdma_memtomem_dma1_channel1, (uint32_t)&aSRC_Buffer, (uint32_t)&aDST_Buffer, BUFFER_SIZE) != HAL_OK)
302  {
303  /* Transfer Error */
304  Error_Handler();
305  }
306 
307  /* Infinite loop */
308  while (1)
309  {
310  if (transferErrorDetected == 1)
311  {
312  /* Turn LED1 on*/
313  BSP_LED_On(BSP_LED1);
314  transferErrorDetected = 0;
315  HAL_Delay(1000);
316  BSP_LED_Off(BSP_LED3);
317  }
318  if (transferCompleteDetected == 1)
319  {
320  CompareBuffer();
321  transferCompleteDetected = 0;
322  HAL_Delay(1000);
323  BSP_LED_Off(BSP_LED3);
324  }
325  }
326 }
