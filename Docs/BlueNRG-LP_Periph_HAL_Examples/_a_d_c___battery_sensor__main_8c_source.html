310  /* Infinite loop */

313  /* Check the ADC flag End Of Down Sampler conversion */
314  if( HAL_ADC_PollForConversion(&hadc, 100) != HAL_OK) {
315 
316  /* Get the battery raw value from the Down Sampler */
317  nVBattRawVal = HAL_ADC_GetValue(&hadc);
318  adcValue = (float)LL_ADC_GetADCConvertedValueBatt(hadc.Instance, nVBattRawVal, USER_DATAWIDTH, offset_vinm0);
319 
320  /* Printout the output value */
