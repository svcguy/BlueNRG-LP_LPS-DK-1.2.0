252  MX_RNG_Init();

255  BSP_PB_Init(BSP_PUSH1, BUTTON_MODE_EXTI);
256 
257  /* Generate eight 16-bit long random numbers */
258  printf("Generate eight 16-bit long random numbers.\n\r");
259  for (counter = 0; counter < 8; counter++)
260  {
261  /* Wait for event on push button to perform following actions */
262  printf("Wait for event on push button.\n\r");
263  while ((ubUserButtonClickEvent) == RESET)
264  {
265  BSP_LED_Toggle(BSP_LED2);
266  HAL_Delay(200);
267  }
268 
269  /* Reset variable for next loop iteration */
270  ubUserButtonClickEvent = RESET;
271 
272  if (HAL_RNG_GenerateRandomNumber(&hrng, &aRandom[counter]) != HAL_OK)
273  {
274  /* Random number generation error */
275  Error_Handler();
276  }
277  else
278  {
279  printf("Value of generated random number: %#X\n\r", aRandom[counter]);
280  }
281  }
282 
283  printf("End of the example.\n\r");
284

287  Error_Handler();
288  }
289 
290  BSP_LED_Off(BSP_LED2);
