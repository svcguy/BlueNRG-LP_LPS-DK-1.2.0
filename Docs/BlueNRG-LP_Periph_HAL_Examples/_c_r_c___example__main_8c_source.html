291  printf("\n\r");
292

294  if (uwCRCValue != uwExpectedCRCValue)
295  {
296  /* Wrong CRC value: Turn LED3 on */
297  printf("Wrong CRC value.\n\r");
298  Error_Handler();
299  }

303  printf("Right CRC value.\n\r");
