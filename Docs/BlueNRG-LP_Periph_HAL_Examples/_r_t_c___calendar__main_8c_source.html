286 static void MX_RTC_Init(void)
287 {
288  RTC_TimeTypeDef sTime = {0};
289  RTC_DateTypeDef sDate = {0};
290 
291  /* Set Date and Time (if not already done before)*/
292  /* Read the Back Up Register 0 Data */
293  if (HAL_RTCEx_BKUPRead(&hrtc, RTC_BKP_DR0) != RTC_BKP_DATA_VALUE)
294  {
295  /* RTC clock enable */
296  __HAL_RCC_RTC_CLK_ENABLE();
297 
298  __HAL_RCC_CLEAR_IT(RCC_IT_RTCRSTRELRDY);
299  /* Force RTC peripheral reset */
300  __HAL_RCC_RTC_FORCE_RESET();
301  __HAL_RCC_RTC_RELEASE_RESET();
302  /* Check if RTC Reset Release flag interrupt occurred or not */
303  while(__HAL_RCC_GET_IT(RCC_IT_RTCRSTRELRDY) == 0)
304  {
305  }
306  __HAL_RCC_CLEAR_IT(RCC_IT_RTCRSTRELRDY);
307 
309  hrtc.Instance = RTC;
310  hrtc.Init.HourFormat = RTC_HOURFORMAT_24;
311  hrtc.Init.AsynchPrediv = RTC_ASYNCH_PREDIV;
312  hrtc.Init.SynchPrediv = RTC_SYNCH_PREDIV;
313  hrtc.Init.OutPut = RTC_OUTPUT_DISABLE;
314  hrtc.Init.OutPutPolarity = RTC_OUTPUT_POLARITY_HIGH;
315  if (HAL_RTC_Init(&hrtc) != HAL_OK)
316  {
317  Error_Handler();
318  }
319  /* Configure RTC Calendar */
320  printf("Configure RTC Calendar\n\r");
321 
323  sTime.Hours = 0x0;
324  sTime.Minutes = 0x59;
325  sTime.Seconds = 0x50;
326  sTime.DayLightSaving = RTC_DAYLIGHTSAVING_NONE;
327  sTime.StoreOperation = RTC_STOREOPERATION_RESET;
328  if (HAL_RTC_SetTime(&hrtc, &sTime, RTC_FORMAT_BCD) != HAL_OK)
329  {
330  Error_Handler();
331  }
332  sDate.WeekDay = RTC_WEEKDAY_MONDAY;
333  sDate.Month = RTC_MONTH_JANUARY;
334  sDate.Date = 0x01;
335  sDate.Year = 0x20;
336 
337  if (HAL_RTC_SetDate(&hrtc, &sDate, RTC_FORMAT_BCD) != HAL_OK)
338  {
339  Error_Handler();
340  }
341 
342  /* Writes a data in a RTC Backup data Register0 */
343  HAL_RTCEx_BKUPWrite(&hrtc, RTC_BKP_DR0, RTC_BKP_DATA_VALUE);
344  }

357 static void RTC_CalendarShow(uint8_t *showtime, uint8_t *showdate)
358 {
359  RTC_DateTypeDef sdatestructureget;
360  RTC_TimeTypeDef stimestructureget;
361 
362  /* Get the RTC current Time */
363  HAL_RTC_GetTime(&hrtc, &stimestructureget, RTC_FORMAT_BIN);
364  /* Get the RTC current Date */
365  HAL_RTC_GetDate(&hrtc, &sdatestructureget, RTC_FORMAT_BIN);
366  /* Display time Format : hh:mm:ss */
367  sprintf((char *)showtime, "%2d:%2d:%2d", stimestructureget.Hours, stimestructureget.Minutes, stimestructureget.Seconds);
368  /* Display date Format : mm-dd-yy */
369  sprintf((char *)showdate, "%2d-%2d-%2d", sdatestructureget.Month, sdatestructureget.Date, 2000 + sdatestructureget.Year);
370 }
