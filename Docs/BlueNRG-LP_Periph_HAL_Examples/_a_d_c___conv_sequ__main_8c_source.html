296 {
297  /* Enable the ADC peripheral */
298  HAL_ADC_StructInit(&adc_handle);
299  adc_handle.Init.DataRatio = USER_RATIO;
300  adc_handle.Init.DataWidth = USER_DATAWIDTH;
301  adc_handle.Init.SampleRate = USER_SAMPLERATE;

303  adc_handle.Init.SampleRateMsb = USER_SAMPLERATE_MSB;
304 #endif
305  adc_handle.Init.SamplingMode = ADC_SAMPLING_AT_START;
306  adc_handle.Init.SequenceLength = ADC_SEQ_LEN_03;
307  adc_handle.DMA_Handle = &hdma_adc;
308 
309  if (HAL_ADC_Init(&adc_handle) != HAL_OK) {
310  Error_Handler();
311  }
312 
313  /* Start the sampling at end of previous sample */
314  LL_ADC_InputSamplingMode(ADC, LL_ADC_SAMPLING_AT_END);
315 
316  /* Set the input channel */
317  xChannel.ChannelType = ADC_CH_VINP0_TO_SINGLE_POSITIVE_INPUT;
318  xChannel.SequenceNumber = ADC_SEQ_POS_01;
319  xChannel.VoltRange = ADC_VIN_RANGE_3V6;
320  if (HAL_ADC_ConfigChannel(&adc_handle, &xChannel)!= HAL_OK) {
321  Error_Handler();
322  }
323 
324  /* Set the GAIN */
325  if(LL_ADC_GET_CALIB_GAIN_FOR_VINPX_3V6() != 0xFFF) {
326  LL_ADC_SetCalibPoint1Gain(ADC, LL_ADC_GET_CALIB_GAIN_FOR_VINPX_3V6() );
327 
328  offset_vinp0 = LL_ADC_GET_CALIB_OFFSET_FOR_VINPX_3V6();
329 #ifdef CONFIG_DEVICE_BLUENRG_LP

331  LL_ADC_SetCalibPoint1Offset(ADC, 0);
332  }
333  else
334 #endif
335  {
336  LL_ADC_SetCalibPoint1Offset(ADC, offset_vinp0);
337  offset_vinp0 = 0;
338  }
339  }
340  else {
341  LL_ADC_SetCalibPoint1Gain(ADC, LL_ADC_DEFAULT_RANGE_VALUE_3V6);
342  }
343  LL_ADC_SetCalibPointForSinglePos3V6(ADC, LL_ADC_CALIB_POINT_1);
344 
345  /* Set the input channe2 */

394 void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc)
395 {
396  HAL_ADC_Stop_DMA(&adc_handle);
397 
398  uint8_t j =0;
399  float vpin = 0.0, vbattery = 0.0;
400  float temp = 0.0;
401 
402  for (int i=0; i<ADC_DMA_BUF_LEN/3; i++) {
403  vpin += LL_ADC_GetADCConvertedValueSingle(ADC, ADC_DMA_buffer[j], LL_ADC_VIN_RANGE_3V6, USER_DATAWIDTH, offset_vinp0);
404  vbattery += LL_ADC_GetADCConvertedValueBatt(ADC, ADC_DMA_buffer[j+1], USER_DATAWIDTH, 0);
405  temp += LL_ADC_GetADCConvertedValueTemp(ADC, ADC_DMA_buffer[j+2], USER_DATAWIDTH)/100.0;
406  j+=3;
407  }
408  vpin /= ADC_DMA_BUF_LEN/3;
409  vbattery /= ADC_DMA_BUF_LEN/3;
410  temp /= ADC_DMA_BUF_LEN/3;
411  printf("%d.%03d mV\t%d.%03d mV\t%d.%02d %cC\r\n", PRINT_INT(vpin),PRINT_FLOAT(vpin), PRINT_INT(vbattery),PRINT_FLOAT(vbattery), PRINT_INT(temp),PRINT_FLOAT(temp), 248);
412 
413 #if defined(CONFIG_DEVICE_BLUENRG_LP)
414  /* Enable the temperature sensor */
415  LL_PWR_EnableTempSens();
416 #endif
417 
418  /* Restart ADC DMA conversion */
419  if (HAL_ADC_Start_DMA(&adc_handle, (uint32_t *)ADC_DMA_buffer, ADC_DMA_BUF_LEN) != HAL_OK) {
420  Error_Handler();
421  }
