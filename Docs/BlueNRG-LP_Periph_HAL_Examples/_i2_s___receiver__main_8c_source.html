296 static void MX_I2S_Init(void)
297 {
298  /* Disable I2S peripheral to allow access to I2S internal registers */
299  __HAL_I2S_DISABLE(&hi2s);
300 
301  /* Configure I2S */
302  hi2s.Instance = I2S_INSTANCE;
303  hi2s.Init.AudioFreq = I2S_AUDIOFREQ_32K; // sampling frequency
304  hi2s.Init.CPOL = I2S_CPOL_LOW;
305  hi2s.Init.DataFormat = I2S_DATAFORMAT_16B;
306  // uncomment to use MCK - MCK enbable
307 // hi2s.Init.MCLKOutput = I2S_MCLKOUTPUT_ENABLE;
308  // comment to use MCK - MCK enbable
309  hi2s.Init.MCLKOutput = I2S_MCLKOUTPUT_DISABLE;
310  hi2s.Init.Mode = I2S_MODE_MASTER_RX;
311  hi2s.Init.Standard = I2S_STANDARD_PHILIPS;
312 
313  I2S_CLK_ENABLE();
314 
315  /* Configure SPIxI2S clock source */
316  I2S_CLK_CONFIG();
317 
318  /* Initialization I2S function */
319  if(HAL_I2S_Init(&hi2s) != HAL_OK)
320  {
321  printf("Initializes the I2S ended with error.\n\r");
322  Error_Handler();
323  }

332 static void EXTI10_IRQHandler_Config(void)

338  /* Enable GPIOC clock */
339  __HAL_RCC_GPIOA_CLK_ENABLE();
340  __HAL_RCC_SYSCFG_CLK_ENABLE();
341 
342  /* Configure PA.10 pin as input floating */
343  GPIO_InitStructure.Mode = GPIO_MODE_INPUT;
344  GPIO_InitStructure.Pull = GPIO_NOPULL;
345  GPIO_InitStructure.Pin = GPIO_PIN_10;
346  HAL_GPIO_Init(GPIOA, &GPIO_InitStructure);
347 
348  EXTI_Config_InitStructure.Line = EXTI_LINE_PA10;
349  EXTI_Config_InitStructure.Trigger = EXTI_TRIGGER_RISING_EDGE;
350  EXTI_Config_InitStructure.Type = EXTI_TYPE_EDGE;
351 
352  HAL_EXTI_SetConfigLine(&HEXTI_InitStructure, &EXTI_Config_InitStructure);
353  HAL_EXTI_RegisterCallback(&HEXTI_InitStructure, HAL_EXTI_COMMON_CB_ID, Example_EXTI_Callback);
354  HAL_EXTI_Cmd(&HEXTI_InitStructure , ENABLE);
355 
356  HAL_EXTI_ClearPending(&HEXTI_InitStructure);
357 
358  /* Enable and set line 10 Interrupt to the lowest priority */
359  HAL_NVIC_SetPriority(GPIOA_IRQn, IRQ_HIGH_PRIORITY);
360  HAL_NVIC_EnableIRQ(GPIOA_IRQn);
