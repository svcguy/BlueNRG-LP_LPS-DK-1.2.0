327 void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
328 {
329  if (htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1)
330  {
331  if(uhCaptureIndex == 0)
332  {
333  /* Get the 1st Input Capture value */
334  uwIC2Value1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
335  uhCaptureIndex = 1;
336  }
337  else if(uhCaptureIndex == 1)
338  {
339  /* Get the 2nd Input Capture value */
340  uwIC2Value2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
341 
342  /* Capture computation */
343  if (uwIC2Value2 > uwIC2Value1)
344  {
345  uwDiffCapture = (uwIC2Value2 - uwIC2Value1);
346  }
347  else if (uwIC2Value2 < uwIC2Value1)
348  {
349  /* 0xFFFF is max TIMx_CCRx value */
350  uwDiffCapture = ((0xFFFF - uwIC2Value1) + uwIC2Value2) + 1;
351  }
352  else
353  {
354  /* If capture values are equal, we have reached the limit of frequency
355  measures */
356  Error_Handler();
357  }
358 
359  /* Frequency computation: for this example TIMx (TIMx) is clocked by
360  64 MHz Clk */
361  uwFrequency = HAL_TIM_GetPeriphClock(htimx.Instance) / uwDiffCapture;
362  uhCaptureIndex = 0;
363  }
364  }
365 }
